2     ************************************************************************
3           REAL*8 FUNCTION ruban(xn,yn,n,x1,y1,d)
4     ************************************************************************
5           IMPLICIT NONE
6     
7     *..  Cette fonction retourne la distance entre le debut de la courbe et
8     *  le point x1,y1.
9     
10     *  arguments
11           INTEGER n
12           REAL*8 x1,y1,xn(n),yn(n),d
13     
14     *  variables locales
15           INTEGER ind
16           REAL*8 dist
17     
18     *  procedures
19           INTEGER indcrb
20           REAL*8 long
21           INTRINSIC SQRT
22           EXTERNAL indcrb, long
23     
24     c=========================
25     c.. xn,yn: tableaux des coordonees des points de la courbe.
26     c.. n  : nombre de points de la courbe.
27     c.. x1,y1: coordonnees du point.
28     c=========================
29     
30           ruban = 0.
31     
32           ind = indcrb(xn,yn,n,x1,y1,d)
33     
34           dist = long(xn,yn,ind)
35     
36           ruban = dist + SQRT((x1-xn(ind))**2 + (y1-yn(ind))**2)
37     
38           RETURN
39           END
