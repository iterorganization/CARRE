1           PROGRAM CARRE
2     c======================================================================
3     c
4     c  version : 1.01.99 15:49
5     c
6     c======================================================================
7     cank -- The comments are translated from French, sorry for errors!
8     
9     *..  Ce programme permet de trouver la localisation des points X
10     *  pour un tokamak. Il parametrise ensuite les separatrices qui passent
11     *  par ces points X et il les trace ainsi que les structures reelles.
12     *  Une maille curviligne orthogonale alignee sur les surfaces de flux
13     *  est alors construite.
14     
15     c*** This program builds an orthogonal curvilinear mesh aligned to the
16     c*** flux surfaces.  It finds the localisation of the X-points,
17     c*** parametrises the separatrices passing through them, and traces
18     c*** them together with the real structures.
19     c======================================================================
20           IMPLICIT NONE
21     
22     c  a modifier pour generaliser la geometrie:
23     c  modifier argsep et maille pour utiliser un critere plus general que
24     c  la droite ou la gauche dans la designation des separatrices.
25     c  Aussi, distribuer les points de facon proportionelle, et les ortho-
26     c  gonaliser par relaxation, avec une protection de proximite.
27     
28     c  a faire:
29     c        1. permettre l'imposition de points fixes ou de coins
30     
31     
32     *  variables.
33     
34     cank-970702: moved dimensions to the CARREDIM.F
35     #include <CARREDIM.F>
36           REAL*8 cstlin,zero,eps_Xpt,rmax,zmax
37           common /geom/ eps_Xpt
38           PARAMETER (cstlin=0.00, zero=0.)
39           INTEGER i, j, nx, ny, nstruc, npstru(strumx),
40          .   ii(gradmx), jj(gradmx), iptx(npxmx), jptx(npxmx),
41          .   npx, npxtot,ptsep(4,npxmx),
42          .   nptot(4,npxmx), indplq(4,npxmx),inddef(nbdmx),
43          .   nbdef, nivtot(nivmx), nbniv,ptxint,nreg,
44          .   np1(nregmx), npr(nregmx), ierror,iflag,limcfg,
45          .   itmp
46     
47           REAL*8 x(nxmax), y(nymax), psi(nxmax,nymax),
48          .   psidx(nxmax,nymax), psidy(nxmax,nymax),
49          .   xstruc(npstmx,strumx), ystruc(npstmx,strumx),
50          .   pointx(gradmx), pointy(gradmx),
51          .   ptx(npxmx), pty(npxmx), xpto, ypto, fctpx(npxmx),
52          .   separx(npnimx,4,npxmx), separy(npnimx,4,npxmx),
53          .   nivx(npnimx,nivmx),nivy(npnimx,nivmx),distnv(5,nivmx),
54          .   xn(npnimx),yn(npnimx),xmail(npmamx,nrmamx,nregmx),
55          .   ymail(npmamx,nrmamx,nregmx),a00(nxmax,nymax,3),
56          .   a10(nxmax,nymax,3),a01(nxmax,nymax,3),a11(nxmax,nymax,3)
57     
58           LOGICAL racord
59           character lign80*80,nomstr(strumx)*80
60     
61           REAL*8 stp0, stpmin
62           PARAMETER (stp0=0.01,stpmin=0.001)
63     
64     *..Procedures
65     *
66           INTRINSIC ABS,MIN,index
67           EXTERNAL pltini,pltend,cadre,motifs,DERIVE,cntour,GRAD0,
68          .   SELPTX,SPTRIS,ARGSEP,FRTIER,MAILLE,trace,entete
69          ,        ,trc_stk_in,trc_stk_out
70     c======================================================================
71     c.. nxmax,nymax: maximum number of the data points in x and y
72     c.. gradmx: maximum number of points where the gradient vanishes
73     c.. npxmx : maximum number of the X-points
74     c.. nbdmx : maximum number of structures being the divertor targets
75     c.. strumx: maximum number of structures
76     c.. npstmx: maximum number of points per structure
77     c.. npnimx: maximum number of tracing points on a curve
78     c.. nivmx : maximum number of limiting level lines
79     c.. npmamx: maximum number of grid points in poloidal direction
80     c.. nrmamx: maximum number of grid points in radial direction
81     c.. nregmx: maximum number of regions
82     c.. cstlin: a linear constant added along y to artificially disconnect
83     c           the X-points. Set to 0 for connected double-nulls
84     c
85     c.. nx,ny : number of data points in x and y
86     c.. x,y   : tables of coordinates of the data points
87     c.. psi   : psi values at each data point
88     c.. psidx,psidy: values of psi derivatives in x and y
89     c                at each data point
90     c.. nstruc: number of structures
91     c.. npstru: number of points per structure
92     c.. xstruc,ystruc: coordinates of the structure points
93     c                  (point index, structure index)
94     c.. npxtot: number of the points where the gradient vanishes
95     c.. pointx,pointy: coordinates of the points where the gradient
96     c                  vanishes
97     c.. ii,jj : x and y indices of the cells where the gradient vanishes
98     c.. npx   : number of the X-points
99     c.. ptx,pty: X-point co-ordinates
100     c.. iptx,jptx: x and y indices of the cells containing the X-points
101     c.. xpto,ypto: coordinates of the O-point
102     c.. racord: determines whether the X-points are connected
103     c.. fctpx: the psi values at each X- or O-point
104     c.. separx,separy: coordinates of the points of the parametrised
105     c               separatrices (point index, branch index, X-point index)
106     c.. nptot : number of parametrisation points for each separatrix
107     c           (separatrix index, point index)
108     c.. ptsep : separatrix pointer, used as index
109     c           (separatrix index, point index)
110     c.. ptxint: index of the internal X-point in the case of disconnected
111     c           double-null
112     c.. nbdef : number of the divertor plates
113     c.. inddef: table of indices of the divertor plates
114     c.. indplq: table of the structure indices (0 means not a target)
115     c           (separatrix index, X-point index)
116     c.. nbniv : number of the limiting level lines
117     c.. nivx,nivy: coordinates of the points of the parametrised
118     c              limiting level lines (point index, curve index)
119     c.. nivtot: number of points for each parametrised limiting level line
120     c.. distnv: distance along a plate between the separatrix strike-point
121     c           and a limiting level line
122     c           (distance selector [1=real, 2=psi], curve index)
123     c.. nreg  : number of the grid regions depending on the configuration
124     c.. np1   : numbers of the grid points in poloidal direction
125     c           (region index)
126     c.. npr   : numbers of the grid points in radial direction
127     c           (region index)
128     c.. xmail,ymail: grid point coordinates
129     c                (poloidal, radial, region)
130     c.. xn,yn : working array for coordinates along a parametrised curve
131     c.. nrelax: maximum number of iterations in the relaxation procedure
132     c           of construction of the orthogonal grid
133     c.. relax : relaxation parameter
134     c.. stpmin: minimum tolerable distance between any two grid points
135     c           in the course of relaxation
136     c.. rlcept: convergence criterion in the relaxation procedure
137     c.. limcfg: indicates when a limiter configuration is considered (when
138     c           non zero). This variable is assigned the index of the
139     c           limiter
140     c     N.B.: When the limiter configuration is selected, npx is set equal
141     c           to 1 and the coordinates of the X-point correspond to the
142     c           innermost point of the limiter.
143     c======================================================================
144     *
145     *..1.0  Initialisation des variables par defaut et de la bibliotheque
146     *       graphique
147     *
148           call defaut
149           CALL pltini
150           CALL cadre
151           CALL motifs
152     *
153     *..2.0  Open the data files
154     *
155           OPEN(UNIT=7, FILE='rzpsi.dat', STATUS='old')
156           OPEN(UNIT=8, FILE='structure.dat', STATUS='old')
157           OPEN(UNIT=9, FILE='carre.dat', STATUS='unknown')
158           OPEN(UNIT=10,FILE='carre.out',STATUS='unknown')
159           open(unit=11,status='scratch')
160     c
161     c  2.1  specify output format
162           write(10,*)'output format: carre70'
163     
164     *
165     *..3.0  Read the data
166     *
167     
168     *..Read the values of x
169     
170     100   format(a)
171           iflag=-1
172           call entete(7,'$r',iflag)
173           read(7,100)lign80
174           i=index(lign80,'=')
175           call rdfrin(11,lign80(i+1:80),nx,ierror)
176           READ(7,*) (x(i), i=1, nx)
177     
178     *..Read the values of y
179     
180           call entete(7,'$z',iflag)
181           read(7,100)lign80
182           i=index(lign80,'=')
183           call rdfrin(11,lign80(i+1:80),ny,ierror)
184           READ(7,*) (y(i), i=1, ny)
185     
186           rmax = 0.0
187           do i = 1, nx-1
188             rmax = max(rmax, (x(i+1)-x(i))**2)
189           enddo
190           zmax = 0.0
191           do i = 1, ny-1
192             zmax = max(zmax, (y(i+1)-y(i))**2)
193           enddo
194           eps_Xpt = sqrt(rmax+zmax)
195     
196     *..Read the values of psi
197     
198           call entete(7,'$psi',iflag)
199           READ(7,*) ((psi(i,j), i=1, nx), j=1, ny)
200     
201           DO 5 j=1,ny
202              DO 4 i=1,nx
203                 psi(i,j) = psi(i,j) + MIN(cstlin*(y(j)-y(ny/2)),zero)
204         4    CONTINUE
205         5 CONTINUE
206     c---
207     c  ceci sert a modifier la symetrie haut-bas de psi, de facon ad hoc.
208     c     write(6,*)'facteur de symetrie haut-bas'
209     c     read(5,*)a00(1,1,1)
210     c     do j=1,ny/2
211     c     do i=1,nx
212     c       xpto=psi(i,j)
213     c       ypto=psi(i,ny-j+1)
214     c       psi(i,j)=0.5*((1.+a00(1,1,1))*xpto+(1.-a00(1,1,1))*ypto)
215     c       psi(i,ny-j+1)=0.5*((1.+a00(1,1,1))*ypto+(1.-a00(1,1,1))*xpto)
216     c     enddo
217     c     enddo
218     c---
219           close(unit=7)
220     
221     *
222     *..3.1  Read the structures.
223     *
224           nstruc=0
225           call listru(8,nstruc,npstru,nomstr,xstruc,ystruc,npstmx,strumx)
226     
227     *
228     *..4.0  Calculate the first partial derivatives in x and y and store
229     *       them in arrays psidx and psidy
230     
231           CALL DERIVE(nx,ny,x,y,psi,psidx,psidy)
232     
233     *
234     *..5.0  Plot the level lines for psidx=0 and psidy=0
235     *
236           CALL cntour(psidx,psidy,nx,ny,x(1),x(nx),y(1),y(ny))
237     c
238     c  interpolation coefficients for psi and its derivatives
239     
240           call inipsi(psi,psidx,psidy,x,y,nxmax,nymax,nx,ny,a00,a10,a01,a11)
241     
242     *
243     *..6.0  Determine the points where the derivatives in x and y vanish
244     *
245     
246           CALL GRAD0(nxmax,nymax,nx,ny,x,y,gradmx,pointx,
247          .          pointy,ii,jj,npxtot,a00,a10,a01,a11)
248     
249     *
250     *..7.0  Select the X-points of interest
251     *
252     
253     cank-970702: moved dimensions to the included file
254           CALL SELPTX(npxtot,npx,pointx,pointy,ii,jj,ptx,
255          .            pty,iptx,jptx,xpto,ypto,racord,limcfg)
256     
257     *
258     *..8.0  Parametrise the separatrices
259     *
260           IF (npx.GT.0 .and. limcfg.eq.0) THEN
261     c
262             CALL SPTRIS(nx,ny,x,y,psi,npx,ptx,pty,
263          .      iptx,jptx,fctpx,separx,separy,nptot,
264          .      nstruc,npstru,xstruc,ystruc,indplq,inddef,nbdef,
265          .      a00,a10,a01,a11)
266     *
267     *..9.0  Arrange the separatrices
268     *
269     
270             CALL ARGSEP(npx,ptx,pty,fctpx,separx,separy,indplq,nptot,npnimx,
271          .            ptsep,racord,ptxint,ypto,nbdef,inddef)
272     
273           ELSEIF(LIMCFG.NE.0) THEN
274     *
275     *  13.   Identify the limiter
276     *
277             call limfnd(xpto,ypto,nivx,nivy,stp0,stpmin,distnv,nivtot,
278          .      nbniv,nx,ny,x,y,psi,npx,ptx,pty,fctpx,
279          .      nstruc,npstru,xstruc,ystruc,indplq,inddef,nbdef,
280          .      a00,a10,a01,a11)
281     
282     	do itmp=1,4
283     	  ptsep(itmp,1) = 0
284     	enddo
285     
286           ENDIF
287     
288           if(npx.gt.0) then
289     *
290     *..10.0  Find the level lines in more detail
291     *
292     c<<<
293     	write(0,*) '=== carre *..10.0 - before frtier'
294     	write(0,'(5h ptx:,1p,8e12.4/(5x,8e12.4))') (ptx(i),i=1,npx)
295     	write(0,'(5h pty:,1p,8e12.4/(5x,8e12.4))') (pty(i),i=1,npx)
296             if(limcfg.eq.0) then
297     	  write(0,*) 'nptot(4,nxpoints)'
298     	  write(0,'(1x,16i5)') ((nptot(i,j),i=1,4),j=1,npx)
299     	  write(0,*) 'Strike points (presumably)'
300     	  write(0,'(3h x:,1p,8e12.4/(3x,8e12.4))')
301          -	   ((separx(nptot(i,j),i,j),i=1,4),j=1,npx)
302     	  write(0,'(3h y:,1p,8e12.4/(3x,8e12.4))')
303          -	   ((separy(nptot(i,j),i,j),i=1,4),j=1,npx)
304     c>>>
305               call trc_stk_in('carre','*..10.0')
306               CALL FRTIER(nx,ny,x,y,psi,nstruc,
307          .      npstru,xstruc,ystruc,inddef,nbdef,npx,separx,
308          .      separy,nptot,ptsep,racord,nivx,nivy,nivtot,
309          .      nbniv,stp0,stpmin,
310          .      distnv,ptxint,a00,a10,a01,a11)
311               call trc_stk_out
312             endif
313     
314     *
315     *..11.0  Grid the regions
316     *
317     
318             CALL MAILLE(nx,ny,x,y,psi,npx,xpto,ypto,racord,
319          .    separx,separy,ptsep,nptot,distnv,ptxint,nstruc,npstru,
320          .    xstruc,ystruc,inddef,nreg,xn,yn,xmail,ymail,
321          .    np1,npr,ptx,pty,nivx,nivy,nivtot,nbniv,
322          .    a00,a10,a01,a11,fctpx,limcfg)
323     
324     c*
325     c* WARNINGS CALCULATION AND OUTPUT
326     c*
327     c        if (npx.EQ.1) CALL WARNINGS(separx,separy,nivx,nivy,
328     c     &  distnv(1,2),xpto,ypto)
329     
330     *
331     *..12.0  Plot the structures, the separatrices, and the limiting level
332     *        lines for each region, together with the resulting grid
333     *
334     
335             call trace(x(1),x(nx),y(1),y(ny),separx,separy,ptsep,npx,nptot,
336          .           nstruc,npstru,xstruc,ystruc,
337          .           nivx,nivy,nivtot,nbniv,np1,npr,xmail,ymail,nreg)
338     
339     
340           ENDIF
341     
342     *
343     *.. Close the graphics
344     *
345           CALL pltend
346     
347           STOP
348           END
