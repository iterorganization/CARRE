2     ************************************************************************
3           REAL*8 FUNCTION aazero(fcn,x0,dx0,dxmin,fmin,xmin,xmax,itmax)
4     ************************************************************************
5           IMPLICIT NONE
6     c
7     c  Purpose: find the zero of a REAL function fcn by the secant method.
8     c
9     c  Prerequisites: none
10     c
11     c  References: none
12     c
13     c  Revision    author              comment
14     c  15/11/91     R. Marchand         initial installation
15     c
16     c  Input/output variables:
17           INTEGER itmax
18           REAL*8 x0,dx0,dxmin,fmin,xmin,xmax
19     c  x0: initial guess
20     c  dx0: initial step
21     c  dxmin: minimum acceptable step: aazero is accepted if dx.le.dxmin
22     c  fmin: minimum acceptable function value: aazero is accepted if
23     c        f.le.fmin*f1
24     c  xmin: lower bound of the x interval in which to find the root
25     c  xmax: upper bound of the x interval in which to find the root
26     c        N.B.: if xmin=xmax, no bounds are assumed.
27     c  itmax: maximum number of iterations allowable
28     c
29     c  Common variables: none
30     c
31     c  Local variables:
32           INTEGER it
33           REAL*8 dxmax,x1,f1,x2,f2,zfmin,dx,dxinv,one,xl,xr,sgl,sgr,sg
34           save one
35     c
36     c  Procedures:
37           REAL*8 fcn
38           external fcn
39           intrinsic abs,sign,min,max
40     c
41     c  Data:
42           data one/1./
43     c
44     c  Computation:
45     c
46           dxmax=100.*abs(dx0)
47           if(xmin.lt.xmax) then
48             x1=max(xmin,min(xmax,x0))
49           else
50             x1=x0
51           endif
52           xl=x1
53           xr=x1
54     c
55           f1=fcn(x1)
56           if(f1.eq.0.) then
57             x2=x0
58             go to 20
59           endif
60           zfmin=abs(fmin*f1)
61           sgl=sign(one,f1)
62           sgr=sgl
63     c
64     c     if(x1-xmin.gt.xmax-x1) then
65     c       dx=abs(dx0)
66     c     else
67     c       dx=-abs(dx0)
68     c     endif
69           if(xmin.lt.xmax) then
70             if(x1.eq.xmax) then
71               dx=abs(dx0)
72             elseif(x1.eq.xmin) then
73               dx=-abs(dx0)
74             else
75               dx=-dx0
76             endif
77           else
78             dx=-dx0
79           endif
80     c
81           do 10 it=1,itmax
82           x2=x1-dx
83           if(xmin.lt.xmax) then
84             if(sgl.ne.sgr) then
85               x2=max(xl,min(xr,x2))
86             else
87               x2=max(xmin,min(xmax,x2))
88             endif
89             if((abs(x2-x1).le.dxmin .and. it.gt.2) .or. x1.eq.x2) go to 30
90           endif
91           f2=fcn(x2)
92           if(abs(f2).le.zfmin) go to 20
93           sg=sign(one,f2)
94           if(sgl.eq.sgr) then
95             xl=min(xl,x2)
96             xr=max(xr,x2)
97             if(sg.ne.sgl) then
98               if(x2.lt.xr) then
99                 sgl=sg
100                 xl=x2
101               else
102                 sgr=sg
103                 xr=x2
104               endif
105             endif
106           endif
107           if(sgl.ne.sgr) then
108             if(sg.eq.sgl) then
109               xl=x2
110               sgl=sg
111             else
112               xr=x2
113               sgr=sg
114             endif
115           endif
116     c
117           dxinv=(f2-f1)/(f2*(x2-x1))
118           if(abs(dxinv)*dxmax.le.1.) then
119             dx=sign(dxmax,dxinv)
120           else
121             dx=1./dxinv
122             if(abs(dxinv)*dxmin.ge.1. .and. it.gt.1) go to 30
123           endif
124     c
125           if(sgl.ne.sgr) then
126             if(sg.ne.sgl) then
127     c         shoot to the left
128               if(dx.gt.0.) then
129                 dx=min(dx,0.8*(x2-xl))
130               else
131                 dx=0.5*(x2-xl)
132               endif
133             else
134     c         shoot to the right
135               if(dx.lt.0.) then
136                 dx=max(dx,0.8*(x2-xr))
137               else
138                 dx=0.5*(x2-xr)
139               endif
140             endif
141           endif
142     c
143           x1=x2
144           f1=f2
145     10    continue
146     c
147     c  no convergence
148           WRITE(6,*)' warning: no convergence in aazero'
149           aazero=x0
150           RETURN
151     c
152     c  good convergence, the function is sufficiently small
153     20    continue
154           aazero=x2
155           RETURN
156     c
157     c  good convergence, step size is sufficiently small
158     30    continue
159           aazero=x2-dx
160           if(xmin.lt.xmax) then
161             aazero=max(xmin,min(xmax,aazero))
162           endif
163           RETURN
164           end
