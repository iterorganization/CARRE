1     ************************************************************************
2           INTEGER FUNCTION ifind(x,t,n,inc)
3     ************************************************************************
4           IMPLICIT NONE
5     
6     *..  Cette fonction retourne l'indice i tel que t(i)<=x<t(i+1) sur un
7     *  tableau t d'indice 1, ..., n par bonds de inc.
8     *  On suppose ici que mod(n-1,inc)=0.
9     
10     *  arguments
11           INTEGER n,inc
12           REAL*8 x, t(n)
13     
14     *  variables locales
15           INTEGER j, jp
16     
17     c=========================
18     c.. x: valeur pour laquelle on veut trouver l'indice de t
19     c.. t: tableau de coordonnees sur lequel on veut interpoler
20     c.. n: nombre de points dans le tableau
21     c.. inc: increment
22     c=========================
23     
24           IF (x.LE.t(1)) THEN
25              ifind=1
26              RETURN
27     
28           ELSE IF (x.GE.t(1+n-inc)) THEN
29              ifind=n-(2*inc)+1
30              RETURN
31     
32           ELSE
33              ifind=0
34              j=n/inc+1
35     
36         2    IF (j-ifind.EQ.1) THEN
37                 ifind=1+inc*(ifind-1)
38                 RETURN
39     
40              ENDIF
41     
42              jp=(j+ifind)/2
43              IF (x.GT.t(1+inc*(jp-1))) THEN
44                 ifind=jp
45     
46              ELSE
47                 j=jp
48     
49              ENDIF
50     
51              GO TO 2
52           ENDIF
53     
54           END
