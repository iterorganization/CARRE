1           SUBROUTINE MARCHE(x0,y0,plaque,fraplq,sens,nivox,nivoy,nvotot,
2          .        nbcrb,xt,yt,nt,stp0,stpmin,nx,ny,x,y,psi,
3          .        nstruc,npstru,xstruc,ystruc,a00,a10,
4          .        a01,a11,indlim)
5     c
6     c  version : 07.07.97 20:42
7     c
8     c======================================================================
9     *..  Cette sous-routine parametrise des lignes de niveau jusqu'a
10     *  ce qu'on obtienne la derniere courbe de niveau qui va d'un deflectur
11     *  a un autre sans rencontrer de structure.
12     *
13     cdps - Translation:
14     * This subroutine parameterises the level lines until one obtains a
15     * final level line which goes from one target to another without 
16     * hitting a structure.
17     c======================================================================
18     
19           IMPLICIT NONE
20     
21     cank-970707: dimensions from the file
22     *  dimensions
23     #include <CARREDIM.F>
24     
25     *  arguments
26           INTEGER nx,ny,nstruc,npstru(strumx),indlim,
27          .        nvotot,plaque,fraplq,sens,nbcrb,nt(2)
28     
29           REAL*8 x(nxmax),y(nymax),psi(nxmax,nymax),xstruc(npstmx,strumx),
30          .     ystruc(npstmx,strumx),x0,y0,stp0,stpmin,nivox(npnimx),
31          .     nivoy(npnimx),a00(nxmax,nymax,3),a10(nxmax,nymax,3),
32          .     a01(nxmax,nymax,3),a11(nxmax,nymax,3),
33          .     xt(npnimx,2),yt(npnimx,2)
34     
35     *  variables locales
36           INTEGER ii,jj,dir,k,j,repart,ind1,ind2,ind3,indstr
37           REAL*8 const1
38           PARAMETER (const1=1.E-08)
39           REAL*8 pas,x1,y1,x2,y2,psi1,psi2,pastmp,ecart1,ecart2,
40          .       niv1x(npnimx),niv1y(npnimx)
41           LOGICAL echec,coin,ouvert
42     
43     *  procedures
44           INTEGER ifind,indsgm
45           LOGICAL in, chgdir, cross
46           INTRINSIC MOD,SQRT
47           EXTERNAL chgdir,CRBNIV,cross,ifind,in,indsgm
48     
49     c=========================
50     c.. plaque: indice de la structure (plaque) sur laquelle on marche.
51     c.. fraplq: indice de la structure (plaque) que la ligne de niveau doit
52     c           frapper.
53     c.. sens: sens de deplacement sur la plaque
54     c.. x0,y0: point de depart.
55     c.. nivox,nivoy: tableaux des coordonnees x et y des points parametrises
56     c                de la derniere ligne de niveau.
57     c.. nvotot: nombre de points de cette courbe.
58     c.. knivtm: nombre de points de cette courbe.
59     c.. x2,y2: point sur la structure ou debute la nouvelle ligne de niveau.
60     c.. x1,y1: point de la ligne precedente.
61     c.. psi1,ps2: valeur de la fonction aux points 1 et 2.
62     c.. ind1,ind2: indices de segments des points 1 et 2.
63     c.. ind3: indice d'un point de structure, coin.
64     c.. pas: valeur du pas auquel on marche.
65     c.. pastmp: valeur temporaire du pas.
66     c.. ii,jj: indice d'identification d'un carre.
67     c.. dir: direction de recherche.
68     c.. k  : indice du point parametrise de chaque courbe.
69     c.. indlim: indice de la derniere structure frappee autre que la
70     c           structure designee fraplq.
71     c.. indstr: indice de la derniere structure frappee.
72     c.. repart: variable necessaire pour signifier a "SAUTE" que l'on
73     c           travaille en distance absolue.
74     c.. ecart1,ecart2: distance entre un point et chaque extremitee du
75     c                  segment sur lequel il se trouve.
76     c.. rien: logique, si oui ou non on n'a pas encore rencontre une
77     c         structure differente de fraplq.
78     c.. echec: logique, si on doit considerer le debut de la courbe comme
79     c          un echec ou non.
80     c.. coin: logique, si on est a un point de structure ou non.
81     c.. ouvert: logique=.t. si la structure sur laquelle on marche est
82     c            ouverte.
83     c
84     cdps Translation:
85     c.. plaque: index of the structure (plate) on which one goes.
86     c.. fraplq: index of the structure (plate) that the level line must strike.
87     c.. sens: direction of movement along the target - 1=the same as the
88     c           target points, -1=opposite.
89     c.. x0,y0: starting point.
90     c.. nivox,nivoy: tables of the coordinates X and y of the points parameterising
91     c                the final level line.
92     c.. nvotot: number of points on this curve. 
93     c.. knivtm: number of points on this curve. 
94     c.. x2,y2: point on the structure or at the start of the new level line.
95     c.. x1,y1: point from the preceding line.
96     c.. psi1,ps2: value of the psi function at points 1 and 2.
97     c.. ind1,ind2: indices of the segments at points 1 et 2.
98     c.. ind3: index of a point on the structure, corner.
99     c.. pas: value of the step we are going to.
100     c.. pastmp: temporary value of the step.
101     c.. ii,jj: cell index.
102     c.. dir: direction of the search.
103     c.. k  : index of the point parameterising each curve. 
104     c.. indlim: index of the final structure hit other than the designated
105     c           fraplq. 
106     c.. indstr: index of the final structure hit.
107     c.. repart: specifies whether "SAUTE" works with distance along the target
108     c           (repart=1) or the new psi value (repart=2).
109     c.. ecart1,ecart2: distance between a point and the boundary segment on 
110     c                  which it is.
111     c.. rien: logical variable indicating whether or not a structure other than
112     c         fraplq has been hit.
113     c.. echec: logical variable indicating whether or not a curve fails to
114     c          start.
115     c.. coin: logical variable indicating whether or not one has a point on 
116     c          on a structure.
117     c.. ouvert: logical variable is true if the structure being followed is
118     c           open.
119     c=========================
120     
121     *..Initialisation
122     c***
123     c     print*,'entree dans marche'
124     c     print*,'nx,ny=',nx,ny
125     c     print*,'strumx=',strumx
126     c     print*,'nbcrb, nt(nbcrb)=',nbcrb,nt(nbcrb)
127     c     do j=1,nt(nbcrb)
128     c       print*,'j, x/yt=',j,xt(j,nbcrb),yt(j,nbcrb)
129     c     enddo
130     c     pause
131     c***
132     
133           x2 = x0
134           y2 = y0
135     
136           ouvert=xstruc(1,plaque).ne.xstruc(npstru(plaque),plaque)
137          .  .or. ystruc(1,plaque).ne.ystruc(npstru(plaque),plaque)
138     
139     *..Calcul de la valeur de la fonction au premier point.
140     cdps
141     c  Calculate psi value at first point.
142     
143           ii = ifind(x2,x,nx,1)
144           jj = ifind(y2,y,ny,1)
145     c***
146     c     print*,'ii, jj=',ii,jj
147     c***
148           psi2 = a00(ii,jj,1) + a10(ii,jj,1)*x2 + a01(ii,jj,1)*y2 +
149          .       a11(ii,jj,1)*x2*y2
150     c***
151     c     print*,'psi2=',psi2
152     c     print*,'plaque=',plaque
153     c***
154     
155           ind2 = indsgm(xstruc(1,plaque),ystruc(1,plaque),npstru(plaque),
156          .              x2,y2)
157     c***
158     c     print*,'ind2=',ind2
159     c***
160     
161     *..Initialisation.
162     
163           pas = stp0
164           repart = 1
165           pastmp = pas
166     
167           DO 5 j=1, nt(nbcrb)
168              nivox(j) = xt(j,nbcrb)
169              nivoy(j) = yt(j,nbcrb)
170         5 CONTINUE
171     c***
172     c     print*,'apres la definition: nbcrb, nt()=',nbcrb,nt(nbcrb)
173     c     do j=1,nt(nbcrb)
174     c       print*,'j, x/yt=',j,xt(j,nbcrb),yt(j,nbcrb)
175     c     enddo
176     c     do j=1,nt(nbcrb)
177     c       print*,'j, nivox/y=',j,nivox(j),nivoy(j)
178     c     enddo
179     c***
180     
181           nvotot = nt(nbcrb)
182     
183        10 CONTINUE
184     
185           IF (pastmp .LT. 0.5*stpmin) RETURN
186     
187           x1 = x2
188           y1 = y2
189           psi1 = psi2
190           ind1 = ind2
191           coin = .FALSE.
192     
193     *..On recherche le prochain point.
194     cdps
195     c  Look for the next point.
196     
197           CALL SAUTE(xstruc(1,plaque),ystruc(1,plaque),npstru(plaque),
198          .           x1,y1,psi1,x2,y2,pas,sens,repart,nx,ny,x,y,
199          .           a00,a10,a01,a11,nxmax,nymax)
200     c***
201     c     print*,' '
202     c     print*,'pas=',pas
203     c     print*,'apres saute: x1,y1=',x1,y1
204     c     print*,'             x2,y2=',x2,y2
205     c     pause
206     c***
207     
208     *..On s'assure que l'on ne depasse pas la maille.
209     cdps
210     c  Make sure we did not go past the end of the mesh.
211     
212           IF ((x2 .LT. x(1)) .OR. (x2 .GT. x(nx)) .OR. (y2 .LT. y(1)) .OR.
213          .      (y2 .GT. y(ny))) THEN
214     
215              pastmp = pastmp/2.
216              pas = pastmp
217              x2 = x1
218              y2 = y1
219     
220              psi2 = psi1
221              ind2 = ind1
222              GO TO 10
223     
224           ENDIF
225     
226     *..Indice du segment de structure sur lequel se trouve x2,y2.
227     cdps
228     c  Index of the segment of the structure containing x2, y2.
229     
230           ind2 = indsgm(xstruc(1,plaque),ystruc(1,plaque),npstru(plaque),
231          .              x2,y2)
232     c***
233     c     print*,'ind1, ind2=',ind1,ind2
234     c***
235     
236           IF (ind1 .NE. ind2) THEN
237     
238              COIN = .TRUE.
239     
240              IF (sens .EQ. 1) THEN
241                 ind3 = ind2
242              ELSE
243                 ind3 = ind1
244              ENDIF
245     
246              x2 = xstruc(ind3,plaque)
247              y2 = ystruc(ind3,plaque)
248     
249              pastmp = pas
250     
251              pas = SQRT((x1-x2)**2 + (y1-y2)**2)
252     
253           ENDIF
254     
255     *..Calcul de la fonction pour le point 2.
256     cdps
257     c  Calculate the psi function at point 2.
258     
259           ii = ifind(x2,x,nx,1)
260           jj = ifind(y2,y,ny,1)
261     
262           psi2 = a00(ii,jj,1) + a10(ii,jj,1)*x2 + a01(ii,jj,1)*y2 +
263          .       a11(ii,jj,1)*x2*y2
264     c***
265     c     print*,'psi2=',psi2
266     c***
267     
268     *..Definition du premier point du vecteur de niveau.
269     cdps
270     c  Define the first point of the level vector.
271     
272           niv1x(1) = x2
273           niv1y(1) = y2
274     
275     *..Recheche du deuxieme point sur le perimetre du carre ii,jj.
276     cdps
277     c  Search for the second point on the perimeter of cross section ii,jj.
278     
279           ecart1 = SQRT((x2-xstruc(ind2,plaque))**2 +
280          .                       (y2-ystruc(ind2,plaque))**2)
281     
282           ecart2 = SQRT((x2-xstruc(ind2+1,plaque))**2 +
283          .                       (y2-ystruc(ind2+1,plaque))**2)
284     
285           echec = .FALSE.
286           dir = 0
287           k = 1
288           CALL CRBNIV(ii,jj,k,dir,nxmax,nymax,nx,ny,x,y,psi,psi2,
289          .            niv1x,niv1y,npnimx,strumx,npstmx,
290          .            nstruc,npstru,xstruc,ystruc,
291          .            indstr,xt,yt,nt,nbcrb,plaque,x2,y2)
292     c***
293     c     print*,'apres crbniv1: k, indstr=',k,indstr
294     c     do j=1,k
295     c       print*,'j, niv1x/y=',j,niv1x(j),niv1y(j)
296     c     enddo
297     c***
298     
299     *..Si on est rendu a l'extremite du segment, on verifie si ce point est
300     *  a l'interieur de la structure ou si le sens est change. Si on est pas
301     *  a l'extremite, on verifie si le point est a l'interieur ou si on a
302     *  traverse un autre segment de la structure. Pour chaque fois, si c'est
303     *  le cas, on doit inverser les directions, revenir d'un carre et on
304     *  oublie le point que l'on vient de prendre. Sinon, on continue en
305     *  conservant le point obtenu, la direction et les coordonnees du carre.
306     cdps
307     c  If one has reached the end of the segment, check to see if the point is
308     c  inside the structure or if the direction has changed.  If not the latter,
309     c  verify that the point is in the interior or if there is a cross-piece
310     c  on another segment of the structure.  If this is the case, the direction
311     c  must be reversed, to return from a cross section and forget the point
312     c  just taken.  If not, one continues, preserving the point just obtained,
313     c  the direction, and the coordinate cross section.
314     
315           IF ((ABS(x2-x(nx)).LT.const1) .OR. (ABS(y2-y(ny)).LT.const1)
316          .  .OR. (ABS(x2-x(1)).LT.const1) .OR. (ABS(y2-y(1)).LT.const1))
317          .                                                         THEN
318     
319              IF (((dir .EQ. 1) .AND. (ii .EQ. nx))
320          .       .OR. ((dir .EQ. 2) .AND. (jj .EQ. ny))
321          .       .OR. ((dir .EQ. 3) .AND. (ii .EQ. 1))
322          .       .OR. ((dir .EQ. 4) .AND. (jj .EQ. 1))) THEN
323     
324                 k = 1
325                 dir = MOD(dir+1,4) + 1
326                 ii = ii - MOD(dir-2,2)
327                 jj = jj - MOD(dir-3,2)
328     
329                 CALL CRBNIV(ii,jj,k,dir,nxmax,nymax,nx,ny,x,y,psi,psi2,
330          .             niv1x,niv1y,npnimx,strumx,npstmx,nstruc,npstru,
331          .             xstruc,ystruc,indstr,xt,yt,nt,nbcrb,plaque,x2,y2)
332                 if(indstr.ne.0 .and. indstr.ne.fraplq) indlim=indstr
333     c***
334     c     print*,'apres crbniv2: k, indstr=',k,indstr
335     c     do j=1,k
336     c       print*,'j, niv1x/y=',j,niv1x(j),niv1y(j)
337     c     enddo
338     c***
339     
340                 IF (cross(ind2,niv1x,niv1y,xstruc(1,plaque),
341          ,                         ystruc(1,plaque),npstru(plaque)))   THEN
342     
343                    echec = .TRUE.
344     c***
345     c        print*,'point 2: echec=',echec
346     c***
347     
348                 ENDIF
349     
350              ENDIF
351     
352           ELSE IF ((ecart1 .LT. const1) .OR. (ecart2 .LT. const1)) THEN
353     
354              IF (chgdir(niv1x,niv1y,nivox,nivoy))     THEN
355     
356                 k = 1
357                 dir = MOD(dir+1,4) + 1
358                 ii = ii - MOD(dir-2,2)
359                 jj = jj - MOD(dir-3,2)
360     
361              ENDIF
362     
363           ELSE
364     c***
365     c     do j=1,k
366     c       print*,'j, niv1x/y=',j,niv1x(j),niv1y(j)
367     c     enddo
368     c     do j=1,k
369     c       print*,'j, nivox/y=',j,nivox(j),nivoy(j)
370     c     enddo
371     c     print*,'chgdir=',chgdir(niv1x,niv1y,nivox,nivoy)
372     c***
373     
374              IF (chgdir(niv1x,niv1y,nivox,nivoy))     THEN
375     
376                 IF ((in(niv1x(2),niv1y(2),xstruc(1,plaque),
377          .                         ystruc(1,plaque),npstru(plaque)))
378          .         .OR. (cross(ind2,niv1x,niv1y,xstruc(1,plaque),
379          .                         ystruc(1,plaque),npstru(plaque)))
380          .         .OR. ouvert)   THEN
381     
382                    k = 1
383                    dir = MOD(dir+1,4) + 1
384                    ii = ii - MOD(dir-2,2)
385                    jj = jj - MOD(dir-3,2)
386     
387                 ELSE
388     
389                    echec = .TRUE.
390     c***
391     c              print*,'point 3: echec=',echec
392     c***
393     
394                 ENDIF
395     
396              ELSE
397     
398                if(.not. ouvert) then
399                 IF ((in(niv1x(2),niv1y(2),xstruc(1,plaque),ystruc
400          .                           (1,plaque),npstru(plaque)))
401          .        .OR. (cross(ind2,niv1x,niv1y,xstruc(1,plaque),
402          .                          ystruc(1,plaque),npstru(plaque)))) THEN
403     
404                    echec = .TRUE.
405     c***
406     c              print*,'point 4: echec=',echec
407     c***
408     
409                 ENDIF
410                endif
411              ENDIF
412           ENDIF
413     
414     *..Pour les points successifs, on poursuit jusqu'a ce qu'on frappe une
415     *  structure.
416     cdps
417     c  For subsequent points, continue until one strikes a structure.
418     
419           IF (.NOT.(echec)) THEN
420     
421              CALL CRBNIV(ii,jj,k,dir,nxmax,nymax,nx,ny,x,y,psi,psi2,
422          .          niv1x,niv1y,npnimx,strumx,npstmx,nstruc,npstru,xstruc,
423          .          ystruc,indstr,xt,yt,nt,nbcrb,plaque,x2,y2)
424              if(indstr.ne.0 .and. indstr.ne.fraplq) indlim=indstr
425     c***
426     c     print*,'apres crbniv3: k, indstr=',k,indstr
427     c     do j=1,k
428     c       print*,'j, niv1x/y=',j,niv1x(j),niv1y(j)
429     c     enddo
430     c***
431     
432           ENDIF
433     
434     *..Si on a frappe une plaque, on verifie si on est rendu a l'extremite
435     *  de notre plaque de depart, sinon on avance d'un pas et on recommence.
436     cdps
437     c  If we do strike a plate, check to see if we have returned to the end of
438     c  our starting plate.  If not, advance one step and start over.
439     
440     c***
441     c     print*,'indstr, fraplq=',indstr,fraplq
442     c     print*,'k=',k
443     c***
444           IF ((indstr .EQ. fraplq) .AND. (.NOT.(echec))) THEN
445     
446             DO 15 j=1, k
447                nivox(j) = niv1x(j)
448                nivoy(j) = niv1y(j)
449        15   CONTINUE
450     
451              nvotot = k
452     
453              IF (pastmp .LT. stp0) THEN
454     
455                 pastmp = pastmp/2.
456                 pas = pastmp
457     
458              ELSE
459     
460                 pas = pastmp
461     
462              ENDIF
463     
464           ELSE
465     
466              IF (coin) THEN
467                 pastmp = pas
468              ENDIF
469     
470              pastmp = pastmp/2.
471              pas = pastmp
472     
473              x2 = x1
474              y2 = y1
475              psi2 = psi1
476              ind2 = ind1
477     
478           ENDIF
479     
480           GO TO 10
481     
482           END
