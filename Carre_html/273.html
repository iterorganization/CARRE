1           integer function coinci(vx,vy,n,rx,ry)
2     
3     c  version : 20.04.2001 23:04
4     
5     c=======================================================================
6     c*** This function checks the points along a curve to find the point
7     c*** which coincides - within the tolerance parameter - with the
8     c*** specified reference point. 
9     c*** If such a point is not found, then the closest point is marked after
10     c*** the confirmation from user.
11     
12     c*** Input:
13     c***  vx, vy  coordinates of points along the curve
14     c***  n       number of these points
15     c***  rx, ry  coordinates of the reference points
16     
17     c*** Internel parameters:
18     c***  const   the tolerance
19     
20     c*** Output:
21     c***  The function value gives the index of the point found
22     
23           implicit none
24     
25     *  arguments
26           integer n
27           real*8 vx(n), vy(n), rx, ry
28     
29     *  variables locales
30           integer i, imn, irp
31           real*8 const, rmn, tol, r
32           parameter (const=0.001, tol=const*const)
33           character c
34     
35     *  procedures
36           intrinsic abs
37           
38           data irp /0/
39     c=======================================================================
40     
41           rmn=1.e16
42           imn=0
43           do i=1, n
44             r=(vx(i)-rx)**2+(vy(i)-ry)**2
45             if (r.le.tol) then
46               coinci = i
47               return
48             else if(r.lt.rmn) then
49               rmn=r
50               imn=i
51             endif
52           end do
53           if(irp.ge.2) then
54             print *,'More than two occasions of the X-point mismatch in ',
55          ,    'coinci'
56             print *,'==> May be an internal error in Carre'
57             call pltend
58             stop
59           end if
60           irp=irp+1
61           rmn=sqrt(rmn)
62           if(irp.eq.1) then
63             print *
64             print *,'== The following message may appear twice:'
65             print *
66           end if
67           print '(1x,1p,2a,2(e10.3,a))',
68          ,  'A separatrix misses the X-point by more than ',
69          ,  'the hard-wired tolerance (',rmn,' vs.',const,' )'
70           print *,'Are you sure the X-points should be connected? (y/n)'
71      10   read '(a1)',c
72           if(c.eq.'y' .or. c.eq.'Y') then
73             coinci = imn
74             return
75           else if(c.eq.'n' .or. c.eq.'N') then
76             call pltend
77             stop
78           end if
79           print *,'Please answer with Y or N'
80           go to 10
81     c=======================================================================
82           end
