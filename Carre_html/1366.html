1           subroutine orddef(inddef,nbdef,xst,yst,ptx,pty,npx)
2     c
3     c  VERSION: 23.06.99 11:27
4     c
5     c=======================================================================
6     c*** This subroutine orders the indices in the vector inddef such that 
7     c*** the first element points to top right, second top left, third bottom
8     c*** right, fourth bottom left. If there is only one X-point, then 1 is 
9     c*** the right and 2 is the left.
10     c***  	NB: the O-point coordinates are in ptx(npx+1), pty(npx+1).
11     c=======================================================================
12           implicit none
13     *  arguments
14           integer nbdef, inddef(nbdef),npx
15           real*8 xst(*),yst(*),ptx(npx+1),pty(npx+1)
16     
17     *  variables en common
18     
19     #include <COMLAN.F>
20     
21     *  variables locales
22           integer n
23           real*8 tt,xx(2),xy(2),zero
24           parameter (n=4, zero=0.)
25           integer i, j, itt, ipx
26           real*8 xt(n), yt(n),wx(4),wy(4)
27           real*8 u_hlp
28     
29     *  procedures
30           real*8 aire
31           external aire
32     c=======================================================================
33     c.. i,j: indices
34     c.. n  : maximum number of targets considered in this routine
35     c.. xt,yt: temporary arrays containing the x and y coordinates of the 
36     c     	   first points of a divertor plate
37     c.. ptx, pty: X-point coordinates
38     c.. xx,xy: X-point coordinates ordered from top to bottom
39     c.. npx: number of X-points
40     c=======================================================================
41     c<<<
42           write(0,*) 'Entering orddef: npx,nbdef= ',npx,nbdef
43           write(0,'(9h inddef: ,20i4/(9x,20i4))') inddef
44           write(0,'(6h xst: ,1p,10e10.2/(6x,10e10.2))') (xst(i),i=1,nbdef)
45           write(0,'(6h yst: ,1p,10e10.2/(6x,10e10.2))') (yst(i),i=1,nbdef)
46           if(npx.gt.0) then !{
47           	write(0,'(6h ptx: ,1p,10e10.2/(6x,10e10.2))') (ptx(i),i=1,npx)
48           	write(0,'(6h pty: ,1p,10e10.2/(6x,10e10.2))') (pty(i),i=1,npx)
49           end if !}
50     c>>>
51     c*** Order the x-points from top to bottom if necessary
52     
53           if(npx.eq.2) then !{
54             if(pty(1).gt.pty(2)) then !{
55               xx(1)=ptx(1)
56               xy(1)=pty(1)
57               xx(2)=ptx(2)
58               xy(2)=pty(2)
59             else !}{
60               xx(1)=ptx(2)
61               xy(1)=pty(2)
62               xx(2)=ptx(1)
63               xy(2)=pty(1)
64             endif !}
65           elseif(npx.eq.1) then !}{
66               xx(1)=ptx(1)
67               xy(1)=pty(1)
68           elseif(npx.gt.2) then !}{
69             if(sellan(1:7).eq.'english') then
70               write(6,*) 'npx=',npx,' too large in orddef'
71             elseif(sellan(1:8).eq.'francais') then
72               write(6,*)'npx=',npx,' trop grand dans orddef'
73             endif
74             stop
75           endif !}
76     
77     c*** Copy inddef, xstruc and ystruc of the first point of the structure 
78     c*** on the temporary arrays
79     
80           do i=1, nbdef !{
81             xt(i) = xst(i)
82             yt(i) = yst(i)
83           end do !}
84     
85     c*** Order the structures according to y
86     
87           do i=1,nbdef-1 !{
88             do j=i+1, nbdef !{
89               if(yt(j).gt.yt(i)) then !{
90                 tt=yt(i)
91                 yt(i)=yt(j)
92                 yt(j)=tt
93                 tt=xt(i)
94                 xt(i)=xt(j)
95                 xt(j)=tt
96                 itt=inddef(i)
97                 inddef(i)=inddef(j)
98                 inddef(j)=itt
99               endif !}
100             end do !}
101           end do !}
102     c<<<
103           write(0,*) 'After ordering in y'
104           write(0,'(9h inddef: ,20i4/(9x,20i4))') inddef
105     c>>>
106     c*** Order the plates 2 by 2 in x
107     c<<<
108           write(0,'(1x,a3,a4,a6,2a9,2x,1h:,18x,a2,16x,1h:,18x,a2)') 'i',
109          ,	'ipx','xy','pty','aire','wx','wy'
110     c>>>
111           do i=1,nbdef-1,2 !{
112             ipx=(i+1)/2
113             wx(1)=xx(ipx)
114             wy(1)=xy(ipx)
115             wx(2)=xt(i)
116             wy(2)=yt(i)
117             wx(3)=xt(i+1)
118             wy(3)=yt(i+1)
119             wx(4)=wx(1)
120             wy(4)=wy(1)
121     c<<<
122     	u_hlp=aire(wx,wy,4)
123     	write(0,'(1x,2i3,1p,1x,11e9.2)') i,ipx,
124          ,	  xy(ipx),pty(npx+1),u_hlp,wx,wy
125     c>>>
126             if((xy(ipx)-pty(npx+1))*aire(wx,wy,4).lt.zero) then !{
127               itt=inddef(i)
128               inddef(i)=inddef(i+1)
129               inddef(i+1)=itt
130             endif !}
131           end do !}
132     c<<<
133           write(0,*) 'After ordering in x'
134           write(0,'(9h inddef: ,20i4/(9x,20i4))') inddef
135     c>>>
136     c=======================================================================
137           end
