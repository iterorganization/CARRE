2     ************************************************************************
3           LOGICAL FUNCTION trvers(x1,y1,x2,y2,vx,vy,n)
4     ************************************************************************
5           IMPLICIT NONE
6     
7     *..  Cette fonction nous dit si le segment de 1 a 2 traverse la courbe v
8     
9     *  arguments
10           INTEGER n
11           REAL*8 x1,y1,x2,y2,vx(n),vy(n)
12     
13     *  variables locales
14           INTEGER i
15           REAL*8 determ, mult1,mult2,zero,un
16           parameter(zero=0.,un=1.)
17     
18     c=========================
19     c.. vx,vy: tableaux des coordonees des points de la courbe.
20     c.. n  : nombre de points de la courbe.
21     c.. x1,y1,x2,y2: coordonnees des 2 points formant le segment.
22     c.. determ: determinant de la matrice des deux equations.
23     c.. mult1: facteur multiplicatif du segment.
24     c.. mult2: facteur multiplicatif du segment de la courbe v.
25     c=========================
26     
27     *..Boucle sur tous les segments de la courbe.
28     
29           DO 10 i=1, n-1
30     
31     *..Calcul du determinant de la matrice.
32     
33              determ = (-(x2 - x1)) * (vy(i+1) - vy(i)) +
34          .               (y2 - y1) * (vx(i+1) - vx(i))
35     
36     *..Si determinant non nul, alors il y a solution.
37     
38              IF (determ .NE. 0.) THEN
39     
40     *..Facteur multiplicatif du segment avec la methode de Cramer.
41     
42                 mult1 = (-(vx(i)-x1) * (vy(i+1)-vy(i)) +
43          .                 (vy(i)-y1) * (vx(i+1)-vx(i)))/determ
44     
45     *..Pour avoir intersection, il faut que mult1 soit entre 0 et 1
46     
47                 IF ((mult1.GT.zero).AND.(mult1.LT.un)) THEN
48     
49     *..Fact. mult. du segment de la courbe v.
50     
51                    mult2= ((x2-x1) * (vy(i)-y1) -
52          .                   (y2-y1) * (vx(i)-x1))/determ
53     
54     *..Intersection si mult2 entre 0 et 1
55     
56                    IF ((mult2.GT.zero).AND.(mult2.LT.un)) THEN
57     
58     *..S'il y a intersection alors le segment traverse la courbe.
59     
60                       trvers = .TRUE.
61                       RETURN
62     
63                    ENDIF
64     
65                 ENDIF
66     
67              ENDIF
68     
69        10 CONTINUE
70     
71           trvers = .FALSE.
72     
73           RETURN
74           END
