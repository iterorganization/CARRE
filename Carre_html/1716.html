2     ************************************************************************
3           LOGICAL FUNCTION tsplaq(vx,vy,k,npx,px,py)
4     ************************************************************************
5           IMPLICIT NONE
6     
7     *..  Cette fonction verifie si l'accroissement d'une courbe est monotone
8     
9     *  arguments
10           INTEGER k, npx
11           REAL*8 vx(k), vy(k), px,py
12     
13     *  variables locales
14           INTEGER i
15           REAL*8 dist1,dist2,eps_Xpt
16           common /geom/ eps_Xpt
17     cxpb      PARAMETER (eps=0.015)
18     c!!!  attention a ce facteur de 0.05. Xavier a du l'augmenter dans
19     c     certains cas.
20     
21     *  procedures
22           INTRINSIC SQRT
23     
24     c=========================
25     c.. vx,vy: tableaux des coordonees des points de la courbe.
26     c.. k  : nombre de points de la courbe.
27     c.. px,py: coordonnees du point X.
28     c.. dist1,dist2: distance entre un point de coordonnee et le point X.
29     c=========================
30           dist1 = 0.
31           DO 10 i=2, k
32              dist2 = (vx(i)-vx(1))**2 + (vy(i)-vy(1))**2
33              IF (dist2 .LT. dist1) THEN
34                 tsplaq = .FALSE.
35                 RETURN
36              ENDIF
37              IF (npx .GT. 1) THEN
38     	   IF (SQRT((vx(i)-px)**2 + (vy(i)-py)**2) .LT. eps_Xpt) THEN
39                    tsplaq = .FALSE.
40                    RETURN
41                 ENDIF
42              ENDIF
43              dist1 = dist2
44        10 CONTINUE
45           tsplaq = .TRUE.
46           RETURN
47           END
