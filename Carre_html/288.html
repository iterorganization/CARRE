1           subroutine coord(vx,vy,n,d,x,y)
2     
3     c  version : 21.07.2000 16:18
4     c=======================================================================
5           implicit none
6     
7     c*** Calculate the x and y coordinates of a point located on a curve v 
8     c*** at a distance d from the origin of the curve
9     
10     *  arguments
11           integer n
12           real*8 vx(n),vy(n),d,x,y
13     
14     *  variables locales
15           integer i
16           real*8 dist,distot,eps
17           parameter (eps=1.e-06)
18     
19     *  procedures
20           intrinsic sqrt
21     
22     c========================
23     c.. n   : number of points in the curve
24     c.. vx,vy: coordinates of the points in the curve
25     c.. d   : required distance along the curve
26     c.. x,y : coordinates of the found point
27     c.. dist: distance between 2 consecutive points on the curve
28     c.. distot: total distance covered
29     c========================
30     
31     *..Initialisation.
32     
33           distot = 0.0
34     
35     *..Loop over the curve segments
36     
37           do i=1, n-1 !{
38     
39              dist = sqrt((vx(i+1)-vx(i))**2 + (vy(i+1)-vy(i))**2)
40              distot = distot + dist
41     
42     *..Check whether the distance d is passed
43     
44              if (distot .ge. d) then !{
45                 x = vx(i+1) - ((distot-d)/dist)*(vx(i+1)-vx(i))
46                 y = vy(i+1) - ((distot-d)/dist)*(vy(i+1)-vy(i))
47     
48                 return
49     
50              end if !}
51     
52           end do  !}
53     
54     c*** Check whether the required distance is equal - within tolerance -
55     c*** to the total curve length
56     
57           if (abs(distot - d) .lt. eps) then !{
58              x = vx(n)
59              y = vy(n)
60     
61              return
62     
63           end if !}
64     
65     *..Error: required distance greater than the total curve length.
66     
67           print *, 'coord: required distance greater than the total length'
68           print*,'distot=',distot
69           print*,'d=',d
70           call pltend
71           stop  'Check the target specification'
72     
73           end
