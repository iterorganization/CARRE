1           subroutine sptris(nx,ny,x,y,psi,npx,ptx,pty,
2          .           iptx,jptx,fctpx,separx,separy,nptot,
3          .           nstruc,npstru,xstruc,ystruc,indplq,inddef,nbdef,
4          .           a00,a10,a01,a11)
5     c
6     c  version : 21.07.2000 13:38
7     c
8     c=======================================================================
9     c*** This subroutine finds first the second starting point for each 
10     c*** separatrix branch (the first point is the X-point). 
11     c*** Once such a point is found, a subroutine CRBNIV is called, which 
12     c*** traces this branch until it strikes a target or comes to an X-point
13     c*** and stores the trace co-ordinates for plotting.
14     c*** Finally, a subroutine ORDDEF is called to order the targets 
15     c*** according to the CARRE conventions, i.e. from right to left, from 
16     c*** top to bottom.
17     c=======================================================================
18     
19           implicit none
20     cank-970707: dimensions from the file
21     *  dimensions
22     #include <CARREDIM.F>
23     
24     *  arguments
25           integer nx,ny,npx,iptx(npx),jptx(npx),
26          .     nptot(4,npxmx),nstruc,npstru(strumx),
27          .     indplq(4,npxmx),inddef(nbdmx),nbdef
28           real*8 x(nxmax),y(nymax),psi(nxmax,nymax),ptx(npxmx),pty(npxmx),
29          .     fctpx(npxmx),separx(npnimx,4,npxmx),separy(npnimx,4,npxmx),
30          .     xstruc(npstmx,strumx),ystruc(npstmx,strumx),
31          .     a00(nxmax,nymax,3),a10(nxmax,nymax,3),
32          .     a01(nxmax,nymax,3),a11(nxmax,nymax,3)
33     
34     *  variables locales
35           integer ipx,ix,jx,isep,i1,i2,j1,j2,iref,jref,k,idir,
36          .  indstr,plaque,nbdfav,nt(2),ixp_hlp(nbdmx),idirloop
37           real*8 xsttmp(nbdmx),ysttmp(nbdmx)
38           integer i,j     !!!
39           integer incri(4),incrj(4),incrir(4),incrjr(4),ic
40           data incri  /0, -1,  0,  1/,  incrj  / 1, 0, -1,  0/
41           data incrir /0,  0, -1, -1/,  incrjr /-1, 0,  0, -1/
42     
43     *  procedures
44           logical milieu
45           real*8 interp
46           intrinsic int,mod
47           external crbniv,insert,interp,milieu,orddef
48     c=======================================================================
49     
50     *..Initialisation
51     c<<<
52           write(0,*) 
53           write(0,*) 'Entering sptris: npx=',npx
54     c>>>
55     
56           plaque = 0
57           nbdef = 0
58           nt(1)=0
59           nt(2)=0
60     
61     *..Loop over the x-points
62     
63           do ipx=1,npx !{
64             ix = iptx(ipx)
65             jx = jptx(ipx)
66     c<<<
67             write(0,*) 
68             write(0,*) 'ipx,ix,jx,ptx,pty =',ipx,ix,jx,ptx(ipx),pty(ipx)
69             write(0,*) 'psi(ix,jx),CCW =',psi(ix,jx),psi(ix+1,jx),
70          ,                                psi(ix+1,jx+1),psi(ix,jx+1)
71     c>>>
72     
73     *..Calculate the flux at the x-point
74     
75             fctpx(ipx) = a00(ix,jx,1) + a10(ix,jx,1)*ptx(ipx)
76          .                            + a01(ix,jx,1)*pty(ipx)
77          .                            + a11(ix,jx,1)*ptx(ipx)*pty(ipx)
78             write(0,*) 'fctpx =',fctpx(ipx)
79     
80     *..Initialise the co-ordinates of each separatrix curve
81     
82             do isep=1, 4 !{
83               separx(1,isep,ipx) = ptx(ipx)
84               separy(1,isep,ipx) = pty(ipx)
85             end do !}
86     
87     *..Loop over the directions for each X-point
88     
89             isep = 0
90     
91     cank-20000720 { here we change the algorithm.
92     c*** Originally, the perimeter of the cell containing the X-point was
93     c*** searched for the intersections with the separatrix branches - that
94     c*** is, for the proper PSI value at each cell boundary. If at some 
95     c*** boundary no intersection was found, then the adjacent cell was also
96     c*** examined in a similar way. Due to the numerical noise (?), this
97     c*** often lead to a confusion when not all the separatrix branches 
98     c*** could be identified properly.
99     c*** Now we check the perimeter of the rim of cells (3x3) surrounding 
100     c*** the one with X-point to reduce the effect of noise.
101     
102             i2=ix+2
103             j2=jx-1
104             do idirloop=1,4 !{
105     	  idir=idirloop
106               nbdfav=nbdef
107     c<<<
108           write(0,*) 
109           write(0,*) 'idir,nbdfav,isep =',idir,nbdfav,isep
110     c>>>
111     
112               do ic=1,3 !{
113                 i1 = i2
114                 i2 = i2+incri(idir)
115                 j1 = j2
116                 j2 = j2+incrj(idir)
117                 iref = i2+incrir(idir)
118                 jref = j2+incrjr(idir)
119     c<<<
120                 write(0,*) 'i1,i2,j1,j2,iref,jref =',i1,i2,j1,j2,iref,jref
121                 write(0,*) 'psi(i1,j1),psi(i2,j2),fctpx(ipx)=',
122          ,                                  psi(i1,j1),psi(i2,j2),fctpx(ipx)
123     c>>>
124     
125     *..Check for crossing the segment
126                 if (milieu(psi(i1,j1),psi(i2,j2),fctpx(ipx))) then !{
127                   isep = isep + 1
128                   k = 2
129                   if (idir.eq.1 .or. idir.eq.3) then !{
130                     separx(k,isep,ipx) = x(i1)
131                     separy(k,isep,ipx) = interp(y(j1),y(j2),psi(i1,j1),
132          .                                psi(i2,j2),fctpx(ipx))
133                   else !}{
134                     separx(k,isep,ipx) = interp(x(i1),x(i2),psi(i1,j1),
135          .                                psi(i2,j2),fctpx(ipx))
136                     separy(k,isep,ipx) = y(j1)
137                   endif !}
138                 
139     c<<<
140           write(0,*) 'Cell crossing: k,isep,ipx,separx,separy =',
141          ,                  k,isep,ipx,separx(k,isep,ipx),separy(k,isep,ipx)
142     c>>>
143     
144     *..Call the routine which traces the line and stores the points
145     
146                   call crbniv(iref,jref,k,idir,nxmax,nymax,nx,ny,x,y,psi,
147          .                fctpx(ipx),separx(1,isep,ipx),separy(1,isep,ipx)
148          .                ,npnimx,strumx,npstmx,nstruc,npstru,xstruc,
149          .                ystruc,indstr,xstruc,ystruc,nt,1,plaque,
150          .                ptx(ipx),pty(ipx))
151     
152                   nptot(isep,ipx) = k
153                   call insert(indstr,inddef,nbdef,ipx)
154                   indplq(isep,ipx) = indstr
155                   if(nbdfav.eq.nbdef-1) then !{
156                     nbdfav=nbdef
157                     ixp_hlp(nbdef)=ipx
158                     xsttmp(nbdef)=separx(k,isep,ipx)
159                     ysttmp(nbdef)=separy(k,isep,ipx)
160                   endif !}
161                 endif !}
162               end do !}
163             end do !}
164             write(0,*) 
165             write(0,*) 'After idir loop: isep,ipx,nbdef,nbdfav = ',
166          ,                                        isep,ipx,nbdef,nbdfav !###
167     cank}
168           end do !}
169     c<<<
170           write(0,*) 
171           write(0,*) 'After X-point loop: isep,npx,nbdef,nbdfav = ',
172          ,                                        isep,npx,nbdef,nbdfav !###
173           if(nbdef.gt.0) then !{
174             write(0,*) 'ixp_hlp :',(ixp_hlp(i),i=1,nbdef)
175             write(0,*) 'xsttmp :',(xsttmp(i),i=1,nbdef)
176             write(0,*) 'ysttmp :',(ysttmp(i),i=1,nbdef)
177           end if !}
178           if(npx.gt.0 .and. isep.gt.0) then !{
179             write(0,*) 'nptot'
180             do j=1,npx !{
181               write(0,'(2x,20i4)') (nptot(i,j),i=1,isep)
182             end do !}
183             write(0,*) 'indplq'
184             do j=1,npx !{
185               write(0,'(2x,20i4)') (indplq(i,j),i=1,isep)
186             end do !}
187           end if !}
188     c>>>
189     
190     *   Calculation of psi at the O-point using the fact that the O-point 
191     *  coordinates are at the end of ptx and pty (see routine SELPTX)
192     
193           ipx=npx+1
194           fctpx(ipx) = a00(ix,jx,1) + a10(ix,jx,1)*ptx(ipx)
195          .           + a01(ix,jx,1)*pty(ipx)
196          .           + a11(ix,jx,1)*ptx(ipx)*pty(ipx)
197     
198     c*** Correct the actual number of targets and corresponding lists
199     
200     c<<<
201     c      write(0,*) 'before trgarng'
202     c>>>
203           call trgarng(inddef,ixp_hlp,xsttmp,ysttmp,nbdef,npx)
204           
205     *..Call the subroutine which orders the divertor plates according to the
206     *   specific order
207     
208     c<<<
209     c      write(0,*) 'before orddef'
210     c>>>
211           call orddef(inddef,nbdef,xsttmp,ysttmp,ptx,pty,npx)
212     
213     c<<<
214           write(0,*) 'Leaving sptris...'
215           write(0,*) 
216     c>>>
217           return
218           end
