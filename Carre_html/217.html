1     ************************************************************************
2           LOGICAL FUNCTION chgdir(nivox,nivoy,nivxtm,nivytm)
3     ************************************************************************
4           IMPLICIT NONE
5     
6     *..  Cette fonction verifie si 2 vecteurs sont de sens inverse ou non.
7     
8     *  arguments
9           REAL*8 nivox(*),nivoy(*),nivxtm(*),nivytm(*)
10     
11     *  variables locales
12           REAL*8 x1,x2,y1,y2,scal
13     
14     c=========================
15     c.. nivox,nivoy: tableaux contenant les coordonnees des points du
16     c                premier vecteur.
17     c.. nivxtm,nivytm: tableaux contenant les coordonnees des points du
18     c                deuxieme vecteur.
19     c.. x1,y1,x2,y2: vecteurs 1 et 2
20     c.. scal: produit scalaire entre les vecteurs 1 et 2.
21     c=========================
22     
23           x1 = nivxtm(2) - nivxtm(1)
24           x2 = nivox(2) - nivox(1)
25           y1 = nivytm(2) - nivytm(1)
26           y2 = nivoy(2) - nivoy(1)
27     
28           scal = x1*x2 + y1*y2
29     
30           IF (scal .LT. 0.) THEN
31              chgdir = .TRUE.
32           ELSE
33              chgdir = .FALSE.
34           ENDIF
35     
36           RETURN
37           END
