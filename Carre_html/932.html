2     ************************************************************************
3           REAL*8 FUNCTION long(vx,vy,n)
4     ************************************************************************
5           IMPLICIT NONE
6     
7     *..  Cette fonction calcule la distance totale que parcourt une courbe v
8     *  de n points.
9     
10     *  arguments
11           INTEGER n
12           REAL*8 vx(n),vy(n)
13     
14     *  variables locales
15           INTEGER i
16           REAL*8 dist
17     
18     *  procedures
19           INTRINSIC SQRT
20     
21     c=========================
22     c.. n   : nombre  de points de la courbe.
23     c.. vx,vy: vecteurs des coordonnees des points de la courbe.
24     c.. dist: distance entre 2 points consecutifs de la courbe.
25     c=========================
26     
27           long = 0.0
28     
29           DO 10 i=1, n-1
30     
31              dist = SQRT((vx(i+1)-vx(i))**2 + (vy(i+1)-vy(i))**2)
32              long = long + dist
33     
34        10 CONTINUE
35     
36           RETURN
37           END
