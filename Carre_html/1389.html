2     ************************************************************************
3           REAL*8 FUNCTION plqdst(x0,y0,x2,y2,xst,yst,npst,doug)
4     ************************************************************************
5     c
6     c  version : 18.06.98 20:20
7     c
8           IMPLICIT NONE
9     
10     *..Cette sous-routine calcule la distance entre deux points sur une meme
11     *  structure, en longeant les segments de structure.
12     
13     *  arguments
14           INTEGER npst
15           REAL*8 x0,y0,x2,y2,xst(npst),yst(npst)
16           CHARACTER*(*) doug
17     
18     *  variables locales
19           INTEGER ind,ind1,ind2,sens
20           REAL*8 x1,y1,x3,y3,dist
21     
22     *  procedures
23           INTEGER indsgm,drctio
24           INTRINSIC MOD,SQRT
25           EXTERNAL drctio,indsgm
26          ,        ,trc_stk_in,trc_stk_out
27     
28     c=========================
29     c.. xst,yst: tableaux des coordonnees des points de la structure.
30     c.. npst: nombre de points de la structure.
31     c.. x0,y0: point de depart sur la structure.
32     c.. x2,y2: point d'arrivee sur la structure.
33     c.. doug: droite ou gauche
34     c.. ind2: indice du segment sur lequel se trouve le point 2.
35     c.. sens: sens dans lequel il faut partir pour la recherche,
36     c         1 = sens des points de la structure, -1 = sens contraire.
37     c.. x1,y1,x3,y3: points intermediaires.
38     c=========================
39     
40     *..Copie d'arguments en variables locales
41     
42           x1 = x0
43           y1 = y0
44     
45     *..Recherche du sens et indices de segment pour les deux points.
46     
47           call trc_stk_in('plqdst','*start')
48           sens = drctio(xst,yst,npst,x1,y1,doug)
49           call trc_stk_out
50     
51           ind1 = indsgm(xst,yst,npst,x1,y1)
52           ind2 = indsgm(xst,yst,npst,x2,y2)
53     
54     *..On verifie si les points sont sur le meme segment.
55     
56           IF (ind1 .EQ. ind2) THEN
57              plqdst = SQRT((x1-x2)**2 + (y1-y2)**2)
58              RETURN
59           ELSE
60     
61     *..Sinon on calcule la distance de coin en coin jusqu'a ce qu'on soit
62     *  sur le segment du point d'arrivee.
63     
64              dist= 0.
65     
66        10    CONTINUE
67     
68              IF (sens .EQ. 1) THEN
69                 ind = MOD(ind1,npst-1) + 1
70              ELSE
71                 ind = ind1
72              ENDIF
73     
74              x3 = xst(ind)
75              y3 = yst(ind)
76     
77              dist = dist + SQRT((x1-x3)**2 + (y1-y3)**2)
78     
79              IF (sens .EQ. 1) THEN
80     
81                 IF (ind .NE. ind2) THEN
82                    x1 = x3
83                    y1 = y3
84                    ind1 = ind
85                    GO TO 10
86                 ELSE
87                    plqdst = dist + SQRT((x2-x3)**2 + (y2-y3)**2)
88                    RETURN
89                 ENDIF
90     
91              ELSE
92     
93                 IF (ind .NE. MOD(ind2,npst-1) + 1) THEN
94                    x1 = x3
95                    y1 = y3
96                    ind1 = ind1 - 1
97                    IF (ind1 .EQ. 0) ind1 = npst - 1
98                    GO TO 10
99                 ELSE
100                    plqdst = dist + SQRT((x2-x3)**2 + (y2-y3)**2)
101                    RETURN
102                 ENDIF
103     
104              ENDIF
105           ENDIF
106     
107           END
