1           SUBROUTINE DERIVE(nx, ny, x, y, psi, psidx, psidy)
2     c
3     c  version : 07.07.97 19:37
4     c
5     c======================================================================
6     *..  Cette sous-routine calcule en chaque point ou psi est defini, la
7     *  derivee selon x et la derivee selon y.
8     c======================================================================
9     
10           IMPLICIT NONE
11     
12     cank-970707: dimensions from the file
13     *  dimensions
14     #include <CARREDIM.F>
15     
16     *  arguments
17           INTEGER nx, ny
18           REAL*8 psi(nxmax,nymax), psidx(nxmax,nymax), psidy(nxmax,nymax),
19          .  x(nx),y(ny)
20     
21     *  variables locales
22           INTEGER i, j, nxymax
23           parameter (nxymax=nymax)
24           REAL*8 tmp(nxymax), tmpp(nxymax)
25     
26     *  procedures
27           EXTERNAL DERIV
28     
29     c=========================
30     c.. x,y: tableaux des coordonnees des points.
31     c.. nx,ny: nombre de points dans chaque tableau.
32     c.. psi: tableau de la valeur de la fonction en chacun de ces points.
33     c.. psidx: tableau de la valeur de la fonction derivee par rapport a x
34     c        en chacun de ces points.
35     c.. psidy: tableau de la valeur de la fonction derivee par rapport a y
36     c        en chacun de ces points.
37     c.. i,j: indices
38     c.. tmp: tableau selon une ligne ou une colonne de psi.
39     c.. tmpp: tableau de la meme ligne ou colonne de derivee de psi par
40     c         rapport a x ou y.
41     c=========================
42     
43     *
44     *.. 1  Calcul des derivees partielles par rapport a X.
45     *
46     
47     *..Boucle sur toutes les lignes.
48     
49           DO 10 j=1, ny
50     
51     *..Copie du vecteur (ligne) dans un vecteur temporaire.
52     
53              DO 15 i=1, nx
54                 tmp(i) = psi(i,j)
55        15    CONTINUE
56     
57     *..Calcul de la derivee de ce vecteur.
58     
59              CALL DERIV(nxmax, x, tmp, tmpp, nx)
60     
61     *..Copie du vecteur de derivees par rapport a x temporaire dans le
62     *  vecteur de derivee par rapport a X: psidx
63     
64              DO 20 i=1, nx
65                 psidx(i,j) = tmpp(i)
66        20    CONTINUE
67     
68        10 CONTINUE
69     
70     *
71     *.. 2  Calcul des derivees partielles par rapport a Y.
72     *
73     
74     *..Boucle sur toutes les colonnes.
75     
76           DO 30 i=1, nx
77     
78     *..Copie du vecteur (colonne) dans un vecteur temporaire.
79     
80              DO 35 j=1, ny
81                 tmp(j) = psi(i,j)
82        35    CONTINUE
83     
84     *..Calcul de la derivee de ce vecteur.
85     
86              CALL DERIV(nymax, y, tmp, tmpp, ny)
87     
88     *..Copie du vecteur de derivees par rapport a Y temporaire dans le
89     *  vecteur de derivee par rapport a Y: psidy
90     
91              DO 40 j=1, ny
92                 psidy(i,j) = tmpp(j)
93        40    CONTINUE
94     
95        30 CONTINUE
96     
97           RETURN
98           END
