1     ************************************************************************
2           subroutine entete(nin,chaine,iflag)
3     ************************************************************************
4     c
5     c  version : 04.04.97 21:21
6     c
7           implicit none
8     c  cette routine lit jusqu'a la ligne qui contient la chaine de
9     c  caracteres specifiee dans chaine. si iflag =1 a l'entree, le fichier
10     c  est rembonine avant la lecture.
11     c
12     c  nin: unite de lecture
13     c  chaine: chaine de caracteres qu'il faut trouver
14     c  iflag:
15     c    entree: 1 pour commencer la lecture a partir du debut du fichier
16     c            2 comme 1, mais le programme arrete en cas d'echec
17     c            0 pour commencer la lecture a partir de la position
18     c              actuelle
19     c           -1 comme 0 mais le programme arrete en cas d'echec
20     c    sortie: 0 succes
21     c            1 echec
22     c
23     c  arguments
24           integer nin,iflag
25           character chaine*(*)
26     c
27     c  variables locales
28           character ligne*200
29     c
30     c  procedures
31           intrinsic index
32     c
33     c  calculs
34           if(nin.eq.5) print *,'starting entete 5'
35           if(iflag.ge.1) rewind nin
36     1     continue
37           read(nin,100,end=99)ligne
38     100   format(a)
39           if(index(ligne,chaine).eq.0) go to 1
40     c
41     c  chaine trouvee
42           iflag=0
43           return
44     c
45     c  chaine non trouvee
46     99    continue
47           if(iflag.eq.-1 .or. iflag.eq.2) then
48             write(6,*)'Incapable de trouver l''entete suivante:'
49             write(6,*)chaine
50             write(6,*)'Arret du programme.'
51             stop
52           endif
53           iflag=1
54           return
55           end
