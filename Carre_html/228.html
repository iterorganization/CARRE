3           subroutine clort(x1,y1,x2,y2,ort,nppol,pasmin,g1,g2,l0,l2)
4     ************************************************************************
5           implicit none
6     c  calcul des fonctions qui s'annulent quand deux series de points sont
7     c  disposees de facon "orthogonale".
8     c
9     c  arguments
10           integer nppol
11           real*8 x1(nppol),y1(nppol),x2(nppol),y2(nppol),ort(nppol),pasmin,
12          .  g1,g2,l0(nppol),l2(nppol)
13     c
14     c  x1, y1: coordonnees des points de maille de reference (sur la courbe
15     c          precedente
16     c  x2, y2: coordonnees des points de maille a ajuster
17     c  ort: tableau de la fonction dont il faut trouver le zero
18     c  pasmin: plus petit espacement entre deux points de maille
19     c  g1, g2: gardes gauche et droite
20     c  l0: tableau des positions initiales
21     c  l2: tableau des positions actuelles
22     c
23     c  variables locales
24           integer i
25           real*8 cs1,cs2,cs3,cs4,l1m,l1p,l2m,l2p,l12,zero,fac1,fac2,fac,
26          .  l12t,un
27           parameter(zero=0.,un=1.)
28     c
29     c  procedures
30           intrinsic sqrt
31     c
32     c  calculs
33           i=1
34           l1p=sqrt((x1(i)-x1(i+1))**2+(y1(i)-y1(i+1))**2)
35           l2p=sqrt((x2(i)-x2(i+1))**2+(y2(i)-y2(i+1))**2)
36     c  on trouve empiriquement que la convergence est meilleure avec une
37     c  valeur de l12 constante.
38           l12t=sqrt((x2(i)-x1(i))**2+(y2(i)-y1(i))**2)
39           if(g1.gt.zero .and. g2.gt.zero) then
40             do i=2,nppol-1
41               fac1=(g1/(g1+l0(i)))**2
42               fac2=(g2/(g2+l0(nppol)-l0(i)))**2
43               fac=fac1*(un-fac2)+fac2
44               l12=fac*l0(nppol)+(1.-fac)*l12t
45     c
46               l1m=l1p
47               l2m=l2p
48               l1p=sqrt((x1(i)-x1(i+1))**2+(y1(i)-y1(i+1))**2)
49               l2p=sqrt((x2(i)-x2(i+1))**2+(y2(i)-y2(i+1))**2)
50               cs1=(x1(i)-x2(i))*(x2(i-1)-x2(i))
51          .       +(y1(i)-y2(i))*(y2(i-1)-y2(i))
52               cs2=(x1(i)-x2(i))*(x2(i+1)-x2(i))
53          .       +(y1(i)-y2(i))*(y2(i+1)-y2(i))
54               cs3=(x2(i)-x1(i))*(x1(i-1)-x1(i))
55          .       +(y2(i)-y1(i))*(y1(i-1)-y1(i))
56               cs4=(x2(i)-x1(i))*(x1(i+1)-x1(i))
57          .       +(y2(i)-y1(i))*(y1(i+1)-y1(i))
58               cs1=cs1/(l2m*l12)
59               cs2=cs2/(l2p*l12)
60               cs3=cs3/(l1m*l12)
61               cs4=cs4/(l1p*l12)
62               ort(i)=
63          .       cs2+cs3-cs1-cs4
64          .      -((g1/l0(i))**2+(g2/(l0(nppol)-l0(i)))**2)
65          .      *(l2(i)-l0(i)/l0(nppol)*l2(nppol))/(pasmin+g1+g2)
66          .      +(pasmin/(l2(i)-l2(i-1)))**2-(pasmin/(l2(i+1)-l2(i)))**2
67     
68             enddo
69           else
70             do i=2,nppol-1
71               l12=l12t
72     c
73               l1m=l1p
74               l2m=l2p
75               l1p=sqrt((x1(i)-x1(i+1))**2+(y1(i)-y1(i+1))**2)
76               l2p=sqrt((x2(i)-x2(i+1))**2+(y2(i)-y2(i+1))**2)
77               cs1=(x1(i)-x2(i))*(x2(i-1)-x2(i))
78          .       +(y1(i)-y2(i))*(y2(i-1)-y2(i))
79               cs2=(x1(i)-x2(i))*(x2(i+1)-x2(i))
80          .       +(y1(i)-y2(i))*(y2(i+1)-y2(i))
81               cs3=(x2(i)-x1(i))*(x1(i-1)-x1(i))
82          .       +(y2(i)-y1(i))*(y1(i-1)-y1(i))
83               cs4=(x2(i)-x1(i))*(x1(i+1)-x1(i))
84          .       +(y2(i)-y1(i))*(y1(i+1)-y1(i))
85               cs1=cs1/(l2m*l12)
86               cs2=cs2/(l2p*l12)
87               cs3=cs3/(l1m*l12)
88               cs4=cs4/(l1p*l12)
89               ort(i)=
90          .       cs2+cs3-cs1-cs4
91          .      +(pasmin/(l2(i)-l2(i-1)))**2-(pasmin/(l2(i+1)-l2(i)))**2
92     
93             enddo
94           endif
95     c
96           return
97           end
