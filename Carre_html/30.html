1           real*8 function aire(x,y,n)
2     c
3     c
4     c  VERSION: 03.08.99 12:12
5     c=======================================================================
6     c*** Calculate the area of a polygon with n-1 nodes (the n-th node 
7     c*** coincides with the 1st one).
8     c*** The area is negative if the nodes are listed clockwise.
9     c*** The area is calculated via decomposition of the polygon into a set
10     c*** of triangles having a common node at the first node of the polygon. 
11     c=======================================================================
12           implicit none
13           integer n
14           real*8 x(n),y(n)
15           integer i,k
16           real*8 x1,y1,x2,y2
17           logical hlp
18           data hlp /.true./, k /0/
19     c=======================================================================
20     
21     c<<<
22     c      k=k+1
23     c      hlp=k.le.4
24     c      if(hlp) then !{
25     c      	write(0,*) 'new aire',k,n
26     c	write(0,'(4h x: ,1p,10e11.4/(4x,10e11.4))') x
27     c	write(0,'(4h y: ,1p,10e11.4/(4x,10e11.4))') y
28     c	write(0,'(a4,7a7,4a10)') 'i','x(1)','y(1)','x(i)','y(i)',
29     c     ,	    'x(i+1)','y(i+1)','x1','y1','x2','y2','aire'
30     c      end if !}
31     c>>>
32           aire=0.
33           do i=2,n-2 !{
34             x1=x(i)-x(1)
35             y1=y(i)-y(1)
36             x2=x(i+1)-x(1)
37             y2=y(i+1)-y(1)
38             aire=aire+(x1*y2-x2*y1)
39     c<<<
40     c      	if(hlp) then !{
41     c	  write(0,'(1x,i3,6f7.3,1p,5e10.3)') i,x(1),y(1),x(i),y(i),
42     c     ,	    x(i+1),y(i+1),x1,y1,x2,y2,aire
43     c      	end if !}
44     c>>>
45           end do !}
46           aire=0.5*aire
47           hlp=.false.
48     c=======================================================================
49           end
