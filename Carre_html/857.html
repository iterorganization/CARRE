1           subroutine limfnd(xpto,ypto,nivx,nivy,stp0,stpmin,distnv,nivtot,
2          .    nbniv,nx,ny,x,y,psi,npx,ptx,pty,fctpx,
3          .    nstruc,npstru,xstruc,ystruc,indplq,inddef,nbdef,
4          .    a00,a10,a01,a11)
5     c
6     c  version : 07.07.97 20:10
7     c
8     c======================================================================
9     c  This routine finds the innermost point of the limitor. It defines
10     c  (ptx,pty), the coordinates of the X-point, to be the coordinates of
11     c  that point. Finally, it parametrises the separatrix (the flux surface
12     c  which goes through (ptx,pty) in curve nivx(*,1,1),nivy(*,1,1).
13     c======================================================================
14           implicit none
15     
16     cank-970707: dimensions from the file
17     *  dimensions
18     #include <CARREDIM.F>
19     
20     c  arguments
21           INTEGER nivtot(*),nbniv,
22          .  nx,ny,npx,
23          .  nstruc,npstru(strumx),
24          .  indplq(4,npxmx),inddef(nbdmx),nbdef
25           REAL*8 xpto,ypto,nivx(npnimx,*),nivy(npnimx,*),stp0,stpmin,
26          .  distnv(5,*),a00(nxmax,nymax,3),a10(nxmax,nymax,3),
27          .  a01(nxmax,nymax,3),a11(nxmax,nymax,3),
28          .  x(nxmax),y(nymax),psi(nxmax,nymax),ptx(npxmx),pty(npxmx),
29          .  fctpx(npxmx),xstruc(npstmx,strumx),ystruc(npstmx,strumx)
30     c
31     c  local variables
32           real*8 zero
33           parameter(zero=0.)
34           integer dir,sens,nt(2),nbcrb,ind0,i,ii,jj,k,plaque,nstrp1,
35          .  idef,idniv,indlim,ipx
36           real*8 xt(npnimx,2),yt(npnimx,2),
37          ,                                       x0,y0,psi0,x2,y2,psi2,norm
38     c
39     c  ind0: indice du point du limiteur qui est le plus rapproche de la
40     c        separatrice
41     c  x0, y0: coordonees du point d'indice ind0
42     
43     c
44     c  procedures
45           integer ifind,horair
46           real*8 aire,plqdst
47           external aire,crbniv,horair,ifind,plqdst
48           intrinsic sqrt
49     c
50     c  calculation
51     c
52     c  1.   definition of a line from the magnetic axis to a point outside
53     c       the separatrix
54           if(nstruc.lt.strumx) then
55             nstrp1=nstruc+1
56             npstru(nstrp1)=3
57             xstruc(1,nstrp1)=xpto
58             ystruc(1,nstrp1)=ypto
59             xstruc(2,nstrp1)=xpto
60             ystruc(2,nstrp1)=max(y(1),y(ny))
61             xstruc(3,nstrp1)=xpto
62             ystruc(3,nstrp1)=1.e20
63           else
64             write(6,*)'Dimension of structures (strumx) must be increased.'
65             write(6,*)'strumx, nstruc=',strumx,nstruc
66             stop
67           endif
68     c
69     c  2.   walk from point 1 towards point 2 until a structure is reached
70           sens=1
71           idniv=1
72     c       courbe a ne pas traverser
73           nbcrb=1
74           nt(1)=2
75           nt(2)=0
76           xt(1,nbcrb)=xpto
77           yt(1,nbcrb)=ypto
78           norm=sqrt((xstruc(2,nstrp1)-xstruc(1,nstrp1))**2+
79          .  (ystruc(2,nstrp1)-ystruc(1,nstrp1))**2)
80           xt(2,nbcrb)=xpto-1.e-2*stpmin*(ystruc(2,nstrp1)-ystruc(1,nstrp1))
81          .  /norm
82           yt(2,nbcrb)=ypto+1.e-2*stpmin*(xstruc(2,nstrp1)-xstruc(1,nstrp1))
83          .  /norm
84     
85           CALL MARCHE(xpto,ypto,nstrp1,nstrp1,sens,nivx(1,idniv),
86          .      nivy(1,idniv),
87          .      nivtot(idniv),nbcrb,xt,yt,nt,stp0,stpmin,
88          .      nx,ny,x,y,psi,nstrp1,npstru,xstruc,
89          .      ystruc,a00,a10,a01,a11,indlim)
90           plaque=indlim
91     c
92     c  2.1  definition des indices de plaque pour usage dans maille
93           indplq(1,1)=plaque
94           indplq(2,1)=plaque
95           inddef(1)=plaque
96           inddef(2)=plaque
97     
98     c
99     c  3.   On trouve le point du limiteur qui est le plus rapproche de la
100     c       derniere surface de flux parametrisee.
101           call plurap(xstruc(1,plaque),ystruc(1,plaque),npstru(plaque),
102          .  nivx(1,idniv),nivy(1,idniv),nivtot(idniv),x0,y0,ind0)
103     c
104     c  4.   parametrisation de la separatrice.
105           ptx(1)=x0
106           pty(1)=y0
107           nbcrb=1
108           nt(1)=0
109           ii = ifind(x0,x,nx,1)
110           jj = ifind(y0,y,ny,1)
111     
112           psi2 = a00(ii,jj,1) + a10(ii,jj,1)*x0 + a01(ii,jj,1)*y0 +
113          .       a11(ii,jj,1)*x0*y0
114     
115           ipx = 1
116           dir = 0
117           k = 1
118           nbdef=0
119     
120           nivx(k,idniv)=x0
121           nivy(k,idniv)=y0
122           CALL CRBNIV(ii,jj,k,dir,nxmax,nymax,nx,ny,x,y,psi,psi2,
123          .  nivx(1,idniv),nivy(1,idniv),npnimx,strumx,npstmx,
124          .  nstruc,npstru,xstruc,ystruc,indlim,xt,yt,nt,nbcrb,plaque,x0,y0)
125           call insert(indlim,inddef,nbdef,ipx)
126           indplq(idniv,ipx) = indlim
127           CALL CRBNIV(ii,jj,k,dir,nxmax,nymax,nx,ny,x,y,psi,psi2,
128          .  nivx(1,idniv),nivy(1,idniv),npnimx,strumx,npstmx,
129          .  nstruc,npstru,xstruc,ystruc,indlim,xt,yt,nt,nbcrb,plaque,x0,y0)
130           nivtot(idniv)=k
131           call insert(indlim,inddef,nbdef,ipx)
132           indplq(idniv,ipx) = indlim
133     
134           if((nivx(1,idniv)-nivx(nivtot(idniv),idniv))**2
135          .  +(nivy(1,idniv)-nivy(nivtot(idniv),idniv))**2. gt.
136          .  2.*(x(2)-x(1))**2) then
137             write(6,*)'attention: la courbe parametrisee est mal fermee.'
138             print*,'nivtot=',nivtot(idniv)
139             print*,'nivx/y(1)=',nivx(1,idniv),nivy(1,idniv)
140             print*,'nivx/y(f)=',nivx(nivtot(idniv),idniv),
141          .    nivy(nivtot(idniv),idniv)
142           endif
143           nivx(nivtot(idniv),idniv)=nivx(1,idniv)
144           nivy(nivtot(idniv),idniv)=nivy(1,idniv)
145           if(aire(nivx,nivy,nivtot(idniv)).lt.zero) then
146             do i=1,nivtot(idniv)/2
147               ii=nivtot(idniv)-i+1
148               psi2=nivx(i,idniv)
149               nivx(i,idniv)=nivx(ii,idniv)
150               nivx(ii,idniv)=psi2
151               psi2=nivy(i,idniv)
152               nivy(i,idniv)=nivy(ii,idniv)
153               nivy(ii,idniv)=psi2
154             enddo
155           endif
156     c
157     c  5.   Recherche de la frontiere exterieure
158           idniv=2
159           sens=1
160           sens=horair(xpto,ypto,x0,y0,xstruc(1,plaque),ystruc(1,plaque),
161          .  npstru(plaque),sens)
162           nbcrb=1
163           nt(nbcrb)=nivtot(1)
164           do i=1,nt(nbcrb)
165             xt(i,nbcrb)=nivx(i,1)
166             yt(i,nbcrb)=nivy(i,1)
167           enddo
168           CALL MARCHE(nivx(1,1),nivy(1,1),plaque,plaque,sens,nivx(1,idniv),
169          .      nivy(1,idniv),
170          .      nivtot(idniv),nbcrb,xt,yt,nt,stp0,stpmin,
171          .      nx,ny,x,y,psi,nstruc,npstru,xstruc,
172          .      ystruc,a00,a10,a01,a11,indlim)
173     
174     c
175     c  5.1  Calcul de la distance entre le point de contact et la derniere
176     c       surface de flux.
177     
178           idef=1
179     
180           distnv(1,idef) = 0.
181           distnv(2,idef) = 0.
182     
183     *..Calcul de la distance absolue entre le point de depart et le
184     *  premier point de la ligne de niveau frontiere.
185     
186           x2 = nivx(1,2)
187           y2 = nivy(1,2)
188     
189           distnv(1,idef) = plqdst(x0,y0,x2,y2,xstruc(1,inddef(idef)),
190          .                   ystruc(1,inddef(idef)),npstru(inddef(idef))
191          .                   ,'droite')
192     
193     *..Calcul de la difference de psi entre ces 2 memes points.
194     
195           ii = ifind(x0,x,nx,1)
196           jj = ifind(y0,y,ny,1)
197     
198           psi0 = a00(ii,jj,1) + a10(ii,jj,1)*x0 + a01(ii,jj,1)*y0 +
199          .       a11(ii,jj,1)*x0*y0
200     
201           ii = ifind(x2,x,nx,1)
202           jj = ifind(y2,y,ny,1)
203     
204           psi2 = a00(ii,jj,1) + a10(ii,jj,1)*x2 + a01(ii,jj,1)*y2 +
205          .       a11(ii,jj,1)*x2*y2
206     
207           distnv(2,idef) = psi2 - psi0
208     c
209     c  6.   Definition du nombre de courbes de niveau calculees.
210           nbniv=2
211     c
212     c  7.   Calcul de psi aux points X et O.
213           ipx=npx
214           ii=ifind(ptx(ipx),x,nx,1)
215           jj=ifind(pty(ipx),y,ny,1)
216           fctpx(ipx) = a00(ii,jj,1) + a10(ii,jj,1)*ptx(ipx)
217          .           + a01(ii,jj,1)*pty(ipx)
218          .           + a11(ii,jj,1)*ptx(ipx)*pty(ipx)
219           ipx=npx+1
220           ii=ifind(ptx(ipx),x,nx,1)
221           jj=ifind(pty(ipx),y,ny,1)
222           fctpx(ipx) = a00(ii,jj,1) + a10(ii,jj,1)*ptx(ipx)
223          .           + a01(ii,jj,1)*pty(ipx)
224          .           + a11(ii,jj,1)*ptx(ipx)*pty(ipx)
225     
226           return
227           end
