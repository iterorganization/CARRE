1           SUBROUTINE SELPTX(npxtot,npx,pointx,pointy,ii,jj,ptx,
2          .                  pty,iptx,jptx,xpto,ypto,racord,limcfg)
3     c
4     c  version : 07.02.99 22:05
5     c
6     c======================================================================
7           IMPLICIT NONE
8     
9     *..  Cette sous-routine imprime la liste des points ou le gradient
10     *  s'annule et nous demande lequel(lesquels) on prendra comme point(s)
11     *  X. S'il y a plus d'un point X alors ils sont ordonnes du haut vers le
12     *  bas. On demande aussi l'indice du point central (point O), ainsi que,
13     *  dans le cas ou il y a 2 points X, si on desire qu'ils soient
14     *  raccordes.
15     
16     cank-970702: moved dimensions into a separate file
17     *  dimensions
18     #include <CARREDIM.F>
19     
20     *  arguments
21           INTEGER npxtot,npx,ii(gradmx),jj(gradmx),iptx(npxmx),
22          .        jptx(npxmx),limcfg
23           REAL*8 pointx(gradmx),pointy(gradmx),ptx(npxmx),pty(npxmx),
24          .       xpto,ypto
25           LOGICAL racord
26     
27     *  variables en common
28     
29     #include <COMLAN.F>
30     
31     *  variables locales
32           INTEGER i, j, k, ipx, ipo, iptxtm, jptxtm
33           REAL*8 ptxtmp,ptytmp,eps_hlp
34           CHARACTER rep*1
35     c*** Tolerance for coinciding X-points
36           parameter (eps_hlp=1.e-5)
37     cank-970701 <<<
38     c*** Automatic definition of the X- and O-points from DG data
39     c***
40     c*** The code tries to identify the pre-selected points with those
41     c*** provided by Carre. If such an identification is unambiguous,
42     c*** then the user is prompted to accept these points. Otherwise, or
43     c*** if the user does not accept the selection, the old scheme works.
44     c*** The pre-selected points should be specified in a file "selptx.inf"
45     c*** in the format suitable for "read *" in the following sequence:
46     c***
47     c***  xpttol
48     c***  nptx
49     c***  (xptxm(i),yptxm(i),i=1,nptx)
50     c***  xptom,yptom
51     c***
52     c*** nptx =< 0 means a limiter configuration
53     c***
54     c***  nptxm         maximum number of the selected X-points ( =< 2 )
55     c***  nptx          real number of the selected X-points
56     c***  xptxm, yptxm  X-point coordinates read from the file
57     c***  xptom, yptom  O-point coordinates read from the file
58     c***  xpttol        tolerance parameter - related to the distance
59     c***                between the points
60     c***  lconn         1: connect X-points, 0: do not, -1: ask user
61     c***  lun           unit number for the point specification file
62     c
63           integer nptxm,nptx,lun,lconn
64           parameter (nptxm=2)
65           integer kptx(nptxm)
66           real*8 xptxm(nptxm),yptxm(nptxm),xptom,yptom,
67          ,                         xpttol,xptdist(gradmx),rdist,x1,x2,y1,y2
68     c      parameter (xpttol=5.e-2)
69           logical ex,op
70     cank>>>
71     c======================================================================
72     c.. npxtot: nombre de points ou le gradient est nul.
73     c.. pointx,pointy: tableaux des coordonnees en x et y des points ou le
74     c                  gradient est nul.
75     c.. ii,jj: tableaux des indices des carres ou le gradient est nul.
76     c.. npx: nombre de points X.
77     c.. ptx,pty: tableaux des coordonnees en x et y des points X.
78     c.. iptx,jptx: tableaux des indices des carres ou il y a un point X.
79     c.. xpto,ypto: coordonnees du point O.
80     c.. racord: determine si oui ou non les points X sont connectes.
81     c.. ipx,ipo: indice du point X et du point O.
82     c.. iptxtm,jptxtm: tableaux temporaires des indices des carres ou il y
83     c                  a un point X.
84     c.. ptxtmp,ptytmp: tableaux temporaires des coordonnees en x et y des
85     c                  points X.
86     c======================================================================
87           rdist(x1,y1,x2,y2)=sqrt((x1-x2)**2+(y1-y2)**2)
88     c======================================================================
89     *..Initialisation
90     
91           npx = 0
92           do i=1,nptxm
93     	kptx(i)=0
94           enddo
95           lconn=-1
96     
97     cank-970701 <<<
98     c*** Check whether the point specification file is present
99           inquire(file='selptx.inf',exist=ex)
100           if(ex) then
101             do lun=7,99
102               inquire(lun,opened=op)
103               if(.not.op) go to 10
104             end do
105             write(*,*) '*** cannot open selptx.inf: no free unit number'
106             lun=0
107             ex=.false.
108             go to 20
109      10     continue
110             open(lun,file='selptx.inf',err=12)
111             rewind(lun)
112             read(lun,*,err=14) xpttol
113             read(lun,*,err=14) nptx
114             ex=nptx.le.nptxm
115             if(ex) then
116               nptx=max0(nptx,0)
117               do i=1,nptx
118                 read(lun,*,err=14) xptxm(i),yptxm(i)
119               end do
120               read(lun,*,err=14) xptom,yptom
121     c*** lconn>0 means "connect x-points")
122               read(lun,*,end=11, err=111) lconn
123     	  go to 11
124      111  	  lconn=-1
125     
126     c*** Set the absolute tolerance for each candidate point
127     
128      11         do j=1,npxtot
129                 x1=1.e30
130                 do i=1,npxtot
131                   if(i.ne.j) then
132                     x2=rdist(pointx(i),pointy(i),pointx(j),pointy(j))
133                     if(x2.gt.eps_hlp) x1=min(x1,x2)
134                   end if
135                 end do
136                 xptdist(j)=x1*xpttol
137               end do
138     c
139     c*** Try to identify the pre-specified points
140     c
141               k=0
142               do i=1,npxtot
143                 if(abs(xptom-pointx(i)).le.xptdist(i) .and.
144          .                         abs(yptom-pointy(i)).le.xptdist(i)) then
145                   if(k.eq.0) then
146                     k=i
147                   else if(rdist(pointx(i),pointy(i),pointx(k),pointy(k))
148          .      	      	      	      	      	      .gt.eps_hlp) then
149                     write(*,*) 'Ambiguous definition of the O-point: ',
150          ,                         'points ',k,'  and ',i,'  are too close'
151                     ex=.false.
152                     go to 20
153                   end if
154                 end if
155               end do
156               if(k.eq.0) then
157                 write(*,*) 'No O-point identified - ',
158          ,                                       'check the specification!'
159                 ex=.false.
160                 go to 20
161               end if
162               ipo=k
163               do j=1,nptx
164                 k=0
165                 do i=1,npxtot
166                   if(abs(xptxm(j)-pointx(i)).le.xptdist(i) .and.
167          ,                      abs(yptxm(j)-pointy(i)).le.xptdist(i)) then
168                     if(k.eq.0) then
169                       k=i
170                   else if(rdist(pointx(i),pointy(i),pointx(k),pointy(k))
171          .      	      	      	      	      	      .gt.eps_hlp) then
172                       write(*,*) 'Ambiguous definition of the X-point',
173          ,                     j,': points ',k,'  and ',i,'  are too close'
174                       ex=.false.
175                       go to 20
176                     end if
177                   end if
178                 end do
179                 if(k.eq.0) then
180                   write(*,*) 'No X-point ',j,' identified - ',
181          ,                                       'check the specification!'
182                   ex=.false.
183                   go to 20
184                 end if
185                 if(k.eq.ipo) then
186                   write(*,*) 'Ambiguous definition of the X-point',
187          ,                              j,': the same as the plasma center'
188                   ex=.false.
189                   go to 20
190                 end if
191                 do i=1,j-1
192                   if(k.eq.kptx(j)) then
193                     write(*,*) 'Ambiguous definition of the X-point',
194          ,                                  j,': the same as the X-point',i
195                     ex=.false.
196                     go to 20
197                   end if
198                 end do
199                 kptx(j)=k
200               end do
201     c
202     c*** Here the o-point and nptx x-points are identified.
203     c*** Ask for confirmation.
204     c
205               write(*,*) 'Pre-selected points are identified.'
206               write(*,'(1x,a9,1p,2e14.4)')
207          -                            'O-point: ',pointx(ipo),pointy(ipo)
208               if(nptx.gt.0) then
209                 write(*,'(1x,a9,1p,2e14.4/(10x,2e14.4))')
210          -           'X-point: ',(pointx(kptx(i)),pointy(kptx(i)),i=1,nptx)
211                 if(nptx.gt.1) then
212     	      if(lconn.gt.0) then
213     	      	write(*,*) 'Connected double-null'
214     	      else if(lconn.eq.0) then
215     	      	write(*,*) 'Disconnected double-null'
216                   end if
217                 end if
218               else
219                 write(*,*) 'No X-point - limiter configuration'
220               end if
221     	  
222               write(*,*)
223               write(*,*) 'Do you accept the selection (y/n)?'
224               read(*,'(a)') rep
225               ex=rep.eq.'y'.or.rep.eq.'Y'
226               if(.not.ex) go to 20
227     c
228     c*** The configuration is accepted. Set the Carre variables
229     c
230               if(nptx.le.0) then
231                 npx=1
232                 limcfg=1
233               else
234                 limcfg=0
235                 npx=nptx
236                 do i=1,npx
237                   ptx(i)=pointx(kptx(i))
238                   pty(i)=pointy(kptx(i))
239                   iptx(i)=ii(kptx(i))
240                   jptx(i)=jj(kptx(i))
241                 end do
242               end if
243               xpto=pointx(ipo)
244               ypto=pointy(ipo)
245               ptx(npx+1)=xpto
246               pty(npx+1)=ypto
247             else
248               write(*,*) 'Too many X-points specified in selptx.inf: ',nptx
249               write(*,*) 'Check the specification or the nptxm parameter ',
250          ,                                                    'in selptx.F'
251               ex=.false.
252             end if
253             close(lun)
254           end if
255           go to 20
256      12   write(*,*) 'selptx: error opening the file selptx.inf'
257           ex=.false.
258           go to 20
259      14   write(*,*) 'selptx: error reading the data from selptx.inf'
260           ex=.false.
261      20   continue
262           if(.not.ex) then
263     cank>>>
264     *..Affichage des points ou le gradient s'annule.
265     
266     c-langue
267             if(sellan(1:8).eq.'francais') then
268               WRITE(6,25) (i,pointx(i),pointy(i), i=1, npxtot)
269        25     FORMAT ('     Voici la liste des points de gradient zero:'/
270          .        '     -------------------------------------------'/
271          .  '  #',T10,' R=',T25,' Z='/
272          .         (I3, T10, 1pe12.5, T25, 1pe12.5))
273               write(6,*)'  0   pour terminer la selection'
274               write(6,*)' -1   pour une configuration limiteur'
275               write(6,*)' Pour une configuration deflecteur, selectionnez',
276          .    ' le ou les points X.'
277     c       write(6,*)' Choisissez en donnant l''indice approprie.'
278             elseif(sellan(1:7).eq.'english') then
279               WRITE(6,225) (i,pointx(i),pointy(i), i=1, npxtot)
280       225     FORMAT ('       List of points where grad-psi vanishes:'/
281          .        '     -------------------------------------------'/
282          .  '  #',T10,' R=',T25,' Z='/
283          .         (I3, T10, 1pe12.5, T25, 1pe12.5))
284               write(6,*)'  0   to stop selection'
285               write(6,*)' -1   for a limiter configuration'
286               write(6,*)' For a divertor configuration, select X-point(s)'
287     c       write(6,*)' Choose option by typing the appropriate index.'
288             end if
289     
290     *..Indexation des points X.
291     
292        30   CONTINUE
293                READ(5,*) ipx
294     
295                IF (ipx.GT.0) THEN
296                   npx = npx + 1
297                   ptx(npx) = pointx(ipx)
298                   pty(npx) = pointy(ipx)
299                   iptx(npx) = ii(ipx)
300                   jptx(npx) = jj(ipx)
301                   if(npx.lt.2) GO TO 30
302                end if
303     
304     c
305             if(npx.eq.0 .and. ipx.eq.-1) then
306               npx=1
307               limcfg=1
308     c  N.B.: dans ce cas, le point X est defini comme etant le premier point
309     c      de contact avec un limiteur. Ce point est calcule dans limfnd.
310             else
311               limcfg=0
312             end if
313     
314     *..On identifie le point O.
315     
316             IF (npx.GT.0 .or. limcfg.ne.0) THEN
317     
318               if(sellan(1:8).eq.'francais') then
319                write(6,*)' Indiquez l''indice du point O.'
320               elseif(sellan(1:7).eq.'english') then
321                write(6,*)' Select index for O-point.'
322               end if
323                READ(5,*) ipo
324     
325                xpto = pointx(ipo)
326                ypto = pointy(ipo)
327     c  on met les coordonnees du point O a la fin des vecteurs ptx et pty.
328     c  ceci servira a distinguer le point X interieur d'un double nul decon-
329     c  necte a la fin de la routine sptris
330                ptx(npx+1)=xpto
331                pty(npx+1)=ypto
332     
333             end if
334     
335     cank-970701
336           end if
337     
338     *..On choisit de raccorder ou ne pas raccorder les points X.
339     
340           if (npx .gt. 1) then
341             if(lconn.ge.0) then
342               racord = lconn.gt.0
343             else
344               if(sellan(1:8).eq.'francais') then
345                 write(6,*)
346          .       ' Voulez-vous que les points X soient raccordes? (o/n)'
347               else if(sellan(1:7).eq.'english') then
348                 write(6,*) 'Do you wish to force connection ',
349          .        	      	      	      	'of the two X-points (y/n)?'
350               end if
351               read(5,35) rep
352        35     format (a)
353     
354               if (rep .EQ. 'n' .OR. rep .EQ. 'N') then
355                 racord = .false.
356               else
357                 racord = .true.
358               end if
359             end if
360     
361     *..On ordonne les points X du haut vers le bas.
362     
363              IF (pty(1) .LT. pty(2)) THEN
364     
365                 ptxtmp = ptx(1)
366                 ptx(1) = ptx(2)
367                 ptx(2) = ptxtmp
368     
369                 ptytmp = pty(1)
370                 pty(1) = pty(2)
371                 pty(2) = ptytmp
372     
373                 iptxtm = iptx(1)
374                 iptx(1) = iptx(2)
375                 iptx(2) = iptxtm
376     
377                 jptxtm = jptx(1)
378                 jptx(1) = jptx(2)
379                 jptx(2) = jptxtm
380     
381              end if
382     
383           ELSE
384              racord = .FALSE.
385           end if
386     
387     cnba  The 'endpag' call below closes the graphics window opened by 'cntour'.
388     cnba  L'appel a 'endpag' ci-dessous ferme la fenetre graphique ouverte par 
389     cnba  'cntour'.
390           call endpag
391     
392           RETURN
393           END
