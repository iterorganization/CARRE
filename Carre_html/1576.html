1           SUBROUTINE SORTIE(nsep,nreg,nptseg,npr,np1,deltp1,deltpn,deltr1,
2          .    deltrn,pntrat,tgarde,distxo,repart,xmail,ymail,
3          .    nx,ny,x,y,a00,a10,a01,a11,ptx,pty,npx,racord,numero,
4          .    fctpx)
5     c
6     c  version : 07.07.97 19:08
7     c
8     c======================================================================
9           IMPLICIT NONE
10     
11     *..  Cette sous-routine imprime les resultats dans un fichier de sortie:
12     *  maille.sor
13     
14     cank-970707: dimensions from the file
15     *  dimensions
16     #include <CARREDIM.F>
17     
18     *  arguments
19           INTEGER nsep,nreg,repart,nptseg(nsep),npr(nreg),np1(nreg),
20          .        nx,ny,npx,numero
21           REAL*8 pntrat,deltp1(nsep),deltpn(nsep),deltr1(nreg),deltrn(nreg),
22          .     xmail(npmamx,nrmamx,nreg),ymail(npmamx,nrmamx,nreg),
23          .     tgarde(4),distxo,x(nxmax),y(nymax),
24          .     a00(nxmax,nymax,3),a10(nxmax,nymax,3),
25          .     a01(nxmax,nymax,3),a11(nxmax,nymax,3),
26          .     ptx(npx+1),pty(npx+1),fctpx(npx+1)
27           logical racord
28     
29     *  variables en common
30     #include <COMRLX.F>
31     #include <COMLAN.F>
32     
33     *  variables locales
34           INTEGER i,j,isep,ireg,ii,jj
35           REAL*8 xx,yy,psimai,psidx,psidy
36     
37     *  procedures
38           INTEGER ifind
39           EXTERNAL ifind
40     c======================================================================
41     c.. i,j: indices
42     c.. isep: indice de separatrice.
43     c.. ireg: indice de region.
44     c.. ii,jj: indices d'identification en x et y du carre.
45     c.. numero: 1= impression des parametres pouvant servir a un essai
46     c           ulterieur. 2= impression des donnees de la maille.
47     c.. xx,yy: coordonnees du premier point d'une courbe de niveau.
48     c.. psimai: valeur de la fonction pour cette courbe de niveau.
49     c======================================================================
50     *..Creation du nouveau fichier.
51     
52           IF (numero .EQ. 1) THEN
53     
54     *..Impression des points X et O.
55     
56             if(sellan(1:8).eq.'francais') then
57               write(10,111)npx,(ptx(i),pty(i),fctpx(i),i=1,npx)
58     111       format(//' Nombre de points X:',i3/
59          .      t5,'x',t25,'y',t45,'psi'/(t4,1pe15.8,t24,1pe15.8,
60          .      t44,1pe15.8))
61              elseif(sellan(1:7).eq.'english') then
62               write(10,211)npx,(ptx(i),pty(i),fctpx(i),i=1,npx)
63     211       format(//' Number of X-points:',i3/
64          .      t5,'x',t25,'y',t45,'psi'/(t4,1pe15.8,t24,1pe15.8,
65          .      t44,1pe15.8))
66              endif
67              write(10,112)ptx(npx+1),pty(npx+1),fctpx(npx+1)
68     112      format(/' Point O:'/(t4,1pe15.8,t24,1pe15.8,t44,1pe15.8))
69              if(npx.gt.1) then
70                 if(racord) then
71                   if(sellan(1:8).eq.'francais') then
72                     write(10,113)' Les points X sont raccordes.'
73     113             format(a//)
74                   elseif(sellan(1:7).eq.'english') then
75                     write(10,113)' X-points are connected.'
76                   endif
77                 else
78                   if(sellan(1:8).eq.'francais') then
79                     write(10,113)' Les points X ne sont pas raccordes.'
80                   elseif(sellan(1:7).eq.'english') then
81                     write(10,113)' X-points are not connected.'
82                   endif
83                 endif
84              endif
85     c
86     *..Ecriture des parametres qui pouront servir au le prochain essai.
87              if(sellan(1:8).eq.'francais') then
88                write(10,*)'$parametres'
89              elseif(sellan(1:7).eq.'english') then
90                write(10,*)'$parameters'
91              endif
92              WRITE(10,100)repart
93       100    FORMAT('repart =',I2)
94     
95              WRITE(10,1015)distxo
96      1015    FORMAT('distxo =',F11.8)
97     
98              WRITE(10,101)pntrat
99       101    FORMAT('pntrat =',F11.8)
100     
101              DO 10 isep=1, nsep
102                 WRITE(10,102)isep,nptseg(isep),isep,deltp1(isep),isep,
103          .                deltpn(isep)
104       102       FORMAT('nptseg(',I1,') =',I3/'deltp1(',I1,') =',F10.7/
105          .          'deltpn(',I1,') =',F10.7)
106        10    CONTINUE
107     
108              DO 20 ireg=1, nreg
109                 WRITE(10,103)ireg,npr(ireg),ireg,deltr1(ireg),ireg,
110          .                deltrn(ireg)
111       103       FORMAT('npr(',I1,') =',I3/'deltr1(',I1,') =',F10.7/
112          .          'deltrn(',I1,') =',F10.7)
113        20    CONTINUE
114     
115              DO 25 i=1, 2
116                 WRITE(10,104)i,tgarde(i)
117       104       FORMAT('tgarde(',I1,') =',F8.5)
118        25    CONTINUE
119     
120              IF (nsep .GT. 3) THEN
121                 DO 26 i=3, 4
122                    WRITE(10,104)i,tgarde(i)
123        26       CONTINUE
124              ENDIF
125              write(10,115)'nrelax=',nrelax,'relax=',relax,'pasmin=',pasmin,
126          .     'rlcept=',rlcept
127      115     format(a,i5/(a,1pe11.4))
128     
129              if(sellan(1:8).eq.'francais') then
130                WRITE(10,105)
131       105      FORMAT('$fin')
132              elseif(sellan(1:7).eq.'english') then
133                write(10,205)
134       205      format('$end')
135              endif
136     
137     *..Ecriture des points de mailles.
138     
139           ELSE IF (numero .EQ. 2) THEN
140              write(10,114)
141     114      format(t2,'x(m)',t18,'y(m)',t34,'psi(SI)',t50,'dpsi/dx',t66,
142          .     'dpsi/dy')
143              write(10,*)'$maille'
144              WRITE(10,106)nreg
145       106    FORMAT('''nreg=''',I5)
146     
147              DO 30 ireg=1, nreg
148                 WRITE(10,107)ireg
149       107       FORMAT('''region:''',I5)
150                 WRITE(10,108)np1(ireg),npr(ireg)
151       108       FORMAT('''nppol=''',I5,'  ''nprad=''',I5)
152     
153                 DO 32 j=1, npr(ireg)
154                 do 31 i=1,np1(ireg)
155                    xx = xmail(i,j,ireg)
156                    yy = ymail(i,j,ireg)
157                    ii = ifind(xx,x,nx,1)
158                    jj = ifind(yy,y,ny,1)
159                    psimai = a00(ii,jj,1)+a10(ii,jj,1)*xx+a01(ii,jj,1)*yy
160          .                + a11(ii,jj,1)*xx*yy
161                    psidx = a00(ii,jj,2)+a10(ii,jj,2)*xx+a01(ii,jj,2)*yy
162          .                + a11(ii,jj,2)*xx*yy
163                    psidy = a00(ii,jj,3)+a10(ii,jj,3)*xx+a01(ii,jj,3)*yy
164          .                + a11(ii,jj,3)*xx*yy
165     
166                    WRITE(10,109)xmail(i,j,ireg),ymail(i,j,ireg),psimai,
167          .           psidx,psidy
168     c 109          FORMAT(1p5E16.9)   mis en commentaire le 13 juin 1994
169       109          FORMAT(1p5E16.8)
170     
171        31       continue
172        32       CONTINUE
173        30    CONTINUE
174     
175              if(sellan(1:8).eq.'francais') then
176                WRITE(10,110)
177       110      FORMAT('$fin')
178              elseif(sellan(1:7).eq.'english') then
179                write(10,210)
180       210      format('$end')
181              endif
182     
183           ENDIF
184     
185           RETURN
186           END
