2     ************************************************************************
3           SUBROUTINE COORD1(vx,vy,n,d,x,y,period)
4     ************************************************************************
5           IMPLICIT NONE
6     
7     *..Cette sous-routine renvoit les coordonnees x et y d'un point situe
8     *  sur une courbe v, et a une distance d du debut de celle-ci.
9     
10     *  arguments
11           INTEGER n
12           REAL*8 vx(n),vy(n),d,x,y,period
13     
14     *  variables locales
15           INTEGER i
16           REAL*8 dist,distot,eps,zero,dd
17           PARAMETER (eps=1.E-06,zero=0.)
18     
19     *  procedures
20           INTRINSIC SQRT
21     
22     c========================
23     c.. n   : nombre de points de la courbe.
24     c.. vx,vy: vecteurs des points de la courbe.
25     c.. d   : distance a laquelle il faut trouver x et y
26     c.. x,y : coordonnees du point situe sur la courbe a une distance d du
27     c         debut de celle-ci.
28     c.. period: periode du circuit si non nul
29     c.. dist: distance entre 2 points consecutifs de la courbe.
30     c.. distot: distance totale parcourue.
31     c========================
32     
33     *..Initialisation.
34     
35           dd=d
36           if(period.gt.zero) then
37             if(d.gt.period) then
38               dd=d-int(d/period)*period
39             elseif(d.lt.zero) then
40               dd=d-(int(d/period)-1)*period
41             endif
42           endif
43           distot=zero
44     
45     *..Boucle sur tous les segments de courbe.
46     
47           DO 10 i=1, n-1
48     
49              dist=SQRT((vx(i+1)-vx(i))**2 + (vy(i+1)-vy(i))**2)
50              distot=distot + dist
51     
52     *..On regarde si on a depasse la distance d.
53     
54              IF (distot .GE. dd) THEN
55                 x=vx(i+1) - ((distot-dd)/dist)*(vx(i+1)-vx(i))
56                 y=vy(i+1) - ((distot-dd)/dist)*(vy(i+1)-vy(i))
57     
58                 RETURN
59     
60              ENDIF
61     
62        10 CONTINUE
63     
64     *..Si on a rien trouve alors on verifie si la distance est la meme que
65     *  la longueur de la courbe. Si oui on prend le dernier point.
66     
67           IF (ABS(distot - dd) .LT. eps) THEN
68              x=vx(n)
69              y=vy(n)
70     
71              RETURN
72     
73           ENDIF
74     
75     *..Si la distance d est plus grande que la distance totale de la courbe,
76     *  alors il y a une erreur.
77     
78           PRINT *, 'distance plus grande que dist. totale de la courbe dans'
79          .  ,' COORD1'
80           print*,'distot=',distot
81           print*,'dd=',dd
82           call pltend
83           STOP
84           END
