2     ************************************************************************
3           integer function horair(xpto,ypto,x0,y0,xstruc,ystruc,npst,sens)
4     ************************************************************************
5           implicit none
6     c  Determination du sens de parcours "sens" sur une structure donnee,
7     c  a partir d'un point de depart x0, y0, pour que le trajet initial
8     c  soit dans le sens horaire (sens=-1) ou anti-horaire (sens=1) tel que
9     c  vu du point de reference xpto, ypto.
10     c  N.B.: la structure est supposee fermee.
11     c
12     c  arguments
13           integer npst,sens
14           real*8 xpto,ypto,x0,y0,xstruc(npst),ystruc(npst)
15     c
16     c  variables locales
17           integer ind,im,ip
18           real*8 v0x,v0y,norm,vmx,vmy,vpx,vpy
19     c
20     c  procedures
21           integer indsgm
22           external indsgm
23           intrinsic sqrt
24     c
25     c  calcul
26     c
27     c  1.   on trouve d'abord l'indice du segment sur lequel se trouve le
28     c       point de depart.
29           ind=indsgm(xstruc,ystruc,npst,x0,y0)
30     
31           if(x0.eq.xstruc(ind) .and. y0.eq.ystruc(ind)) then
32             if(ind.eq.1) then
33               im=npst-1
34               ip=ind+1
35             elseif(ind.eq.npst-1) then
36               im=ind-1
37               ip=1
38             else
39               im=ind-1
40               ip=ind+1
41             endif
42           elseif(x0.eq.xstruc(ind+1) .and. y0.eq.ystruc(ind+1)) then
43             if(ind.eq.npst-1) then
44               im=ind
45               ip=2
46             else
47               im=ind
48               ip=ind+2
49             endif
50           else
51             im=ind
52             ip=ind+1
53           endif
54           v0x=x0-xpto
55           v0y=y0-ypto
56     
57           vmx=xstruc(im)-x0
58           vmy=ystruc(im)-y0
59           norm=sqrt(vmx*vmx+vmy*vmy)
60           vmx=vmx/norm
61           vmy=vmy/norm
62     
63           vpx=xstruc(ip)-x0
64           vpy=ystruc(ip)-y0
65           norm=sqrt(vpx*vpx+vpy*vpy)
66           vpx=vpx/norm
67           vpy=vpy/norm
68     
69           if((v0x*vmy-v0y*vmx).gt.(v0x*vpy-v0y*vpx)) then
70             horair=-sens
71           else
72             horair=sens
73           endif
74     
75           return
76           end
