1           INTEGER FUNCTION drctio(xst,yst,npst,x1,y1,doug)
2     c
3     c  version : 30.06.98 17:18
4     c
5     c======================================================================
6           IMPLICIT NONE
7     
8     *..Determination de la direction anti-horaire (dir=+1) ou horaire (-1)
9     *  qui correspond a un deplacement vers la droite ou vers la gauche
10     *  le long d'une structure, a partir d'un point x,y
11     
12     *  arguments
13           INTEGER npst
14           REAL*8 xst(npst),yst(npst),x1,y1
15           CHARACTER*(*) doug
16     
17     *  variables locales
18           INTEGER ind,nrot,irot
19           REAL*8 sg,delta,xin,yin,xin1,yin1,zero,norm,pi,cs,sn
20           parameter(delta=1.e-4,zero=0.,nrot=10,pi=3.141592654)
21     
22     *  procedures
23           intrinsic sqrt,sin,cos
24           INTEGER indsgm
25           logical in
26           EXTERNAL indsgm,in
27          ,         ,trc_stk
28     
29     c<<<
30           integer n_call
31           logical l_dbg
32           data n_call /0/, l_dbg /.false./
33     c>>>
34     
35     c=========================
36     c.. xst,yst: tableaux des coordonnees des points de la structure.
37     c.. npst: nombre de points de la structure.
38     c.. x1,y1: point de depart sur la structure.
39     c.. doug: drote ou gauche
40     c.. ind: indice du segment de structure.
41     c=========================
42     *  calculs
43     
44           n_call=n_call+1
45     cc<<<
46     c      l_dbg=n_call.le.4
47     c      if(l_dbg) then
48     c        write(0,*) '==== Entering drctio: ',n_call
49     c        write(0,*) 'doug,npst,x1,y1: ',doug,' ',npst,x1,y1
50     c        write(0,'(5h xst:,1p,7e12.4/(5x,7e12.4))') xst
51     c        write(0,'(5h yst:,1p,7e12.4/(5x,7e12.4))') yst
52     c      end if
53     cc>>>
54     
55     cank-19980618
56           cs=cos((2.*pi)/nrot)
57           sn=sin((2.*pi)/nrot)
58     cank
59     
60           ind=indsgm(xst,yst,npst,x1,y1)
61     
62           IF (doug(1:1).EQ.'d' .OR. doug(1:1).EQ.'D') then
63              sg=+1
64           else
65              sg=-1
66           end if
67           if(y1.gt.zero) sg=-sg
68     cc<<<
69     c      if(l_dbg) then
70     c        write(0,*) 'ind,sg =',ind,sg
71     c      end if
72     cc>>>
73           IF (xst(ind).NE.x1 .or. yst(ind).ne.y1) then
74     c  on trouve d'abord les coordonnees d'un point interieur
75             xin=y1-yst(ind)
76             yin=xst(ind)-x1
77             norm=sqrt(xin*xin+yin*yin)
78             xin=delta*xin/norm
79             yin=delta*yin/norm
80     cc<<<
81     c        if(l_dbg) then
82     c          write(0,*) 'xin,yin,norm=',xin,yin,norm
83     c          write(0,'(1x,a4,4(5x,a6,3x))') 'irot','x1+xin','y1+yin',
84     c     ,                                            'xin','yin'
85     c        end if
86     cc>>>
87             do 2 irot=1,nrot
88     cc<<<
89     c          if(l_dbg) then
90     c            write(0,'(1x,i3,1x,1p,4e14.6)') irot,
91     c     ,                                      x1+xin,y1+yin,xin,yin
92     c          end if
93     cc>>>
94               if(in(x1+xin,y1+yin,xst,yst,npst)) then
95                 go to 4
96               else
97     cank-19980618
98     c            cs=cos((2.*pi*irot)/nrot)
99     c            sn=sin((2.*pi*irot)/nrot)
100     cank
101                 norm=xin
102                 xin=xin*cs-yin*sn
103                 yin=norm*sn+yin*cs
104               end if
105       2     continue
106     c        write(0,*)'on ne trouve pas de point interieur dans drctio'
107             write(0,*) 'Interior point not found in drctio ',
108          ,                                  '- an internal error in Carre?'
109             write(0,*) '*2: n_call,x1,y1=',n_call,x1,y1
110             write(0,'(5h xst:,1p,7e12.4/(5x,7e12.4))') xst
111             write(0,'(5h yst:,1p,7e12.4/(5x,7e12.4))') yst
112             call trc_stk
113             write(0,*) '==> Check whether the structures representing ',
114          ,                                         'the targets are closed'
115             call pltend
116             stop
117       4     continue
118     c  on teste maintenant l'orientation
119             if(sg*(xin*(y1-yst(ind))-yin*(x1-xst(ind))).gt.zero) then
120               drctio=1
121             else
122               drctio=-1
123             end if
124           else if(xst(ind+1).NE.x1 .or. yst(ind+1).ne.y1) then
125             xin1=xst(ind+1)-x1
126             yin1=yst(ind+1)-y1
127             norm=sqrt(xin1*xin1+yin1*yin1)
128             xin1=xin1/norm
129             yin1=yin1/norm
130             if(ind.eq.1) then
131               irot=npst-1
132             else
133               irot=ind-1
134             end if
135             xin=xst(irot)-x1
136             yin=yst(irot)-y1
137             norm=sqrt(xin*xin+yin*yin)
138             xin=xin/norm+xin1
139             yin=yin/norm+yin1
140             norm=sqrt(xin*xin+yin*yin)
141             xin=delta*xin/norm
142             yin=delta*yin/norm
143     
144             do 6 irot=1,nrot
145               if(in(x1+xin,y1+yin,xst,yst,npst)) then
146                 go to 8
147               else
148     cank-19980618
149     c            cs=cos((2.*pi*irot)/nrot)
150     c            sn=sin((2.*pi*irot)/nrot)
151     cank
152                 norm=xin
153                 xin=xin*cs-yin*sn
154                 yin=norm*sn+yin*cs
155               end if
156       6     continue
157     c        write(0,*)'on ne trouve pas de point interieur dans drctio'
158             write(0,*) 'Interior point not found in drctio ',
159          ,                                  '- an internal error in Carre?'
160             write(0,*) '*6: n_call,x1,y1=',n_call,x1,y1
161             write(0,'(5h xst:,1p,7e12.4/(5x,7e12.4))') xst
162             write(0,'(5h yst:,1p,7e12.4/(5x,7e12.4))') yst
163             call trc_stk
164             write(0,*) '==> Check whether the structures representing ',
165          ,                                         'the targets are closed'
166             call pltend
167             stop
168       8     continue
169     c  on teste maintenant l'orientation
170             if(sg*(xin*(y1-yst(ind))-yin*(x1-xst(ind))).gt.zero) then
171               drctio=1
172             else
173               drctio=-1
174             end if
175           else
176     c        PRINT*,'Segment degenere detecte dans drctio: fin du programme'
177             call pltend
178             STOP 'Degenerated segment detected in drctio: the program stops'
179           end if
180           RETURN
181           END
