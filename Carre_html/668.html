3           INTEGER FUNCTION indcrb(vx,vy,n,x,y,d)
4     ************************************************************************
5           IMPLICIT NONE
6     
7     *..  Cette fonction renvoie l'indice du point du debut du segment entre
8     *  2 points parametrises sur lequel se trouve le point x,y. On effectue
9     *  un produit scalaire entre les vecteurs allant du point a chaque
10     *  extremite du segment. Sur le segment de contact ou se trouve le
11     *  point, l'angle sera le plus grand. L'indice trouve est le plus grand
12     *  tel que la distance parcourue jusqu'a cet indice est inferieure ou
13     *  egale a la distance d
14     
15     *  arguments
16           INTEGER n
17           REAL*8 vx(n),vy(n),x,y,d
18     
19     *  variables locales
20           INTEGER i
21           REAL*8 mumin, mu, ax, ay, bx, by, dist, eps,dd
22           PARAMETER (eps=1.E-6)
23     
24     *  procedures
25           INTRINSIC SQRT
26     
27     c=========================
28     c.. n   : nombre  de points de la courbe.
29     c.. vx,vy: vecteurs des coordonnees des points de la courbe.
30     c.. x,y : point sur la courbe dont on recherche l'indice.
31     c.. ax,ay,bx,by: composantes x et y des vecteurs a et b.
32     c.. mu: angle entre les vecteurs a et b.
33     c=========================
34     
35     *..Initialisation.
36     
37           mumin = 1.
38           indcrb = 1
39     
40     *..Test pour savoir si le point x,y est egal au premier point parame-
41     *  trise.
42     
43           ax = vx(1) - x
44           ay = vy(1) - y
45     
46           dist = SQRT(ax*ax + ay*ay)
47           dd=sqrt((vx(2)-vx(1))**2+(vy(2)-vy(1))**2)
48     
49           IF (dist .LT. eps .and. dd.gt.d) THEN
50     
51              indcrb = 1
52              RETURN
53     
54           ENDIF
55     
56     *..Boucle sur tous les points parametrises.
57     
58           dd=0.
59           DO 10 i=1, n-1
60           dd=dd+sqrt((vx(i+1)-vx(i))**2+(vy(i+1)-vy(i))**2)
61           if(dd.ge.d) then
62     
63     *..Def. des 2 vecteurs a et b.
64     
65              ax = vx(i) - x
66              ay = vy(i) - y
67              bx = vx(i+1) - x
68              by = vy(i+1) - y
69     
70     *..Test pour voir si le point x,y tombe sur le point parametrise suivant
71     
72              dist = SQRT(bx*bx + by*by)
73              IF (dist .LT. eps) THEN
74                 IF (i .NE. n-1) THEN
75                    indcrb = i + 1
76                    RETURN
77                 ELSE
78                    indcrb = i
79                    RETURN
80                 ENDIF
81              ELSE
82     
83     *..On trouve le cosinus de l'angle entre les 2 vecteurs
84     
85                 mu = (((ax*bx) + (ay*by))
86          .                /SQRT(((ax**2)+(ay**2)) * ((bx**2)+(by**2))))
87     c
88     
89     *..L'indice de segment correspond au cosinus le plus petit
90     
91                 IF (mu .lt. mumin) THEN
92                    indcrb = i
93                    mumin = mu
94                 ENDIF
95     
96              ENDIF
97     
98           endif
99        10 CONTINUE
100     
101           RETURN
102           END
