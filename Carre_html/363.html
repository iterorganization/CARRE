2     ************************************************************************
3           LOGICAL FUNCTION cross(ind,xx,yy,xst,yst,n)
4     ************************************************************************
5           IMPLICIT NONE
6     
7     *..  Cette fonction verifie si un segment traverse un segment de
8     *  structure autre que celui sur lequel on est en train de marcher.
9     
10     *  arguments
11           INTEGER ind, n
12           REAL*8 xx(*),yy(*),xst(n),yst(n)
13     
14     *  variables locales
15           INTEGER i
16           REAL*8 mult1,mult2,determ
17     
18     c=========================
19     c.. xst,yst: tableaux des coordonnees des points de la structure.
20     c.. n  : nombre de points de la structure.
21     c.. xx,yy: tableaux des coordonnees contenant les 2 points du segment.
22     c.. ind: indice du segment de structure.
23     c.. determ: determinant de la matrice des deux equations.
24     c.. mult1: facteur multiplicatif du segment de courbe.
25     c.. mult2: facteur multiplicatif du segment de structure.
26     c=========================
27     
28     *..Boucle sur la structure.
29           DO 10 i=1, n-1
30     
31     *..Calcul du determinant de la matrice.
32     
33              determ = (-(xx(2) - xx(1))) * (yst(i+1) - yst(i)) +
34          .                   (yy(2) - yy(1)) * (xst(i+1) - xst(i))
35     
36     *..Si determinant non nul, alors il y a solution.
37     
38              IF (determ .NE. 0.) THEN
39     
40     *..Facteur multiplicatif du segment de courbe avec la methode de Cramer.
41     
42                 mult1 = ((-(xst(i)-xx(1))) * (yst(i+1)-yst(i)) +
43          .              (yst(i)-yy(1)) * (xst(i+1)-xst(i)))/determ
44     
45     *..Pour avoir intersection, il faut que mult1 soit entre 0 et 1
46     
47                 IF ((mult1.GT.0.).AND.(mult1.LT.1.)) THEN
48     
49     *..Fact. mult. du segment de structure.
50     
51                    mult2= ((xx(2)-xx(1)) * (yst(i)-yy(1)) -
52          .                (yy(2)-yy(1)) * (xst(i)-xx(1)))/determ
53     
54     *..Intersection si mult2 entre 0 et 1
55     
56                    IF ((mult2.GT.0.).AND.(mult2.LT.1.)) THEN
57     
58     *..On verifie si le segment de structure traversee est different de
59     *  celui sur lequel on marche.
60     
61                       IF (i .NE. ind) THEN
62     
63                          cross = .TRUE.
64                          RETURN
65     
66                       ENDIF
67     
68                    ENDIF
69     
70                 ENDIF
71     
72              ENDIF
73     
74        10 CONTINUE
75     
76           cross = .FALSE.
77     
78           RETURN
79           END
