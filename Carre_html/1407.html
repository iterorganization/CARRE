3           subroutine plurap(tx1,ty1,n1,tx2,ty2,n2,x0,y0,ind0)
4     ************************************************************************
5           implicit none
6     c  Determination des coordonnees (x0,y0) et de l'indice de segment ind0
7     c  du point de la structure parametrisee par [tx1, ty1, n1], qui est
8     c  le plus rapproche de la courbe parametrisee par [tx2, ty2,n2].
9     c  N.B.: On suppose qu'il n'y a pas intersection entre les deux courbes.
10     c
11     c  arguments
12           integer n1,n2,ind0
13           real*8 tx1(n1),ty1(n1),tx2(n2),ty2(n2),x0,y0
14     c
15     c  variables locales
16           integer i1,i2
17           real*8 h,hmin,x,y,x1,y1,x2,y2,base,zero,un
18           parameter(zero=0.,un=1.)
19     c
20     c  procedures
21           intrinsic abs,sqrt
22     c
23     c  calcul
24     c
25     c  0.   initialisation
26           hmin=sqrt((tx1(1)-tx2(1))**2+(ty1(1)-ty2(1))**2)
27           x0=tx1(1)
28           y0=ty1(1)
29           ind0=1
30     c
31     c  1.   boucle sur tous les points de la courbe parametrisee et, pour
32     c       chaque point, boucle sur chaque segment de la structure.
33           do i2=1,n2
34             x=tx2(i2)
35             y=ty2(i2)
36             do i1=1,n1-1
37               x1=tx1(i1)
38               y1=ty1(i1)
39               x2=tx1(i1+1)
40               y2=ty1(i1+1)
41               base=((x-x1)*(x2-x1)+(y-y1)*(y2-y1))
42          .      /((x2-x1)*(x2-x1)+(y2-y1)*(y2-y1))
43     c***
44     c         print*,'i2, i1=',i2,i1
45     c         print*,' x,  y=',x,y
46     c         print*,'x1, y1=',x1,y1
47     c         print*,'x2, y2=',x2,y2
48     c         print*,'base=',base
49     c***
50               if(base.gt.un) then
51                 h=sqrt((x-x2)*(x-x2)+(y-y2)*(y-y2))
52                 if(h.lt.hmin) then
53                   hmin=h
54                   x0=x2
55                   y0=y2
56                   ind0=i1
57                 endif
58               elseif(base.lt.zero) then
59                 h=sqrt((x-x1)*(x-x1)+(y-y1)*(y-y1))
60                 if(h.lt.hmin) then
61                   hmin=h
62                   x0=x1
63                   y0=y1
64                   ind0=i1
65                 endif
66               else
67                 h=abs((x-x1)*(y2-y1)-(y-y1)*(x2-x1))
68          .        /sqrt((x2-x1)*(x2-x1)+(y2-y1)*(y2-y1))
69                 if(h.lt.hmin) then
70                   hmin=h
71                   x0=x1+base*(x2-x1)
72                   y0=y1+base*(y2-y1)
73                   ind0=i1
74                 endif
75               endif
76             enddo
77           enddo
78     c
79     c  2.   boucle sur tous les points de la structure et, pour chaque point,
80     c       boucle sur chaque segment de la courbe parametrisee.
81           do i1=1,n1
82             x=tx1(i1)
83             y=ty1(i1)
84             do i2=1,n2-1
85               x1=tx2(i2)
86               y1=ty2(i2)
87               x2=tx2(i2+1)
88               y2=ty2(i2+1)
89               base=((x-x1)*(x2-x1)+(y-y1)*(y2-y1))
90          .      /((x2-x1)*(x2-x1)+(y2-y1)*(y2-y1))
91     c***
92     c         print*,'i1, i2=',i1,i2
93     c         print*,' x,  y=',x,y
94     c         print*,'x1, y1=',x1,y1
95     c         print*,'x2, y2=',x2,y2
96     c         print*,'base=',base
97     c***
98               if(base.gt.un) then
99                 h=sqrt((x-x2)*(x-x2)+(y-y2)*(y-y2))
100                 if(h.lt.hmin) then
101                   hmin=h
102                   x0=x
103                   y0=y
104                   ind0=min(i1,n1-1)
105                 endif
106               elseif(base.lt.zero) then
107                 h=sqrt((x-x1)*(x-x1)+(y-y1)*(y-y1))
108                 if(h.lt.hmin) then
109                   hmin=h
110                   x0=x
111                   y0=y
112                   ind0=min(i1,n1-1)
113                 endif
114               else
115                 h=abs((x-x1)*(y2-y1)-(y-y1)*(x2-x1))
116          .        /sqrt((x2-x1)*(x2-x1)+(y2-y1)*(y2-y1))
117                 if(h.lt.hmin) then
118                   hmin=h
119                   x0=x
120                   y0=y
121                   ind0=min(i1,n1-1)
122                 endif
123               endif
124             enddo
125           enddo
126           return
127           end
