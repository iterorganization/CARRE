1           SUBROUTINE RAPPEL(nptseg,deltp1,deltpn,repart,npr,deltr1,deltrn,
2          .             pntrat,tgarde,lg,difpsi,distnv,nreg,nsep,npx,
3          .             dpmin,dpmax,drmin,drmax,distxo,iusor,correct)
4     c
5     c  version : 07.07.97 18:37
6     c
7     c======================================================================
8           IMPLICIT NONE
9     
10     *..  Cette sous-routine imprime a l'ecran un tableau contenant tous les
11     *  parametres choisis, ainsi que quelques resultats intermediaires.
12     *  nouvelle valeur desiree.
13     
14     cank-970707: dimensions from the file
15     *  dimensions
16     #include <CARREDIM.F>
17     
18     *  arguments
19           INTEGER nptseg(*),repart,npr(*),nreg,nsep,npx,iusor
20           REAL*8 deltp1(*),deltpn(*),deltr1(*),deltrn(*),pntrat,lg(*),
21          .       distnv(5,*),difpsi,dpmin(*),dpmax(*),drmin(*),drmax(*),
22          .       tgarde(4),distxo
23           LOGICAL correct
24     
25     *  variables en common
26     
27     #include <COMRLX.F>
28     #include <COMLAN.F>
29     
30     *  variables locales
31           INTEGER i
32     c======================================================================
33     *.. Impression des donnees se rapportant aux separatrices.
34     
35           if(sellan(1:8).eq.'francais') then
36             write(iusor,105) 'SEPARATRICES:'
37       105   FORMAT(//T10,A)
38           elseif(sellan(1:7).eq.'english') then
39             write(iusor,205) 'SEPARATRICES:'
40      205    FORMAT(//T10,A)
41           endif
42             write(iusor,110)'#','nptseg(i)','lg(i)','deltp1(i)','deltpn(i)',
43          .          'dpmin','dpmax'
44       110 FORMAT(/T2,A1,T6,A9,T18,A5,T29,A9,T41,A9,T54,A6,T66,A6/75('='))
45     
46           DO 12 i=1, nsep
47              write(iusor,114)i,nptseg(i),lg(i),deltp1(i),deltpn(i),dpmin(i),
48          .             dpmax(i)
49       114    FORMAT(T2,I1,T8,I3,T15,1p5E12.4)
50        12 CONTINUE
51     
52     *.. Impression des donnees se rapportant aux directions radiales de
53     *   chaque region.
54     
55           if(sellan(1:8).eq.'francais') then
56             write(iusor,120) 'DIRECTION RADIALE POUR CHAQUE REGION:'
57       120   FORMAT(//T10,A)
58           elseif(sellan(1:7).eq.'english') then
59             write(iusor,120) 'RADIAL DISTRIBUTIONS FOR EACH REGION:'
60           endif
61     
62           IF (repart .EQ. 1) THEN
63             if(sellan(1:8).eq.'francais') then
64              write(iusor,121) 'Repartition selon la distance:',
65          .     ' repart=',repart
66             elseif(sellan(1:7).eq.'english') then
67              write(iusor,221) 'Distribution in displacement along plates:',
68          .     ' repart=',repart
69             endif
70       121    FORMAT(T14,A,i5)
71       221    FORMAT(T14,2A,i5)
72     
73              write(iusor,122)'region','npr(i)','distance','deltr1(i)',
74          .     'deltrn(i)','drmin','drmax'
75       122    FORMAT(/T2,A6,T9,A6,T17,A8,T29,A9,T41,A9,T54,A6,T66,A6/
76          .            75('='))
77     
78           ELSE IF (repart .EQ. 2) THEN
79             if(sellan(1:8).eq.'francais') then
80               write(iusor,121) 'Repartition selon psi: repart=',repart
81               write(iusor,124)'region','npr(i)','difference','deltr1(i)',
82          .      'deltrn(i)','drmin','drmax'
83       124     FORMAT(/T2,A6,T9,A6,T17,A10,T29,A9,T41,A9,T54,A6,T66,A6/
84          .            75('='))
85             elseif(sellan(1:7).eq.'english') then
86               write(iusor,121) 'Distribution in psi: repart=',repart
87               write(iusor,124)'region','npr(i)','  width   ','deltr1(i)',
88          .      'deltrn(i)','drmin','drmax'
89             endif
90           ENDIF
91     
92           DO 26 i=1, nreg-1
93              write(iusor,128)i,npr(i),distnv(repart,i),deltr1(i),deltrn(i),
94          .            drmin(i),drmax(i)
95       128    FORMAT(T2,I1,T8,I3,T15,1p5E12.4)
96        26 CONTINUE
97     
98     *.. Impression des donnees se rapportant a la region centrale.
99     
100           if(sellan(1:8).eq.'francais') then
101             write(iusor,134) nreg,distxo
102       134   FORMAT(//T10,'REGION CENTRALE: region i=',T37,I1,T43,
103          .       'pntrat max.=',F11.8)
104           elseif(sellan(1:7).eq.'english') then
105             write(iusor,234) nreg,distxo
106       234   FORMAT(//T10,'CENTRAL REGION: region i=',T37,I1,T43,
107          .       'pntrat max.=',F11.8)
108           endif
109     
110           IF (repart .EQ. 1) THEN
111              write(iusor,138)'pntrat','npr(i)','deltr1(i)','deltrn(i)',
112          .            'drmin','drmax'
113       138    FORMAT(/T2,A6,T9,A6,T17,A9,T29,A9,T41,A6,T54,A6/75('='))
114     
115              write(iusor,140)pntrat,npr(nreg),deltr1(i),deltrn(i),drmin(i),
116          .            drmax(i)
117       140    FORMAT(/T2,F6.3,T9,I3,T15,1p4E12.4)
118     
119           ELSE IF (repart .EQ. 2) THEN
120     
121              if(sellan(1:8).eq.'francais') then
122                write(iusor,148)'pntrat','npr(i)','difference','deltr1(i)',
123          .       'deltrn(i)','drmin','drmax'
124              elseif(sellan(1:7).eq.'english') then
125                write(iusor,148)'pntrat','npr(i)','  width','deltr1(i)',
126          .       'deltrn(i)','drmin','drmax'
127              endif
128       148    FORMAT(/T2,A6,T9,A6,T17,A,T29,A9,T41,A9,T54,A6,T66,A6/
129          .            75('='))
130     
131              write(iusor,150)pntrat,npr(nreg),difpsi,deltr1(nreg),
132          .      deltrn(nreg),drmin(nreg),drmax(nreg)
133       150    FORMAT(T2,F6.3,T9,I3,T15,1p5E12.4)
134     
135           ENDIF
136     
137     *.. Impression des donnees se rapportant a la longueur de garde de
138     *   chaque plaque de deflecteur.
139     
140           if(sellan(1:8).eq.'francais') then
141             write(iusor,152)
142          .  'LONGUEUR DE GARDE DE CHAQUE PLAQUE DE DEFLECTEUR:'
143       152   FORMAT(//T10,A)
144           elseif(sellan(1:7).eq.'english') then
145             write(iusor,152)'GUARD LENGTH FOR EACH DIVERTOR PLATE:'
146           endif
147     
148           IF (nsep .EQ. 1 .or. nsep .EQ. 3) THEN
149     
150              write(iusor,154) 'tgarde(1)','tgarde(2)'
151       154    FORMAT(/T2,A9,T16,A9/75('='))
152     
153              write(iusor,156) tgarde(1),tgarde(2)
154       156    FORMAT(T2,F8.5,T17,F8.5)
155     
156           ELSE
157     
158              write(iusor,158)'tgarde(1)','tgarde(2)','tgarde(3)','tgarde(4)'
159       158    FORMAT(/T2,A9,T16,A9,T31,A9,T46,A9/75('='))
160     
161              write(iusor,160) tgarde(1),tgarde(2),tgarde(3),tgarde(4)
162       160    FORMAT(T2,F8.5,T17,F8.5,T32,F8.5,T47,F8.5)
163           ENDIF
164     
165     c  Impression des parametres servant au calcul de la maille orthogonale
166     c  par relaxation
167           if(sellan(1:8).eq.'francais') then
168             write(iusor,162)
169      162    format(//t10,
170          .    'PARAMETRES SERVANT AU CALCUL DE LA MAILLE PAR RELAXATION:'//
171          .    75('=')/
172          .    'nrelax',t15,'relax',t30,'pasmin',t45,'rlcept')
173           elseif(sellan(1:7).eq.'english') then
174             write(iusor,262)
175      262    format(//t10,
176          .    'RELAXATION PARAMETERS USED TO CONSTRUCT THE MESH:'//
177          .    75('=')/
178          .    'nrelax',t15,'relax',t30,'pasmin',t45,'rlcept')
179           endif
180           write(iusor,164)nrelax,relax,pasmin,rlcept
181      164  format(t2,i5,t15,f6.3,t30,1pe10.3,t45,1pe10.3)
182     
183     cxpb  Verifier que les valeurs presentees sont correctes
184     
185           correct = .true.
186           do i = 1, nsep
187             if (dpmin(i)*dpmax(i).le.0.0 .or.
188          .      dpmin(i)*lg(i).le.0.0 .or.
189          .      dpmax(i)*lg(i).le.0.0) then
190               correct = .false.
191               if(sellan(1:8).eq.'francais') then
192                 WRITE(6,281) i, dpmin(i), dpmax(i), lg(i)
193       281       FORMAT(//T2,'Donnees invalides pour le segment',i2,'!'//
194          .   T2,
195          .   'Les nombres dpmin, dpmax et lg doivent avoir le meme signe.'//
196          .   T2,'Modifiez deltp1 et deltpn en consequence.'//
197          .   T2,'dpmin,dpmax,lg = ',1p,3(1e11.4,1x))
198               elseif(sellan(1:7).eq.'english') then
199                 WRITE(6,280) i, dpmin(i), dpmax(i), lg(i)
200       280       FORMAT(//T2,'Invalid data for segment',i2,'!'//
201          .   T2,'The numbers dpmin, dpmax and lg must have the same sign.'//
202          .   T2,'Modify deltp1 and deltpn accordingly.'//
203          .   T2,'dpmin,dpmax,lg = ',1p,3(1e11.4,1x))
204               endif
205             endif 
206             if (nptseg(i).le.2) then
207               correct=.false.
208               if(sellan(1:8).eq.'francais') then
209                 write(6,2811) i
210      2811       format(//T2,'Donnees invalides pour le segment',i2,'!'//
211          .        T2,'Il faut plus de deux points sur ce segment!')
212               elseif(sellan(1:7).eq.'english') then
213                 write(6,2810) i
214      2810       format(//T2,'Invalid data for segment',i2,'!'//
215          .        T2,'You need more than 2 points on this segment!')
216               endif
217             endif
218             if (abs(dpmin(i)).lt.pasmin .or.
219          .      abs(dpmax(i)).lt.pasmin) then
220               correct = .false.
221               if(sellan(1:8).eq.'francais') then
222                 WRITE(6,2821) i, dpmin(i), dpmax(i), pasmin
223      2821       FORMAT(//T2,'Donnees invalides pour le segment',i2,'!'//
224          .   T2,
225          .   'Les nombres dpmin et dpmax doivent etre superieurs a pasmin'//
226          .   ' en valeur absolue.'//
227          .   T2,'Modifiez deltp1, deltpn or pasmin en consequence.'//
228          .   T2,'dpmin,dpmax,pasmin = ',1p,3(1e11.4,1x))
229               elseif(sellan(1:7).eq.'english') then
230                 WRITE(6,2820) i, dpmin(i), dpmax(i), pasmin
231      2820       FORMAT(//T2,'Invalid data for segment',i2,'!'//
232          .   T2,'The numbers dpmin and dpmax must be larger than pasmin'//
233          .      ' in absolute value.'//
234          .   T2,'Modify deltp1, deltpn or pasmin accordingly.'//
235          .   T2,'dpmin,dpmax,pasmin = ',1p,3(1e11.4,1x))
236               endif
237             endif
238           enddo
239           do i = 1, nreg-1
240             if (drmin(i)*drmax(i).le.0.0 .or.
241          .      drmin(i)*distnv(repart,i).le.0.0 .or.
242          .      drmax(i)*distnv(repart,i).le.0.0) then
243               correct = .false.
244               if(sellan(1:8).eq.'francais') then
245                 WRITE(6,291) i, drmin(i), drmax(i), distnv(repart,i)
246       291       FORMAT(//T2,'Donnees invalides pour la region',i2,'!'//
247          .   T2,'Les nombres drmin, drmax et difference doivent ',
248          .   'avoir le meme signe.'//
249          .   T2,'Modifiez deltr1 et deltrn en consequence.'//
250          .   T2,'drmin,drmax,difference = ',1p,3(1e11.4,1x))
251               elseif(sellan(1:7).eq.'english') then
252                 WRITE(6,290) i, drmin(i), drmax(i), distnv(repart,i)
253       290       FORMAT(//T2,'Invalid data for region',i2,'!'//
254          .   T2,
255          .   'The numbers drmin, drmax and width must have the same sign.'//
256          .   T2,'Modify deltr1 and deltrn accordingly.'//
257          .   T2,'drmin,drmax,width = ',1p,3(1e11.4,1x))
258               endif
259             endif
260           enddo
261           i = nreg
262           if (drmin(i)*drmax(i).le.0.0 .or.
263          .    drmin(i)*difpsi.le.0.0 .or.
264          .    drmax(i)*difpsi.le.0.0) then
265             correct = .false.
266             if(sellan(1:8).eq.'francais') then
267               WRITE(6,2911) i, drmin(i), drmax(i), difpsi
268      2911     FORMAT(//T2,'Donnees invalides pour la region',i2,'!'//
269          .   T2,'Les nombres drmin, drmax et difference doivent ',
270          .   'avoir le meme signe.'//
271          .   T2,'Modifiez deltr1 et deltrn en consequence.'//
272          .   T2,'drmin,drmax,difference = ',1p,3(1e11.4,1x))
273             elseif(sellan(1:7).eq.'english') then
274               WRITE(6,2900) i, drmin(i), drmax(i), difpsi
275      2900     FORMAT(//'Invalid data for region',i2,'!'//
276          .   T2,
277          .   'The numbers drmin, drmax and width must have the same sign.'//
278          .   T2,'Modify deltr1 and deltrn accordingly.'//
279          .   T2,'drmin,drmax,width = ',1p,3(1e11.4,1x))
280             endif
281           endif
282           do i = 1, 2*max(1,npx)
283             if (tgarde(i).lt.0.0) then
284               correct = .false.
285               if(sellan(1:8).eq.'francais') then
286                 WRITE(6,3011) i, tgarde(i)
287      3011       FORMAT(//T2,'Donnees invalides pour la plaque',i2,'!'//
288          .     T2,'La longueur tgarde doit etre positive ou nulle.'//
289          .     T2,'Modifiez tgarde en consequence.'//
290          .     T2,'tgarde = ',1pe11.4)
291               elseif(sellan(1:7).eq.'english') then
292                 WRITE(6,3000) i, tgarde(i)
293      3000       FORMAT(//'Invalid data for plate',i2,'!'//
294          .     T2,'The length tgarde must be positive or zero.'//
295          .     T2,'Modify tgarde accordingly.'//
296          .     T2,'tgarde = ',1pe11.4)
297               endif
298             endif
299           enddo
300           if (nrelax.lt.1 .or. relax.le.0.0 .or. relax.gt.1.0 .or. 
301          &    pasmin.le.0.0 .or. rlcept.le.0.0) then
302             correct = .false.
303             if(sellan(1:8).eq.'francais') then
304               WRITE(6,3111) nrelax, relax, pasmin, rlcept
305      3111     FORMAT(//
306          .   T2,'Donnees invalides pour l''algorithme de relaxation!'//
307          .   T2,'Il faut au moins une iteration.'//
308          .   T2,'Les parametres relax, pasmin, et rlcept ',
309          .      'doivent etre positifs.'//
310          .   T2,'Le parametre relax ne doit pas exceder 1.'//
311          .   T2,'Modifiez-les en consequence.'//
312          .   T2,'nrelax, relax, pasmin, rlcept = ',i6,1p,3(1e11.4,1x))
313             elseif(sellan(1:7).eq.'english') then
314               WRITE(6,3100) nrelax, relax, pasmin, rlcept
315      3100     FORMAT(//T2,'Invalid data for relaxation algorithm!'//
316          .   T2,'The length tgarde must be positive or zero.'//
317          .   T2,'At least one iteration is needed.'//
318          .   T2,'The relax, pasmin, and rlcept parameters ',
319          .      'must be positive.'//
320          .   T2,'The relax parameter must not exceed 1.'//
321          .   T2,'Modify these accordingly.'//
322          .   T2,'nrelax, relax, pasmin, rlcept = ',i6,1p,3(1e11.4,1x))
323             endif
324           endif
325     
326           RETURN
327           END
