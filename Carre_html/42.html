1           SUBROUTINE ARGSEP(npx,ptx,pty,fctpx,separx,separy,indplq,nptot,
2          .                 npnimx,ptsep,racord,ptxint,ypto,nbdef,inddef)
3     c
4     c  version : 23.06.98 20:28
5     c
6     c======================================================================
7     cank -- The comments are translated from French, sorry for errors!
8     
9     c*** This sub-routine arranges the separatrices in the required way
10     c*** and indexes them in correct order
11     
12     *..  Cette sous-routine arrange les separatrices de la facon demandee
13     *  et les ordonnes selon un ordre precis.
14     c======================================================================
15           IMPLICIT NONE
16     
17     *  arguments
18           INTEGER npnimx,npx,indplq(4,npx),nptot(4,npx),ptsep(4,npx),
19          ,                                       ptxint,nbdef,inddef(nbdef)
20           REAL*8 ptx(npx),pty(npx),separx(npnimx,4,npx),
21          .       separy(npnimx,4,npx),fctpx(npx+1),ypto
22           LOGICAL racord
23     
24     *  variables locales
25           INTEGER kkmax
26           PARAMETER (kkmax=10)
27           real*8 wx(4),wy(4),zero
28           parameter(zero=0.)
29           INTEGER ipx, ipx2, j, i, k0, kk(kkmax), kkp, k, ind,
30          ,        i1, i2, ipi, j1, j2, k1, k2, l1, l2
31     
32     *  procedures
33           INTEGER coinci
34           real*8 aire
35           LOGICAL tsplaq
36           INTRINSIC MOD, ABS
37           EXTERNAL coinci,tsplaq,aire
38     
39     c======================================================================
40     c*** Input
41     c.. npx   : number of the X-points
42     c.. ptx,pty: x and y coordinates of the X-points
43     c.. fctpx: the psi values at each X- or O-point
44     c.. separx,separy: coordinates of the points of the parametrised
45     c                  separatrices
46     c                  (point index, separatrix index, X-point index)
47     c.. nptot : number of parametrisation points for each separatrix
48     c           (separatrix index, X-point index)
49     c.. npnimx: maximum number of tracing points on a curve
50     c.. racord: determines whether the X-points are connected
51     c.. ypto  : vertical coordinate of the magnetic axis (the O-point)
52     c.. nbdef : number of divertor targets
53     c.. inddef: index of the target structures
54     c
55     c*** Output
56     c.. ptsep : separatrix pointer, used for indexation
57     c           (separatrix index, X-point index)
58     c.. ptxint: index of the X-point which is inner with respect to the
59     c           other in case of disconnected double-null
60     c.. indplq: structure indices; 0 if it is not a divertor plate
61     c           (separatrix index, X-point index)
62     c
63     c*** Internal variables
64     c.. ind,i,j: indices
65     c.. k0    : counter
66     c.. kkmax : maximum number which the counter can reach
67     c.. kk    : pointers of separatrices having to be copied in ptsep
68     c.. kkp   : separatrix pointer
69     c.. k     : index of the point where 2 separatrices coincide
70     c.. ipx,ipx2: indices of each X-point
71     c======================================================================
72     cc<<<
73     c      write(0,*) '=== Entering argsep. npx,racord=',npx,racord
74     c      write(0,*) 'nbdef,inddef: ',nbdef,inddef
75     c      write(0,*) 'nptot : ',nptot
76     c      write(0,*)
77     c      write(0,'(1x,a16,1p,4e12.4)') 'separx(i,1)',
78     c     ,                     (separx(nptot(i,1),i,1),i=1,4)
79     c      write(0,'(1x,a16,1p,4e12.4)') 'separy(i,1)',
80     c     ,                     (separy(nptot(i,1),i,1),i=1,4)
81     c      write(0,*)
82     c      write(0,'(1x,a16,1p,4e12.4)') 'separx(i,2)',
83     c     ,                     (separx(nptot(i,2),i,2),i=1,4)
84     c      write(0,'(1x,a16,1p,4e12.4)') 'separy(i,2)',
85     c     ,                        (separy(nptot(i,2),i,2),i=1,4)
86     c      write(0,*)
87     cc>>>
88     *..Initialisation.
89     
90           ptxint = 0
91     
92     *..To have angles of 90 degrees at the X-points, one must drop the
93     *  second point from each separatrix
94     
95           do 1 i=1,npx
96              do 2 j=1, 4
97                 do 3 k=2, nptot(j,i)-1
98                    separx(k,j,i) = separx(k+1,j,i)
99                    separy(k,j,i) = separy(k+1,j,i)
100         3       CONTINUE
101                 nptot(j,i) = nptot(j,i) - 1
102         2    CONTINUE
103         1 CONTINUE
104     cc<<<
105     c      write(0,*) '** after 1 continue'
106     c      write(0,*) 'nptot : ',nptot
107     c      write(0,'(13h indplq in : ,8i5)') indplq
108     cc>>>
109     
110     *..Find the separatrices which go directly to a target
111           do 10 ipx=1, npx
112     cc<<<
113     c        write(0,*) '** do 10  ipx = ',ipx
114     cc>>>
115             IF (npx .EQ. 1) then
116               do 14 j=1, 4
117                 IF (.NOT.(tsplaq(separx(1,j,ipx),separy(1,j,ipx),
118          .           nptot(j,ipx),npx,ptx(ipx),pty(ipx)))) then
119                   indplq(j,ipx) = 0
120                 end if
121        14     CONTINUE
122             else
123               ipx2 = MOD(ipx,2) + 1
124               do 15 j=1, 4
125                 IF (.NOT.(tsplaq(separx(1,j,ipx),separy(1,j,ipx),
126          .          nptot(j,ipx),npx,ptx(ipx2),pty(ipx2)))) then
127                   indplq(j,ipx) = 0
128                 end if
129        15     CONTINUE
130             end if
131     cc<<<
132     c        write(0,'(13h indplq out: ,8i5)') indplq
133     cc>>>
134             k0 = 0
135             do 20 j=1, 4
136               IF (indplq(j,ipx) .NE. 0) then
137                 k0 = k0 + 1
138                 kk(k0) = j
139               end if
140        20   CONTINUE
141     cc<<<
142     c        write(0,*) 'ipx,k0,kk : ',ipx,k0,' ',kk
143     cc>>>
144     *..Check whether there are exactly 2 separatrices which go to the plate
145             IF (npx .EQ. 1 .AND. k0 .NE. 2) then
146               PRINT *, 'Error in argsep. k0=', k0
147               call pltend
148               STOP
149             end if
150     *..Index the separatrices which go to the targets (right=1, left=2)
151             wx(1)=separx(1,kk(1),ipx)
152             wy(1)=separy(1,kk(1),ipx)
153             wx(2)=separx(nptot(kk(1),ipx),kk(1),ipx)
154             wy(2)=separy(nptot(kk(1),ipx),kk(1),ipx)
155             wx(3)=separx(nptot(kk(2),ipx),kk(2),ipx)
156             wy(3)=separy(nptot(kk(2),ipx),kk(2),ipx)
157             wx(4)=wx(1)
158             wy(4)=wy(1)
159     cc<<<
160     c        write(0,'(17h ypto,pty,aire = ,1p,4e12.4)') ypto,pty,
161     c     ,                                                    aire(wx,wy,4)
162     c        write(0,'(5h wx: ,1p,4e12.4)') wx
163     c        write(0,'(5h wy: ,1p,4e12.4)') wy
164     cc>>>
165             if((pty(ipx)-ypto)*aire(wx,wy,4).ge.zero) then
166               ptsep(1,ipx)=kk(1)
167               ptsep(2,ipx)=kk(2)
168             else
169               ptsep(1,ipx)=kk(2)
170               ptsep(2,ipx)=kk(1)
171             end if
172        10 CONTINUE
173     cc<<<
174     c      write(0,*) '** After 10 continue. npx = ',npx
175     c      write(0,*) 'ptsep: ',ptsep
176     cc>>>
177     *..First case: ONE X-point
178           IF (npx .EQ. 1) then
179     c----------------------------------------------------------------------
180     cc<<<
181     c        write(0,*) '-- Single null'
182     cc>>>
183     *..Initialisation
184              ipx = 1
185              k0 = 0
186     *..Identify the separatrices which do not go to the targets
187              do 25 j=1, 4
188                 IF (indplq(j,ipx) .EQ. 0) then
189                    k0 = k0 + 1
190                    kk(k0) = j
191                 end if
192        25    CONTINUE
193     *..Index the separatrices which do not go to the targets
194              wx(1)=separx(1,kk(1),ipx)
195              wy(1)=separy(1,kk(1),ipx)
196              wx(2)=separx(2,kk(1),ipx)
197              wy(2)=separy(2,kk(1),ipx)
198              wx(3)=separx(2,kk(2),ipx)
199              wy(3)=separy(2,kk(2),ipx)
200              wx(4)=wx(1)
201              wy(4)=wy(1)
202              if((pty(ipx)-ypto)*aire(wx,wy,4).le.zero) then
203                ptsep(3,ipx) = kk(1)
204                kkp = kk(2)
205              else
206                ptsep(3,ipx) = kk(2)
207                kkp = kk(1)
208              end if
209     *..Parametrise the third separatrix (which meets itself)
210              k = coinci(separx(1,ptsep(3,ipx),ipx),separy(1,ptsep(3,ipx),
211          .              ipx),nptot(ptsep(3,ipx),ipx),separx(2,kkp,ipx),
212          .              separy(2,kkp,ipx)) + 1
213              nptot(ptsep(3,ipx),ipx) = k
214              separx(k,ptsep(3,ipx),ipx) = separx(1,1,ipx)
215              separy(k,ptsep(3,ipx),ipx) = separy(1,1,ipx)
216     *..Cancel the separatrix which is not used (number of points
217     *  equal to 1 means that the separatrix is not traced)
218              ptsep(4,ipx) = kkp
219              nptot(ptsep(4,ipx),ipx) = 1
220              separx(1,ptsep(4,ipx),ipx) = separx(1,1,ipx)
221              separy(1,ptsep(4,ipx),ipx) = separy(1,1,ipx)
222     c----------------------------------------------------------------------
223     *..Second case: TWO X-points
224           else if (npx .EQ. 2) then
225     *..If the X-points are connected...
226              IF (racord) then
227     c----------------------------------------------------------------------
228     cc<<<
229     c        write(0,*) '-- Connected double null'
230     cc>>>
231     *..Identification of each curve which does not go to a plate
232                 k0 = 0
233                 do 30 ipx=1, npx
234                    do 35 j=1, 4
235                       IF (indplq(j,ipx) .EQ. 0) then
236                          k0 = k0 + 1
237                          kk(k0) = j
238                       end if
239        35          CONTINUE
240        30       CONTINUE
241     *..   Index each curve not going to a plate (curve connecting the X-poin
242     *     from right:  3, from left:  4)
243                 do 40 ipx=1, npx
244                    wx(1)=separx(1,kk(2*(ipx-1)+1),ipx)
245                    wy(1)=separy(1,kk(2*(ipx-1)+1),ipx)
246                    wx(2)=separx(2,kk(2*(ipx-1)+1),ipx)
247                    wy(2)=separy(2,kk(2*(ipx-1)+1),ipx)
248                    wx(3)=separx(2,kk(2*(ipx-1)+2),ipx)
249                    wy(3)=separy(2,kk(2*(ipx-1)+2),ipx)
250                    wx(4)=wx(1)
251                    wy(4)=wy(1)
252                    if((pty(ipx)-ypto)*aire(wx,wy,4).le.zero) then
253                      ptsep(3,ipx) = kk(2*(ipx-1)+1)
254                      ptsep(4,ipx) = kk(2*(ipx-1)+2)
255                    else
256                      ptsep(3,ipx) = kk(2*(ipx-1)+2)
257                      ptsep(4,ipx) = kk(2*(ipx-1)+1)
258                    end if
259        40       CONTINUE
260     *..Connect the end of the separatrices to the other X-point
261                 ipx = 1
262                 do 50 ind=3, 4
263                    k = coinci(separx(1,ptsep(ind,ipx),ipx),
264          .                    separy(1,ptsep(ind,ipx),ipx),
265          .                    nptot(ptsep(ind,ipx),ipx),
266          .                    separx(2,ptsep(ind,ipx+1),ipx+1),
267          .                    separy(2,ptsep(ind,ipx+1),ipx+1)) + 1
268                    nptot(ptsep(ind,ipx),ipx) = k
269                    separx(k,ptsep(ind,ipx),ipx) = separx(1,1,ipx+1)
270                    separy(k,ptsep(ind,ipx),ipx) = separy(1,1,ipx+1)
271        50       CONTINUE
272     *..Cancel the separatrices which are not used (number of points
273     *  equal to 1 means that the separatrix is not traced)
274                 ipx = 2
275                 nptot(ptsep(3,ipx),ipx) = 1
276                 separx(1,ptsep(3,ipx),ipx) = separx(1,1,ipx)
277                 separy(1,ptsep(3,ipx),ipx) = separy(1,1,ipx)
278                 nptot(ptsep(4,ipx),ipx) = 1
279                 separx(1,ptsep(4,ipx),ipx) = separx(1,1,ipx)
280                 separy(1,ptsep(4,ipx),ipx) = separy(1,1,ipx)
281     c----------------------------------------------------------------------
282     *..If the X-points are not connected...
283              else if (.NOT.(racord)) then
284     c----------------------------------------------------------------------
285     cc<<<
286     c        write(0,*) '-- Disconnected double null'
287     cc>>>
288     *..Determine the X-point which is inner to the other.
289     *  Inner X-point is that which has the value of the psi closer to
290     *  that of psi on the axis
291                 IF (abs(fctpx(1)-fctpx(npx+1)) .lt.
292          .          abs(fctpx(2)-fctpx(npx+1))) then
293                    ptxint = 1
294                    ipx = ptxint
295                 else if (abs(fctpx(1)-fctpx(npx+1)) .gt.
296          .          abs(fctpx(2)-fctpx(npx+1))) then
297                    ptxint = 2
298                    ipx = ptxint
299                 else
300                    WRITE(0,*) 'Equal psi values at both X-points for the ',
301          ,                                       'disconnected double null'
302     c  100          FORMAT('Double nul deconnecte mais valeur de fonction',
303     c     .                ' egale'/'pour chaque point X. PROBLEME !!')
304                    call pltend
305                    STOP
306                 end if
307     *..Initialisation
308                 k0 = 0
309     *..Identify the separatrices which do not go to the targets
310                 do 60 j=1, 4
311                    IF (indplq(j,ipx) .EQ. 0) then
312                       k0 = k0 + 1
313                       kk(k0) = j
314                    end if
315        60       CONTINUE
316     cc<<<
317     c        write(0,*) '** after 60 continue. ptxint = ',ptxint
318     c        write(0,*) 'ipx,k0,kk : ',ipx,k0,' ',kk
319     cc>>>
320     *..Index the inner separatrices
321                 wx(1)=separx(1,kk(1),ipx)
322                 wy(1)=separy(1,kk(1),ipx)
323                 wx(2)=separx(2,kk(1),ipx)
324                 wy(2)=separy(2,kk(1),ipx)
325                 wx(3)=separx(2,kk(2),ipx)
326                 wy(3)=separy(2,kk(2),ipx)
327                 wx(4)=wx(1)
328                 wy(4)=wy(1)
329     cc<<<
330     c            write(0,'(17h ypto,pty,aire = ,1p,4e12.4)') ypto,pty,
331     c     ,                                                    aire(wx,wy,4)
332     c            write(0,'(5h wx: ,1p,4e12.4)') wx
333     c            write(0,'(5h wy: ,1p,4e12.4)') wy
334     cc>>>
335                 if((pty(ipx)-ypto)*aire(wx,wy,4).le.zero) then
336                   ptsep(3,ipx) = kk(1)
337                   kkp = kk(2)
338                 else
339                   ptsep(3,ipx) = kk(2)
340                   kkp = kk(1)
341                 end if
342     *..Parametrise the inner separatrix
343                 k = coinci(separx(1,ptsep(3,ipx),ipx),separy(1,ptsep(3,ipx),
344          .              ipx),nptot(ptsep(3,ipx),ipx),separx(2,kkp,ipx),
345          .              separy(2,kkp,ipx)) + 1
346                 nptot(ptsep(3,ipx),ipx) = k
347                 separx(k,ptsep(3,ipx),ipx) = separx(1,1,ipx)
348                 separy(k,ptsep(3,ipx),ipx) = separy(1,1,ipx)
349     *..Cancel the separatrix which is not used (number of points
350     *  equal to 1 means that the separatrix is not traced)
351                 ptsep(4,ipx) = kkp
352                 nptot(ptsep(4,ipx),ipx) = 1
353                 separx(1,ptsep(4,ipx),ipx) = separx(1,1,ipx)
354                 separy(1,ptsep(4,ipx),ipx) = separy(1,1,ipx)
355     
356     *..Arrange the separatrices which surround the inner one
357     
358     *..Consider the outer X-point
359     
360                 ipx = MOD(ipx,2) + 1
361     
362     c*** Find the targets not connected to the inner X-point
363     
364               i1=0
365               i2=0
366               ipi=mod(ipx,2)+1
367               do i=1,nbdef
368                 k=inddef(i)
369                 do j=1,4
370                   if(k.eq.indplq(j,ipi)) k=0
371                 end do
372                 if(k.ne.0) then
373                   if(i1.eq.0) then
374                     i1=k
375                   else if(i2.eq.0) then
376                     i2=k
377                   else
378                     write(0,*) 'Error detected in argsep: more than two ',
379          ,                  'targets not intersecting the inner separatrix'
380                     call pltend
381                     STOP
382                   end if
383                 end if
384               end do
385               if(i2.eq.0) then
386                 write(0,*) 'Error detected in argsep: less than two (',i1,
387          ,               ' ) targets not intersecting the inner separatrix'
388                 call pltend
389                 STOP
390               end if
391     
392               j1=0
393               j2=0
394               do i=1,4
395                 if(i1.eq.indplq(i,ipx)) j1=i
396                 if(i2.eq.indplq(i,ipx)) j2=i
397               end do
398               if(j1.eq.0 .or. j2.eq.0) then
399                 write(0,*) 'Error detected in argsep: primary targets for',
400          ,                                ' the outer separatrix not found'
401                 write(0,*) 'i1,i2,inddef: ',i1,i2,' ',inddef
402                 write(0,*) 'indplq :',indplq
403                 call pltend
404                 STOP
405               end if
406     
407               k1=0
408               k2=0
409               do i=1,4
410                 if(i.ne.j1 .and. i.ne.j2) then
411                   if(k1.eq.0) then
412                     k1=i
413                   else
414                     k2=i
415                   end if
416                 end if
417               end do
418               l1=0
419               l2=0
420               do i=1,4
421                 k=inddef(i)
422                 if(k.eq.indplq(k1,ipx)) then
423                   if(l1.eq.0) then
424                     l1=k1
425                   else
426                     l2=k1
427                   end if
428                 end if
429                 if(k.eq.indplq(k2,ipx)) then
430                   if(l1.eq.0) then
431                     l1=k2
432                   else
433                     l2=k2
434                   end if
435                 end if
436               end do
437     cc<<<
438     c          write(0,*) 'i1,i2, j1,j2, k1,k2, l1,l2: ',
439     c     ,                              i1,i2,' ',j1,j2,' ',k1,k2,' ',l1,l2
440     cc>>>
441     c
442     c*** Index the curves which skirt the inner separatrix
443     c*** (each of them must intersect a target)
444     
445               ptsep(1,ipx) = j1
446               ptsep(2,ipx) = j2
447     
448               ptsep(3,ipx) = l1
449               ptsep(4,ipx) = l2
450     cank
451     c----------------------------------------------------------------------
452              end if
453           end if
454     cc<<<
455     c      write(0,*) 'Leaving argsep. ptsep : ',ptsep
456     cc>>>
457           RETURN
458     c======================================================================
459           END
