1           subroutine listru(iunit,nstr,npstru,nomstr,xstruc,ystruc,npstmx,
2          .  nstrmx)
3     c
4     c  version : 05.04.97 16:19
5     c
6     c======================================================================
7           implicit none
8     c  read data concerning structures
9     c
10     c  arguments
11     
12           integer nstrmx,npstmx,nstr,npstru(nstrmx),iunit
13           real*8 xstruc(npstmx,nstrmx),ystruc(npstmx,nstrmx)
14           character nomstr(nstrmx)*80
15     c
16     c  iunit: unit of file containing structure data
17     c  nstr: number of structures
18     c  npstru(i): number of points in structure i
19     c             a positive (negative) number indicates a closed (open)
20     c             structure. When this number is zero, the code counts
21     c             the number of points and the structure must be closed.
22     c             When -1, the code counts the number of points and the
23     c             structure is understood to be closed.
24     c  nomstr(i): name of structure i
25     c  xstruc(j,i): j-th x coordinate of structure i
26     c  ystruc(j,i): j-th y coordinate of structure i
27     c  nstrmx: maximum number of structures allowed
28     c  npstmx: maximum number of coordinates per structure
29     c
30     c  local variables
31           integer i,ifail
32           character line*80
33     c
34     c  procedures
35           external entete
36     c======================================================================
37     c  calculation
38     c
39           ifail=1
40           call entete(iunit,'$structures',ifail)
41           if(ifail.eq.1) then
42             write(6,*)'unable to find structure header in file: stop'
43             stop
44           endif
45      1    continue
46           read(iunit,100,end=99)line
47     100   format(a)
48           if(index(line,'$fin')+index(line,'$end').eq.0) then
49             nstr=nstr+1
50             nomstr(nstr)=line
51     c       call rdfrin(11,line,npstru(nstr),ifail)
52             read(iunit,*)npstru(nstr)
53             if(abs(npstru(nstr)).le.1) then
54               i=0
55      3        continue
56               i=i+1
57               read(iunit,*,err=5)xstruc(i,nstr),ystruc(i,nstr)
58               go to 3
59      5        continue
60               if(npstru(nstr).eq.-1) then
61                 npstru(nstr)=-(i-1)
62               else
63                 npstru(nstr)=i-1
64               endif
65               backspace iunit
66             else
67               do i=1,abs(npstru(nstr))
68                 read(iunit,*,end=98)xstruc(i,nstr),ystruc(i,nstr)
69               enddo
70             endif
71     c
72     c  structure names containing $n are suppressed
73             if(index(nomstr(nstr),'$n').gt.0) nstr=nstr-1
74     c
75             go to 1
76           endif
77      99   continue
78     c
79     c  close input unit
80           close(unit=iunit)
81           return
82     c
83      98   continue
84           write(6,*)'error in structure file'
85           stop
86           end
