2     ************************************************************************
3           LOGICAL FUNCTION milieu(f1, f2, niv)
4     ************************************************************************
5           IMPLICIT NONE
6     
7     *    Cette fonction verifie si la valeur d'une fonction niv est presente
8     *  entre deux valeurs de fonction: f1 et f2.
9     
10     *  arguments
11           REAL*8 f1, f2, niv
12     
13     c=========================
14     c.. f1,f2: valeurs de la fonction aux points 1 et 2.
15     c.. niv: fonction pour laquelle on doit verifier si elle est entre
16     c         f1 et f2.
17     c=========================
18     
19           IF ((f1.LE.niv .AND. f2.GE.niv) .OR.
20          .      (f1.GE.niv .AND. f2.LE.niv)) THEN
21              milieu = .TRUE.
22           ELSE
23              milieu = .FALSE.
24           ENDIF
25     
26           RETURN
27           END
