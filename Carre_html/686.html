1     ************************************************************************
2           INTEGER FUNCTION indsgm(xst,yst,n,xx,yy)
3     ************************************************************************
4           IMPLICIT NONE
5     
6     *..  Cette fonction trouve le segment d'une structure auquel est associe
7     *  un point de contact (xx,yy). Les erreurs numeriques font que ce point
8     *  n'est pas exactement sur le segment. Pour trouver le segment, on
9     *  effectue (pour chaque segment) un produit scalaire entre les 2
10     *  vecteurs suivants: 1) du point jusqu'a une des extremites du segment;
11     *  2) du point vers l'autre extremite. Ainsi l'angle qu'on mesurera pour le
12     *  segment de contact sera pres de 180 degrees et pour chaque autre on
13     *  aura un angle moindre. Le segment de contact sera celui qui
14     *  presentera l'angle le plus grand.
15     
16     *  arguments
17           INTEGER n
18           REAL*8 xst(n), yst(n), xx, yy
19     
20     *  variables locales
21           INTEGER i
22           REAL*8 mumin, mu, ax, ay, bx, by,eps,dist
23           PARAMETER (eps=1.E-6)
24     
25     *  procedures
26           INTRINSIC sqrt
27     
28     c=========================
29     c.. xst,yst: tableaux des coordonnees des points de la structure.
30     c.. n  : nombre de points de la structure.
31     c.. xx,yy: point de contact sur un segment de structure.
32     c.. ax,ay: vecteur allant du point de contact au debut d'un segment.
33     c.. bx,by: vecteur allant du point de contact au debut du segment
34     c          suivant.
35     c.. mu, mumin: cosinus de l'angle et de l'angle maximum
36     c=========================
37     
38     *..Initialisation.
39     
40           mumin = 1.
41     
42     *..Test pour savoir si le point x,y est egal au premier point parame-
43     *  trise.
44     
45           ax = xst(1) - xx
46           ay = yst(1) - yy
47     
48           dist = SQRT(ax*ax + ay*ay)
49     
50           IF (dist .LT. eps) THEN
51     
52              indsgm = 1
53              RETURN
54     
55           ENDIF
56     
57     *..Boucle sur tous les segments
58     
59           DO 10 i=1, n-1
60     
61     *..Def. des 2 vecteurs a et b.
62     
63              ax = xst(i) - xx
64              ay = yst(i) - yy
65              bx = xst(i+1) - xx
66              by = yst(i+1) - yy
67     
68     *..Test pour voir si le point x,y tombe sur le point parametrise suivant
69     
70              dist = SQRT(bx*bx + by*by)
71              IF (dist .LT. eps) THEN
72                 IF (i .NE. n-1) THEN
73                    indsgm = i + 1
74                    RETURN
75                 ELSE
76                    indsgm = i
77                    RETURN
78                 ENDIF
79              ELSE
80     
81     *..On trouve le cosinus de l'angle entre les 2 par le produit scalaire.
82     
83                mu = ((ax*bx) + (ay*by))
84          .                  /SQRT(((ax**2)+(ay**2)) * ((bx**2)+(by**2)))
85     
86     *..L'indice de segment correspond au cosinus le plus petit
87     
88                IF (mu .lt. mumin) THEN
89                   indsgm = i
90                   mumin = mu
91                ENDIF
92     
93           endif
94     
95        10 CONTINUE
96     
97           RETURN
98           END
