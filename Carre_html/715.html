1           subroutine insert(indstr,inddef,nbdef,ipx)
2     c
3     c  version: 17.11.99 23:56
4     c
5     c=======================================================================
6           implicit none
7     #include <CARREDIM.F>
8     
9     *.. This routine creates a vector of indices of the divertor targets
10     
11     *  arguments
12           integer indstr,nbdef,inddef(nbdmx),indxpt(nbdmx),ipx,npx
13           real*8 xst(nbdmx),yst(nbdmx),xtt(nbdmx),ytt(nbdmx)
14     
15     *  variables locales
16           integer npxmx4
17           parameter (npxmx4=4*npxmx)
18           integer i,j,k,l,m,iipx(npxmx4),
19          ,  ii(strumx),jj(nbdmx),kk(npxmx),ll(nbdmx),mm(npxmx)
20           
21           data iipx /npxmx4*0/
22           save iipx
23     
24     c=======================================================================
25     c.. indstr: structure index
26     c.. inddef: table of target indices
27     c.. nbdef:  number of targets
28     c=======================================================================
29     
30     c*** If the structure index is non-zero, it points to a structure 
31     c*** intersecting the separatrix branch. Store all such cases for a 
32     c*** while - we will select the primary strikepoints after all x-points
33     c*** have been treated.
34     cank-19990727: symptomatic treatment
35     c*** ... However, multiple intersections of separatrix branches starting
36     c*** at the same x-point with the same target should be screened
37     
38           write(0,*) 'insert: indstr,nbdef,ipx',indstr,nbdef,ipx !###
39           if (indstr .gt. 0)  then !{
40             do i=1, nbdef !{
41               if (inddef(i).eq.indstr .and. ipx.eq.iipx(i)) return
42             end do !}
43             iipx(i)=ipx
44             nbdef = nbdef + 1
45             inddef(nbdef) = indstr
46           end if !}
47           return
48     
49     c=======================================================================
50     
51           entry trgarng(inddef,indxpt,xst,yst,nbdef,npx)
52     c=======================================================================
53     c*** Here, we determine the "primary" strikepoints - i.e., those 
54     c*** connected to the innermost x-point for each target, and check for  
55     c*** the consistency of the strike-points and targets. 
56     c*** Each x-point must be connected to two "primary" strike-points.
57     c***
58     c*** Input:
59     c***  inddef  - list of targets (pieces of structure having at least one
60     c***            intersection with a separatrix branch (strike-point)
61     c***  indxpt  - list of x-points connected to the corresponding targets
62     c***  xst,yst - strike-point co-ordinates
63     c***  nbdef   - number of strike-points
64     c***  npx     - number of x-points
65     c***
66     c*** Output:
67     c***  the same arrays corrected so that only the "primary" strike-points
68     c***  are included and counted
69     c***
70     c*** Internal arrays:
71     c***  ii  - strike-point counters for each target
72     c***  jj  - index of targets being "private" - i.e., having only one
73     c***        strike-point
74     c***  kk  - index of x-points connected to two "private" targets
75     c***  ll  - index of x-points corresponding to the "private" targets
76     c***  mm  - number of "private" targets for each x-point
77     c=======================================================================
78     c<<<
79           write(0,*) 'Entering trgarng: nbdef=',nbdef
80           if(nbdef.gt.0) then !{
81             write(0,'(1x,a8,i7,9i10)') 'inddef: ',(inddef(i),i=1,nbdef)
82             write(0,'(1x,a8,i7,9i10)') 'indxpt: ',(indxpt(i),i=1,nbdef)
83             write(0,'(1x,a4,4x,1p,10e10.3)') 'xst:',(xst(i),i=1,nbdef)
84             write(0,'(1x,a4,4x,1p,10e10.3)') 'yst:',(yst(i),i=1,nbdef)
85           end if !}
86     c>>>
87     c*** Select the primary x-points and correct the number of targets
88     c*** First, check which targets are intersected only once
89     
90           do i=1,nbdmx !{
91             ii(i)=0
92           end do !}
93           do i=1,nbdef !{
94             ii(inddef(i))=ii(inddef(i))+1
95           end do !}
96           do i=1,npx !{
97             mm(i)=0
98           end do !}
99     
100     c*** k is the number of "private" targets
101           k=0
102           do j=1,nbdef !{
103             if(ii(inddef(j)).eq.1) then !{
104               k=k+1
105               jj(k)=j
106               ll(k)=indxpt(j)
107               mm(indxpt(j))=mm(indxpt(j))+1
108             else if(ii(inddef(j)).gt.2) then !}{
109               write(0,*) 'Error found in INSERT (entry TRGARNG): too many ',
110          ,      'intersections with separatrix branches for one target - ',
111          ,      ii(inddef(j)), 'for structure ',inddef(j)
112               stop
113             end if !}
114           end do !}
115     c<<<
116           write(0,*) 'Single intersections : ',k
117           if(k.gt.0) then !{
118             write(0,*) 'jj : ',(jj(i),i=1,k)
119             write(0,*) 'll : ',(ll(i),i=1,k)
120           end if !}
121     c>>>
122     
123     c*** Check the count of private targets for each x-point
124     c*** m is the number of x-points having two "private" targets
125           m=0
126           do j=1,npx !{
127             if(mm(j).eq.2) then !{
128               m=m+1
129               kk(m)=j
130             else if(mm(j).gt.2) then !}{
131               write(0,*) 'Error found in INSERT (entry TRGARNG): too many ',
132          ,      '"private" targets for one x-point: ',j
133               stop '==> Check the targets related to the inner x-point'
134             end if !}
135           end do !}
136     c<<<
137           write(0,*) 'Complete x-points: ',m
138           if(m.gt.0) then !{
139             write(0,*) 'kk : ',(kk(i),i=1,m)
140           end if !}
141     c>>> 
142     c*** Now look for targets intersected twice
143           do j=1,nbdef !{
144             if(ii(inddef(j)).eq.2) then !{
145               l=0
146               do i=1,m !{
147                 if(kk(i).eq.indxpt(j)) l=l+1
148               end do !}
149               if(l.eq.0) then !{
150                 k=k+1
151                 jj(k)=j
152               end if !}
153             end if !}
154           end do !}
155     c<<<
156           write(0,*) 'Before re-arrangement. k = ',k
157           if(k.gt.0) write(0,*) 'jj : ',(jj(i),i=1,k)
158     c>>>
159     
160           do i=1,k !{
161             j=jj(i)
162             kk(i)=inddef(j)
163             mm(i)=indxpt(j)
164             xtt(i)=xst(j)
165             ytt(i)=yst(j)
166           end do !}
167           do i=1,k !{
168             inddef(i)=kk(i)
169             indxpt(i)=mm(i)
170             xst(i)=xtt(i)
171             yst(i)=ytt(i)
172           end do !}
173           nbdef=k
174     c<<<
175           write(0,*) 'After re-arrangement: ',nbdef
176           if(nbdef.gt.0) then !{
177             write(0,'(1x,a8,i7,9i10)') 'inddef: ',(inddef(i),i=1,nbdef)
178             write(0,'(1x,a4,4x,1p,10e10.3)') 'xst:',(xst(i),i=1,nbdef)
179             write(0,'(1x,a4,4x,1p,10e10.3)') 'yst:',(yst(i),i=1,nbdef)
180           end if !}
181     c>>> 
182     
183     c*** Check the consistency of the "primary targets" once more
184           
185           do i=1,npx !{
186             mm(i)=0
187           end do !}
188           do i=1,nbdef !{
189             mm(indxpt(i))=mm(indxpt(i))+1
190           end do !} 
191           m=0
192           do i=1,npx !{
193             if(mm(i).ne.2) then !{
194               m=m+1
195               write(0,*) 'Error found in INSERT (entry TRGARNG): wrong ',
196          ,      'number of "private" targets. ipx,n = ',i,mm(i)
197             end if !}
198           end do !}
199           if(m.gt.0) stop '==> check the target specification' 
200     c<<<
201           write(0,*) 'Leaving trgarng: ',nbdef
202     c>>> 
203     c=======================================================================
204           end
