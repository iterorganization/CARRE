2     ************************************************************************
3           subroutine intrsc(x1,y1,x2,y2,vx,vy,n,x,y,iint,nint,sens)
4     ************************************************************************
5           implicit none
6     
7     *..  cette routine determine le point d'intersection entre un segment
8     c    1-2 et une courbe parametrisee de n points de coordonnees vx vy.
9     
10     *  arguments
11           integer n,iint(2),nint,sens
12           real*8 x1,y1,x2,y2,vx(n),vy(n),x(2),y(2)
13     
14     *  variables locales
15           integer i,i1,i2,ii
16           real*8 determ, mult1,mult2,zero,un,multmn,zx,zy
17           parameter(zero=0.,un=1.)
18     c  procedures
19           integer tourne
20           external tourne
21     
22     c=========================
23     c.. vx,vy: tableaux des coordonees des points de la courbe.
24     c.. n  : nombre de point de la courbe.
25     c.. x1,y1,x2,y2: coordonnees des 2 points formant le segment.
26     c   x,y: tableaux de coordonnees des points d'intersection
27     c   iint: tableau des indices de segment de la courbe v pour lesquels il
28     c         y a intersection
29     c   nint: nombre d'intersections
30     c   sens: sens du balayage
31     c.. determ: determinant de la matrice des deux equations.
32     c.. mult1: facteur multiplicatif du segment 1-2
33     c.. mult2: facteur multiplicatif du segment de la courbe v.
34     c=========================
35     *..boucle sur tous les segments de la courbe.
36           multmn=2.
37           nint=0
38           i2=1
39           do 10 i=1, n-1
40              i1=i2
41              i2=tourne(i1,sens,n)
42     *..calcul du determinant de la matrice.
43              determ = (-(x2 - x1)) * (vy(i2) - vy(i1)) +
44          .               (y2 - y1) * (vx(i2) - vx(i1))
45     *..il y a solution seulement si le determinant n'est pas nul
46              if (determ .ne. 0.) then
47     *..facteur multiplicatif du segment avec la methode de Cramer.
48                 mult1 = (-(vx(i1)-x1) * (vy(i2)-vy(i1)) +
49          .                (vy(i1)-y1) * (vx(i2)-vx(i1)))/determ
50     *..pour avoir intersection, il faut que mult1 soit entre 0 et 1
51                 if ((mult1.ge.zero).and.(mult1.le.un)) then
52     *..fact. mult. du segment de la courbe v.
53                    mult2= ((x2-x1) * (vy(i1)-y1) -
54          .                 (y2-y1) * (vx(i1)-x1))/determ
55     *..intersection si mult2 entre 0 et 1
56                    if ((mult2.ge.zero).and.(mult2.le.un)) then
57     *..s'il y a intersection alors le segment traverse la courbe.
58                       nint=nint+1
59                       iint(nint)=i1
60                       x(nint)=x1+mult1*(x2-x1)
61                       y(nint)=y1+mult1*(y2-y1)
62                       if(mult1.lt.multmn) then
63                         multmn=mult1
64                         ii=iint(1)
65                         zx=x(1)
66                         zy=y(1)
67                         iint(1)=iint(nint)
68                         x(1)=x(nint)
69                         y(1)=y(nint)
70                         iint(nint)=ii
71                         x(nint)=zx
72                         y(nint)=zy
73                      endif
74                    endif
75                 endif
76              endif
77        10 continue
78           return
79           end
