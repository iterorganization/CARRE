3           SUBROUTINE NUNIFO(npoint,l,d1,dn,interv,dmin,dmax)
4     ************************************************************************
5           IMPLICIT NONE
6     
7     *..  Cette sous-routine repartit les points de facon non uniforme sur
8     *  une courbe en faisant passer une parabole par le premier et le
9     *  dernier intervalle donnee. Les intervalles intermediaires sont
10     *  determinees par la parabole.
11     
12     *  arguments
13           INTEGER npoint
14           REAL*8 l,d1,dn,interv(npoint-1),dmin,dmax
15     
16     *  variables locales
17           INTEGER i,n
18           REAL*8 denom,aa,bb,cc
19     
20     *  procedures
21           INTRINSIC MIN,MAX
22     
23     c=========================
24     c.. npoint: nombre de points de la courbe.
25     c.. n  : nombre d'intervalles.
26     c.. l  : longueur en metres de la courbe, ou difference en psi entre le
27     c         premier et le dernier point.
28     c.. d1 : grandeur du premier intervalle
29     c.. dn : grandeur du dernier intervalle
30     c.. dmin: intervalle minimum.
31     c.. dmax: intervalle maximum.
32     c.. interv: tableau contenant les intervalles de chaque point.
33     c.. aa,bb,cc: coefficients de l'equation: di = aa + bb*i + cc*i*i
34     c.. denom: denominateur.
35     c=========================
36     
37     *..Definition du nombre d'intervalles.
38     
39           n = npoint - 1
40     
41     *..Definition des coefficients.
42     
43           if(n.eq.1) then
44     c       aa=0.5*(d1+dn)
45             aa=l
46             bb=0.
47             cc=0.
48           elseif(n.eq.2) then
49     c       aa=2.*d1-dn
50     c       bb=dn-d1
51             bb=l/(3.*(1.-d1/dn)-2.*(1.-2.*d1/dn))
52             aa=-(1.-2.*d1/dn)*bb
53             bb=(1.-d1/dn)*bb
54             cc=0.
55           else
56           denom = -((n-1)**2)*(n-2)/6.
57             aa = (d1*(-n)*(n+1)*(n-1)/6. - dn*(n+1)*(n-1)/3. + l*(n-1))
58          .    /denom
59             bb = (d1*(-(n+1)*(2*n+1)/6. + n*n) + dn*((n+1)*(2*n+1)/6. - 1)
60          .        + l*(1-n*n)/n)/denom
61             cc = (n-1)*(-d1/2. - dn/2. + l/n)/denom
62           endif
63     
64     *..Calcul de l'intervalle
65     
66           dmin = aa + bb + cc
67           dmax = aa + bb + cc
68     
69           DO 10 i=1, n
70              interv(i) = aa + bb*i + cc*i*i
71              dmin = MIN(interv(i),dmin)
72              dmax = MAX(interv(i),dmax)
73        10 CONTINUE
74     
75           RETURN
76           END
