2           SUBROUTINE WARNINGS(separx,separy,nivx,nivy,
3          &  distnv,xpto,ypto)
4     c****************************************************************
5     C written by Heimo Buerbaumer December 1998
6     C Calculation of the distance separatrix-outer border limit
7     C in the SOL-region and output on the file 'warnings.dat'
8     C for the use in the automatic grid generator AGG
9     c****************************************************************
10     #include <CARREDIM.F>
11           REAL*8 nivx(npnimx,nivmx),nivy(npnimx,nivmx),nxu,nyu,nxo,nyo
12           REAL*8 separx(npnimx,4,npxmx),separy(npnimx,4,npxmx),xpto,ypto,
13          & distnv
14           
15           open (33,file='warnings.dat')
16           rewind 33
17           
18     
19          
20           adzu=100.
21           adzo=100.
22           bdzu=100.
23           bdzo=100.
24           do i=1,npnimx
25              do j=1,4
26                 if (separx(i,j,1).GT.xpto) then
27                    dz=ypto-separy(i,j,1)
28                    adz=ABS(dz)
29                    if ((dz.GE.0.).AND.(adz.LT.adzu)) then
30                    adzu=adz
31                    sxu=separx(i,j,1)
32                    syu=separy(i,j,1)
33                    endif
34                    if ((dz.LT.0.).AND.(adz.LT.adzo)) then
35                    adzo=adz
36                    sxo=separx(i,j,1)
37                    syo=separy(i,j,1)
38                    endif
39           
40                 endif
41               enddo
42           if (nivx(i,1).GT.xpto) then
43              dz=ypto-nivy(i,1)
44              adz=ABS(dz)
45              if ((dz.GE.0.).AND.(adz.LT.bdzu)) then
46                 bdzu=adz
47                 nxu=nivx(i,1)
48                 nyu=nivy(i,1)
49              endif
50              if ((dz.LT.0.).AND.(adz.LT.bdzo)) then
51                 bdzo=adz
52                 nxo=nivx(i,1)
53                 nyo=nivy(i,1)
54              endif
55     C      write (33,*) i,bdzu,nxu,nyu,bdzo,nxo,nyo
56           endif
57           enddo
58     
59           z=ABS(syo-syu)
60           if (z.LT.0.000001) then
61              rs=syo
62           else
63              rs=(ypto-syo)*(sxu-sxo)/(syu-syo)+sxo
64           endif
65           z=ABS(nxo-nxu)
66           if (z.LT.0.000001) then
67              rn=nyo
68           else
69              rn=(ypto-nyo)*(nxu-nxo)/(nyu-nyo)+nxo
70           endif
71           distsn=100.*(rn-rs)
72           distnv=100.*distnv
73            write (33,120) distnv,distsn
74            write (33,*) rs,rn,syo,syu,nxo,nyo,xpto,ypto
75            
76           
77           close(33)
78      120   FORMAT(1X,F8.2,1X,F8.2)
79           RETURN
80           END
