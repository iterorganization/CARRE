2     ************************************************************************
3           SUBROUTINE UNTANG(xn,yn,npni,x,y,ux,uy,d,period)
4     ************************************************************************
5           IMPLICIT NONE
6     
7     *..  Cette sous-routine calcule le vecteur unite tangent a la courbe
8     *  parametrisee au point x,y. La direction de ce vecteur est obtenue par
9     *  interpolation lineaire avec les points parametrises adjacents.
10     
11     * arguments
12           INTEGER npni
13           REAL*8 xn(npni), yn(npni), x, y, ux, uy,d,period
14     
15     *  variables en common
16     
17     #include <COMLAN.F>
18     
19     *  variables locales
20           INTEGER indp1, ind, jnd, jndp, ipp
21           REAL*8 unx1,uny1,unx2,uny2,z,w1
22     
23     *  procedures
24           INTEGER indcrb
25           EXTERNAL indcrb
26           INTRINSIC SQRT
27     
28     c=========================
29     c.. xn,yn: coordonnees des points de la courbe parametrisee.
30     c.. npni: nombre de points dans la courbe.
31     c.. x,y: coordonnees du point sur la courbe.
32     c.. ind : indice du point parametrise precedent le point de la courbe
33     c.. ux,uy: vecteur unite tangent a la courbe au point x,y.
34     c.. unx1,uny1: vecteur unite dependant de l'indice precedant.
35     c.. unx2,uny2: vecteur unite dependant de l'indice suivant.
36     c=========================
37     
38     *..Calculs.
39     
40     *..Initialisation.
41     
42           ind = indcrb(xn,yn,npni,x,y,d)
43     
44           indp1 = ind + 1
45     c
46     c  indice du point le plus proche
47           if((xn(ind)-x)**2+(yn(ind)-y)**2 .lt.
48          .  (xn(indp1)-x)**2+(yn(indp1)-y)**2) then
49             ipp=ind
50           else
51             ipp=indp1
52           endif
53     c
54     c  calcul du 1-er vecteur unite
55           unx1=xn(indp1)-xn(ind)
56           uny1=yn(indp1)-yn(ind)
57           z=sqrt(unx1*unx1+uny1*uny1)
58           unx1=unx1/z
59           uny1=uny1/z
60     c
61     c  pas d'interpolation aux extremites
62           if(ipp.eq.1 .or. ipp.eq.npni) then
63             if(period.eq.0.) then
64               ux=unx1
65               uy=uny1
66               return
67             else
68               if(ipp.eq.1) then
69                 jndp=npni
70                 jnd=npni-1
71               else
72                 jndp=2
73                 jnd=1
74               endif
75             endif
76           elseif(ipp.eq.ind) then
77             jndp=ind
78             jnd=ind-1
79           elseif(ipp.eq.indp1) then
80             jndp=ind+2
81             jnd=indp1
82           else
83             if(sellan(1:8).eq.'francais') then
84               write(6,*)'Cas impossible dans untang: arret force'
85             elseif(sellan(1:7).eq.'english') then
86               write(6,*)'Impossible case in untang: forced stop'
87             endif
88             call pltend
89             stop
90           endif
91     c
92     c  calcul du 2-e vecteur unite
93           unx2=xn(jndp)-xn(jnd)
94           uny2=yn(jndp)-yn(jnd)
95           z=sqrt(unx2*unx2+uny2*uny2)
96           unx2=unx2/z
97           uny2=uny2/z
98     
99     c
100     c  ponderation du 1-er vecteur et moyenne
101           w1=0.5+sqrt(((xn(ipp)-x)**2+(yn(ipp)-y)**2)/
102          .  ((xn(indp1)-xn(ind))**2+(yn(indp1)-xn(ind))**2))
103           ux=w1*unx1+(1.-w1)*unx2
104           uy=w1*uny1+(1.-w1)*uny2
105           z=sqrt(ux*ux+uy*uy)
106           ux=ux/z
107           uy=uy/z
108           return
109           end
