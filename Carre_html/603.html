2     ************************************************************************
3           SUBROUTINE GRAD0(nxmax,nymax,nx,ny,x,y,gradmx,
4          .          pointx,pointy,ii,jj,npxtot,a00,a10,a01,a11)
5     ************************************************************************
6           IMPLICIT NONE
7     
8     *..  Cette sous-routine verifie pour chaque carre si il y a intersection
9     *  entre les lignes de niveau psidx=0 et psidy=0 et si oui elle garde
10     *  les coordonnees en memoire.
11     
12     *  arguments
13           INTEGER nxmax,nymax,nx,ny,gradmx,ii(gradmx),jj(gradmx),npxtot
14           REAL*8 pointx(gradmx), pointy(gradmx),x(nx),y(ny)
15          .  ,a00(nxmax,nymax,3),a10(nxmax,nymax,3),a01(nxmax,nymax,3)
16          .  ,a11(nxmax,nymax,3)
17     
18     *  variables locales
19           INTEGER i, j
20           REAL*8 aa,bb,cc,x1,x2,y1,y2
21     
22     *  procedures
23           INTRINSIC SQRT
24     
25     c=========================
26     c.. npxtot: nombre de points ou le gradient est nul.
27     c.. pointx,pointy: tableaux des coordonnees en x et y des points ou le
28     c                  gradient est nul.
29     c.. ii,jj: tableaux des indices en x et en y des carres ou le gradient
30     c          est nul.
31     c.. i,j: indices
32     c.. ax,bx,cx,dx(i,j,2): coefficients de l'equation:
33     c                           ax + bx*X + cx*Y + dx*X*Y = psidx
34     c.. ay,by,cy,dy(i,j,3): coefficients de l'equation:
35     c                           ay + by*X + cy*Y + dy*X*Y = psidy
36     c.. aa,bb,cc: coefficients de l'equation des racines:
37     c             ( -bb +- SQRT(bb**2 - 4*aa*cc)) / 2*aa
38     c.. y1,y2: points ou la fonction s'annule selon y.
39     c.. x1,x2: points correspondants ou la fonction s'annule selon x.
40     c=========================
41     
42     *..balayage en x et en y.
43     
44           npxtot = 0
45           DO 10 j=1, ny-1
46              DO 20 i=1, nx-1
47     
48     *..recherche des racines.
49     
50                 aa = (a01(i,j,3)*a11(i,j,2) - a11(i,j,3)*a01(i,j,2))
51                 bb = (a00(i,j,3)*a11(i,j,2) - a10(i,j,3)*a01(i,j,2)
52          .           +a01(i,j,3)*a10(i,j,2) - a11(i,j,3)*a00(i,j,2))
53                 cc = (a00(i,j,3)*a10(i,j,2) - a10(i,j,3)*a00(i,j,2))
54     
55                 IF (4.*aa*cc .LT. bb*bb) THEN
56     
57                    IF (bb.GT.0. .AND. aa.NE.0.) THEN
58                       y1 = - 2.*cc/(SQRT(bb*bb - 4.*aa*cc) + bb)
59                       y2 = (-bb - SQRT(bb*bb - 4.*aa*cc))/(2.*aa)
60                    ELSE IF (aa.NE.0.) THEN
61                       y1 = 2.*cc/(SQRT(bb*bb - 4.*aa*cc) - bb)
62                       y2 = (-bb + SQRT(bb*bb - 4.*aa*cc))/(2.*aa)
63                    ELSE IF (bb.ne.0.) THEN
64                       y1 = -cc/bb
65                    ELSE
66                       y1 = 2.*y(1)-y(2)
67                    ENDIF
68     
69     *..Test pour savoir si le carre contient une de ces racines.
70     
71                    IF (y1.GE.y(j).AND.y1.LE.y(j+1)) THEN
72                       if(a10(i,j,2)+a11(i,j,2)*y1.ne.0.) then
73                         x1 = (-a00(i,j,2) - a01(i,j,2)*y1)/(a10(i,j,2)
74          .                   + a11(i,j,2)*y1)
75                       else
76                         x1 = 2.*x(1)-x(2)
77                       endif
78                       IF (x1.GE.x(i).AND.x1.LE.x(i+1)) THEN
79     
80     *..Sauvegarde de chaque point ou le gradient s'annule.
81     
82                          npxtot = npxtot + 1
83                          pointx(npxtot) = x1
84                          pointy(npxtot) = y1
85                          ii(npxtot) = i
86                          jj(npxtot) = j
87     
88                       ENDIF
89                    ENDIF
90     
91                    IF (y2.GE.y(j).AND.y2.LE.y(j+1).AND.aa.NE.0.) THEN
92                       if(a10(i,j,2)+a11(i,j,2)*y2.ne.0.) then
93                         x2 = (-a00(i,j,2)-(a01(i,j,2)*y2))/(a10(i,j,2)
94          .                   +(a11(i,j,2)*y2))
95                       else
96                         x2 = 2.*x(1)-x(2)
97                       endif
98                       IF (x2.GE.x(i).AND.x2.LE.x(i+1)) THEN
99     
100                          npxtot = npxtot + 1
101                          pointx(npxtot) = x2
102                          pointy(npxtot) = y2
103                          ii(npxtot) = i
104                          jj(npxtot) = j
105     
106                       ENDIF
107                    ENDIF
108                 ENDIF
109     
110        20    CONTINUE
111        10 CONTINUE
112     
113           RETURN
114           END
