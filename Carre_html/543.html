990              x0=separx(nptot(ptsep(1,ipx),ipx),ptsep(1,ipx),ipx)
991              y0=separy(nptot(ptsep(1,ipx),ipx),ptsep(1,ipx),ipx)
992     
993     *..Direction of the sweep.
994     
995              call trc_stk_in('frtier','*98')
996              sens = drctio(xstruc(1,inddef(idef)),ystruc(1,inddef(idef)),
997          .                 npstru(inddef(idef)),x0,y0,'gauche')
998              call trc_stk_out
999     
1000     *..Call the subroutine to find the limiting level line
1001     
1002              CALL MARCHE(x0,y0,inddef(idef),fraplq,sens,nivx(1,idniv)
1003          .      ,nivy(1,idniv),nivtot(idniv),nbcrb,xt,yt,nt,stp0,stpmin,
1004          .      nx,ny,x,y,psi,nstruc,npstru,xstruc,
1005          .      ystruc,a00,a10,a01,a11,indlim)
1006     
1007     *..Initialise the distances
1008     
1009              distnv(1,idniv) = 0.
1010              distnv(2,idniv) = 0.
1011     
1012     *..Calculate the absolute distance from the starting point to the
1013     *  first point of the limiting level line
1014     
1015              x2 = nivx(1,idniv)
1016              y2 = nivy(1,idniv)
1017     
1018              distnv(1,idniv) = plqdst(x0,y0,x2,y2,xstruc(1,inddef(idef)),
1019          .                      ystruc(1,inddef(idef)),npstru(inddef(idef))
1020          .                      ,'gauche')
1021     
1022     *..Calculate the psi difference between these two points.
1023     
1024              ii = ifind(x0,x,nx,1)
1025              jj = ifind(y0,y,ny,1)
1026     
1027              psi0 = a00(ii,jj,1) + a10(ii,jj,1)*x0 + a01(ii,jj,1)*y0 +
1028          .          a11(ii,jj,1)*x0*y0
1029     
1030              ii = ifind(x2,x,nx,1)
1031              jj = ifind(y2,y,ny,1)
1032     
1033              psi2 = a00(ii,jj,1) + a10(ii,jj,1)*x2 + a01(ii,jj,1)*y2 +
1034          .          a11(ii,jj,1)*x2*y2
1035     
1036              distnv(2,idniv) = psi2 - psi0
1037     
1038           ENDIF

1042     cc>>>
